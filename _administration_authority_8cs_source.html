93 public async ValueTask<AuthorityResponse<AdministrationResponse>> GetUpdateInformation(bool forceFresh, CancellationToken cancellationToken)
94 {
95 try
96 {
97 async Task<AdministrationResponse> CacheFactory()
98 {
99 Version? greatestVersion = null;
100 Uri? repoUrl = null;
101 var scopeCancellationToken = CancellationToken.None; // DCT: None available
102 try
103 {
104 var gitHubService = await gitHubServiceFactory.CreateService(scopeCancellationToken);
105 var repositoryUrlTask = gitHubService.GetUpdatesRepositoryUrl(scopeCancellationToken);
106 var releases = await gitHubService.GetTgsReleases(scopeCancellationToken);
107
108 foreach (var kvp in releases)
109 {
110 var version = kvp.Key;
111 var release = kvp.Value;
112 if (version.Major > 3 // Forward/backward compatible but not before TGS4
113 && (greatestVersion == null || version > greatestVersion))
114 greatestVersion = version;
115 }
116
117 repoUrl = await repositoryUrlTask;
118 }
119 catch (NotFoundException e)
120 {
121 Logger.LogWarning(e, "Not found exception while retrieving upstream repository info!");
122 }
123
124 return new AdministrationResponse
125 {
126 LatestVersion = greatestVersion,
127 TrackedRepositoryUrl = repoUrl,
128 GeneratedAt = DateTimeOffset.UtcNow,
129 };
130 }
131
132 var ttl = TimeSpan.FromMinutes(30);
133 Task<AdministrationResponse> task;
134 if (forceFresh || !cacheService.TryGetValue(ReadCacheKey, out var rawCacheObject))
135 {
136 using var entry = cacheService.CreateEntry(ReadCacheKey);
137 entry.AbsoluteExpirationRelativeToNow = ttl;
138 entry.Value = task = CacheFactory();
139 }
140 else
141 task = (Task<AdministrationResponse>)rawCacheObject!;
142
143 var result = await task.WaitAsync(cancellationToken);
144 return new AuthorityResponse<AdministrationResponse>(result);
145 }
146 catch (RateLimitExceededException e)
147 {
148 return RateLimit<AdministrationResponse>(e);

153 return new AuthorityResponse<AdministrationResponse>(

163 public async ValueTask<AuthorityResponse<ServerUpdateResponse>> TriggerServerVersionChange(Version targetVersion, bool uploadZip, CancellationToken cancellationToken)
164 {
165 var attemptingUpload = uploadZip == true;
166 if (attemptingUpload)
167 {
168 if (!AuthenticationContext.PermissionSet.AdministrationRights!.Value.HasFlag(AdministrationRights.UploadVersion))
169 return Forbid<ServerUpdateResponse>();
170 }
171 else if (!AuthenticationContext.PermissionSet.AdministrationRights!.Value.HasFlag(AdministrationRights.ChangeVersion))
172 return Forbid<ServerUpdateResponse>();
173
174 if (targetVersion.Major < 4)
175 return BadRequest<ServerUpdateResponse>(ErrorCode.CannotChangeServerSuite);
176
177 if (!serverControl.WatchdogPresent)
178 return new AuthorityResponse<ServerUpdateResponse>(
179 new ErrorMessageResponse(ErrorCode.MissingHostWatchdog),
180 HttpFailureResponse.UnprocessableEntity);
181
182 IFileUploadTicket? uploadTicket = attemptingUpload
183 ? fileTransferService.CreateUpload(FileUploadStreamKind.None)
184 : null;
185
186 ServerUpdateResult updateResult;
187 try
188 {
189 try
190 {
191 updateResult = await serverUpdateInitiator.InitiateUpdate(uploadTicket, targetVersion, cancellationToken);
192 }
193 catch
194 {
195 if (attemptingUpload)
196 await uploadTicket!.DisposeAsync();
197
198 throw;
199 }
200 }
201 catch (RateLimitExceededException ex)
202 {
203 return RateLimit<ServerUpdateResponse>(ex);
204 }
205 catch (ApiException e)
206 {
207 Logger.LogWarning(e, OctokitException);
208 return new AuthorityResponse<ServerUpdateResponse>(
209 new ErrorMessageResponse(ErrorCode.RemoteApiError)
210 {
211 AdditionalData = e.Message,
212 },
213 HttpFailureResponse.FailedDependency);
214 }
215
216 return updateResult switch
217 {
218 ServerUpdateResult.Started => new AuthorityResponse<ServerUpdateResponse>(new ServerUpdateResponse(targetVersion, uploadTicket?.Ticket.FileTicket), HttpSuccessResponse.Accepted),
219 ServerUpdateResult.ReleaseMissing => Gone<ServerUpdateResponse>(),
220 ServerUpdateResult.UpdateInProgress => BadRequest<ServerUpdateResponse>(ErrorCode.ServerUpdateInProgress),
221 ServerUpdateResult.SwarmIntegrityCheckFailed => new AuthorityResponse<ServerUpdateResponse>(
222 new ErrorMessageResponse(ErrorCode.SwarmIntegrityCheckFailed),
223 HttpFailureResponse.FailedDependency),
224 _ => throw new InvalidOperationException($"Unexpected ServerUpdateResult: {updateResult}"),
225 };
