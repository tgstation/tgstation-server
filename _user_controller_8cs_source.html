84 public async Task<IActionResult> Create([FromBody] UserCreateRequest model, CancellationToken cancellationToken)
85 {
86 if (model == null)
87 throw new ArgumentNullException(nameof(model));
88
89 if (model.OAuthConnections?.Any(x => x == null) == true)
90 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure));
91
92 if ((model.Password != null && model.SystemIdentifier != null)
93 || (model.Password == null && model.SystemIdentifier == null && model.OAuthConnections?.Any() != true))
94 return BadRequest(new ErrorMessageResponse(ErrorCode.UserMismatchPasswordSid));
95
96 if (model.Group != null && model.PermissionSet != null)
97 return BadRequest(new ErrorMessageResponse(ErrorCode.UserGroupAndPermissionSet));
98
99 model.Name = model.Name?.Trim();
100 if (model.Name?.Length == 0)
101 model.Name = null;
102
103 if (!(model.Name == null ^ model.SystemIdentifier == null))
104 return BadRequest(new ErrorMessageResponse(ErrorCode.UserMismatchNameSid));
105
106 var fail = CheckValidName(model, true);
107 if (fail != null)
108 return fail;
109
110 var totalUsers = await DatabaseContext
111 .Users
112 .AsQueryable()
113 .CountAsync(cancellationToken)
114 .ConfigureAwait(false);
115 if (totalUsers >= generalConfiguration.UserLimit)
116 return Conflict(new ErrorMessageResponse(ErrorCode.UserLimitReached));
117
118 var dbUser = await CreateNewUserFromModel(model, cancellationToken).ConfigureAwait(false);
119 if (dbUser == null)
120 return Gone();
121
122 if (model.SystemIdentifier != null)
123 try
124 {
125 using var sysIdentity = await systemIdentityFactory.CreateSystemIdentity(dbUser, cancellationToken).ConfigureAwait(false);
126 if (sysIdentity == null)
127 return Gone();
128 dbUser.Name = sysIdentity.Username;
129 dbUser.SystemIdentifier = sysIdentity.Uid;
130 }
131 catch (NotImplementedException)
132 {
133 return RequiresPosixSystemIdentity();
134 }
135 else if (!(model.Password?.Length == 0 && model.OAuthConnections?.Any() == true))
136 {
137 var result = TrySetPassword(dbUser, model.Password, true);
138 if (result != null)
139 return result;
140 }
141
142 dbUser.CanonicalName = Models.User.CanonicalizeName(dbUser.Name);
143
144 DatabaseContext.Users.Add(dbUser);
145
146 await DatabaseContext.Save(cancellationToken).ConfigureAwait(false);
147
148 Logger.LogInformation("Created new user {0} ({1})", dbUser.Name, dbUser.Id);
149
150 return Created(dbUser.ToApi());
151 }

172 public async Task<IActionResult> Update([FromBody] UserUpdateRequest model, CancellationToken cancellationToken)
173 {
174 if (model == null)
175 throw new ArgumentNullException(nameof(model));
176
177 if (!model.Id.HasValue || model.OAuthConnections?.Any(x => x == null) == true)
178 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure));
179
180 if (model.Group != null && model.PermissionSet != null)
181 return BadRequest(new ErrorMessageResponse(ErrorCode.UserGroupAndPermissionSet));
182
183 var callerAdministrationRights = (AdministrationRights)AuthenticationContext.GetRight(RightsType.Administration);
184 var canEditAllUsers = callerAdministrationRights.HasFlag(AdministrationRights.WriteUsers);
185 var passwordEdit = canEditAllUsers || callerAdministrationRights.HasFlag(AdministrationRights.EditOwnPassword);
186 var oAuthEdit = canEditAllUsers || callerAdministrationRights.HasFlag(AdministrationRights.EditOwnOAuthConnections);
187
188 var originalUser = !canEditAllUsers
189 ? AuthenticationContext.User
190 : await DatabaseContext
191 .Users
192 .AsQueryable()
193 .Where(x => x.Id == model.Id)
194 .Include(x => x.CreatedBy)
195 .Include(x => x.OAuthConnections)
196 .Include(x => x.Group)
197 .ThenInclude(x => x.PermissionSet)
198 .Include(x => x.PermissionSet)
199 .FirstOrDefaultAsync(cancellationToken)
200 .ConfigureAwait(false);
201
202 if (originalUser == default)
203 return NotFound();
204
205 if (originalUser.CanonicalName == Models.User.CanonicalizeName(Models.User.TgsSystemUserName))
206 return Forbid();
207
208 // Ensure they are only trying to edit things they have perms for (system identity change will trigger a bad request)
209 if ((!canEditAllUsers
210 && (model.Id != originalUser.Id
211 || model.Enabled.HasValue
212 || model.Group != null
213 || model.PermissionSet != null
214 || model.Name != null))
215 || (!passwordEdit && model.Password != null)
216 || (!oAuthEdit && model.OAuthConnections != null))
217 return Forbid();
218
219 if (model.SystemIdentifier != null && model.SystemIdentifier != originalUser.SystemIdentifier)
220 return BadRequest(new ErrorMessageResponse(ErrorCode.UserSidChange));
221
222 if (model.Password != null)
223 {
224 var result = TrySetPassword(originalUser, model.Password, false);
225 if (result != null)
226 return result;
227 }
228
229 if (model.Name != null && Models.User.CanonicalizeName(model.Name) != originalUser.CanonicalName)
230 return BadRequest(new ErrorMessageResponse(ErrorCode.UserNameChange));
231
232 if (model.Enabled.HasValue)
233 {
234 if (originalUser.Enabled.Value && !model.Enabled.Value)
235 originalUser.LastPasswordUpdate = DateTimeOffset.UtcNow;
236
237 originalUser.Enabled = model.Enabled.Value;
238 }
239
240 if (model.OAuthConnections != null
241 && (model.OAuthConnections.Count != originalUser.OAuthConnections.Count
242 || !model.OAuthConnections.All(x => originalUser.OAuthConnections.Any(y => y.Provider == x.Provider && y.ExternalUserId == x.ExternalUserId))))
243 {
244 if (originalUser.CanonicalName == Models.User.CanonicalizeName(DefaultCredentials.AdminUserName))
245 return BadRequest(new ErrorMessageResponse(ErrorCode.AdminUserCannotOAuth));
246
247 if (model.OAuthConnections.Count == 0 && originalUser.PasswordHash == null && originalUser.SystemIdentifier == null)
248 return BadRequest(new ErrorMessageResponse(ErrorCode.CannotRemoveLastAuthenticationOption));
249
250 originalUser.OAuthConnections.Clear();
251 foreach (var updatedConnection in model.OAuthConnections)
252 originalUser.OAuthConnections.Add(new Models.OAuthConnection
253 {
254 Provider = updatedConnection.Provider,
255 ExternalUserId = updatedConnection.ExternalUserId,
256 });
257 }
258
259 if (model.Group != null)
260 {
261 originalUser.Group = await DatabaseContext
262 .Groups
263 .AsQueryable()
264 .Where(x => x.Id == model.Group.Id)
265 .Include(x => x.PermissionSet)
266 .FirstOrDefaultAsync(cancellationToken)
267 .ConfigureAwait(false);
268
269 if (originalUser.Group == default)
270 return Gone();
271
272 DatabaseContext.Groups.Attach(originalUser.Group);
273 if (originalUser.PermissionSet != null)
274 {
275 Logger.LogInformation("Deleting permission set {0}...", originalUser.PermissionSet.Id);
276 DatabaseContext.PermissionSets.Remove(originalUser.PermissionSet);
277 originalUser.PermissionSet = null;
278 }
279 }
280 else if (model.PermissionSet != null)
281 {
282 if (originalUser.PermissionSet == null)
283 {
284 Logger.LogTrace("Creating new permission set...");
285 originalUser.PermissionSet = new Models.PermissionSet();
286 }
287
288 originalUser.PermissionSet.AdministrationRights = model.PermissionSet.AdministrationRights ?? AdministrationRights.None;
289 originalUser.PermissionSet.InstanceManagerRights = model.PermissionSet.InstanceManagerRights ?? InstanceManagerRights.None;
290
291 originalUser.Group = null;
292 originalUser.GroupId = null;
293 }
294
295 var fail = CheckValidName(model, false);
296 if (fail != null)
297 return fail;
298
299 originalUser.Name = model.Name ?? originalUser.Name;
300
301 await DatabaseContext.Save(cancellationToken).ConfigureAwait(false);
302
303 Logger.LogInformation("Updated user {0} ({1})", originalUser.Name, originalUser.Id);
304
305 // return id only if not a self update and cannot read users
306 var canReadBack = AuthenticationContext.User.Id == originalUser.Id
307 || callerAdministrationRights.HasFlag(AdministrationRights.ReadUsers);
308 return canReadBack
309 ? (IActionResult)Json(originalUser.ToApi())
310 : NoContent();
311 }
