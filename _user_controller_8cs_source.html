98 public async ValueTask<IActionResult> Create([FromBody] UserCreateRequest model, CancellationToken cancellationToken)
99 {
100 ArgumentNullException.ThrowIfNull(model);
101
102 if (model.OAuthConnections?.Any(x => x == null) == true)
103 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure));
104
105 if ((model.Password != null && model.SystemIdentifier != null)
106 || (model.Password == null && model.SystemIdentifier == null && (model.OAuthConnections?.Count > 0) != true))
107 return BadRequest(new ErrorMessageResponse(ErrorCode.UserMismatchPasswordSid));
108
109 if (model.Group != null && model.PermissionSet != null)
110 return BadRequest(new ErrorMessageResponse(ErrorCode.UserGroupAndPermissionSet));
111
112 model.Name = model.Name?.Trim();
113 if (model.Name?.Length == 0)
114 model.Name = null;
115
116 if (!(model.Name == null ^ model.SystemIdentifier == null))
117 return BadRequest(new ErrorMessageResponse(ErrorCode.UserMismatchNameSid));
118
119 var fail = CheckValidName(model, true);
120 if (fail != null)
121 return fail;
122
123 var totalUsers = await DatabaseContext
124 .Users
125 .AsQueryable()
126 .CountAsync(cancellationToken);
127 if (totalUsers >= generalConfiguration.UserLimit)
128 return Conflict(new ErrorMessageResponse(ErrorCode.UserLimitReached));
129
130 var dbUser = await CreateNewUserFromModel(model, cancellationToken);
131 if (dbUser == null)
132 return this.Gone();
133
134 if (model.SystemIdentifier != null)
135 try
136 {
137 using var sysIdentity = await systemIdentityFactory.CreateSystemIdentity(dbUser, cancellationToken);
138 if (sysIdentity == null)
139 return this.Gone();
140 dbUser.Name = sysIdentity.Username;
141 dbUser.SystemIdentifier = sysIdentity.Uid;
142 }
143 catch (NotImplementedException ex)
144 {
145 return RequiresPosixSystemIdentity(ex);
146 }
147 else if (!(model.Password?.Length == 0 && (model.OAuthConnections?.Count > 0) == true))
148 {
149 var result = TrySetPassword(dbUser, model.Password!, true);
150 if (result != null)
151 return result;
152 }
153
154 dbUser.CanonicalName = Models.User.CanonicalizeName(dbUser.Name!);
155
156 DatabaseContext.Users.Add(dbUser);
157
158 await DatabaseContext.Save(cancellationToken);
159
160 Logger.LogInformation("Created new user {name} ({id})", dbUser.Name, dbUser.Id);
161
162 return Created(dbUser.ToApi());
163 }

184 public async ValueTask<IActionResult> Update([FromBody] UserUpdateRequest model, CancellationToken cancellationToken)
185 {
186 ArgumentNullException.ThrowIfNull(model);
187
188 if (!model.Id.HasValue || model.OAuthConnections?.Any(x => x == null) == true)
189 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure));
190
191 if (model.Group != null && model.PermissionSet != null)
192 return BadRequest(new ErrorMessageResponse(ErrorCode.UserGroupAndPermissionSet));
193
194 var callerAdministrationRights = (AdministrationRights)AuthenticationContext.GetRight(RightsType.Administration);
195 var canEditAllUsers = callerAdministrationRights.HasFlag(AdministrationRights.WriteUsers);
196 var passwordEdit = canEditAllUsers || callerAdministrationRights.HasFlag(AdministrationRights.EditOwnPassword);
197 var oAuthEdit = canEditAllUsers || callerAdministrationRights.HasFlag(AdministrationRights.EditOwnOAuthConnections);
198
199 var originalUser = !canEditAllUsers
200 ? AuthenticationContext.User
201 : await DatabaseContext
202 .Users
203 .AsQueryable()
204 .Where(x => x.Id == model.Id)
205 .Include(x => x.CreatedBy)
206 .Include(x => x.OAuthConnections)
207 .Include(x => x.Group!)
208 .ThenInclude(x => x.PermissionSet)
209 .Include(x => x.PermissionSet)
210 .FirstOrDefaultAsync(cancellationToken);
211
212 if (originalUser == default)
213 return NotFound();
214
215 if (originalUser.CanonicalName == Models.User.CanonicalizeName(Models.User.TgsSystemUserName))
216 return Forbid();
217
218 // Ensure they are only trying to edit things they have perms for (system identity change will trigger a bad request)
219 if ((!canEditAllUsers
220 && (model.Id != originalUser.Id
221 || model.Enabled.HasValue
222 || model.Group != null
223 || model.PermissionSet != null
224 || model.Name != null))
225 || (!passwordEdit && model.Password != null)
226 || (!oAuthEdit && model.OAuthConnections != null))
227 return Forbid();
228
229 var originalUserHasSid = originalUser.SystemIdentifier != null;
230 if (originalUserHasSid && originalUser.PasswordHash != null)
231 {
232 // cleanup from https://github.com/tgstation/tgstation-server/issues/1528
233 Logger.LogDebug("System user ID {userId}'s PasswordHash is polluted, updating database.", originalUser.Id);
234 originalUser.PasswordHash = null;
235 originalUser.LastPasswordUpdate = DateTimeOffset.UtcNow;
236 }
237
238 if (model.SystemIdentifier != null && model.SystemIdentifier != originalUser.SystemIdentifier)
239 return BadRequest(new ErrorMessageResponse(ErrorCode.UserSidChange));
240
241 if (model.Password != null)
242 {
243 if (originalUserHasSid)
244 return BadRequest(new ErrorMessageResponse(ErrorCode.UserMismatchPasswordSid));
245
246 var result = TrySetPassword(originalUser, model.Password, false);
247 if (result != null)
248 return result;
249 }
250
251 if (model.Name != null && Models.User.CanonicalizeName(model.Name) != originalUser.CanonicalName)
252 return BadRequest(new ErrorMessageResponse(ErrorCode.UserNameChange));
253
254 bool userWasDisabled;
255 if (model.Enabled.HasValue)
256 {
257 userWasDisabled = originalUser.Require(x => x.Enabled) && !model.Enabled.Value;
258 if (userWasDisabled)
259 originalUser.LastPasswordUpdate = DateTimeOffset.UtcNow;
260
261 originalUser.Enabled = model.Enabled.Value;
262 }
263 else
264 userWasDisabled = false;
265
266 if (model.OAuthConnections != null
267 && (model.OAuthConnections.Count != originalUser.OAuthConnections!.Count
268 || !model.OAuthConnections.All(x => originalUser.OAuthConnections.Any(y => y.Provider == x.Provider && y.ExternalUserId == x.ExternalUserId))))
269 {
270 if (originalUser.CanonicalName == Models.User.CanonicalizeName(DefaultCredentials.AdminUserName))
271 return BadRequest(new ErrorMessageResponse(ErrorCode.AdminUserCannotOAuth));
272
273 if (model.OAuthConnections.Count == 0 && originalUser.PasswordHash == null && originalUser.SystemIdentifier == null)
274 return BadRequest(new ErrorMessageResponse(ErrorCode.CannotRemoveLastAuthenticationOption));
275
276 originalUser.OAuthConnections.Clear();
277 foreach (var updatedConnection in model.OAuthConnections)
278 originalUser.OAuthConnections.Add(new Models.OAuthConnection
279 {
280 Provider = updatedConnection.Provider,
281 ExternalUserId = updatedConnection.ExternalUserId,
282 });
283 }
284
285 if (model.Group != null)
286 {
287 originalUser.Group = await DatabaseContext
288 .Groups
289 .AsQueryable()
290 .Where(x => x.Id == model.Group.Id)
291 .Include(x => x.PermissionSet)
292 .FirstOrDefaultAsync(cancellationToken);
293
294 if (originalUser.Group == default)
295 return this.Gone();
296
297 DatabaseContext.Groups.Attach(originalUser.Group);
298 if (originalUser.PermissionSet != null)
299 {
300 Logger.LogInformation("Deleting permission set {permissionSetId}...", originalUser.PermissionSet.Id);
301 DatabaseContext.PermissionSets.Remove(originalUser.PermissionSet);
302 originalUser.PermissionSet = null;
303 }
304 }
305 else if (model.PermissionSet != null)
306 {
307 if (originalUser.PermissionSet == null)
308 {
309 Logger.LogTrace("Creating new permission set...");
310 originalUser.PermissionSet = new Models.PermissionSet();
311 }
312
313 originalUser.PermissionSet.AdministrationRights = model.PermissionSet.AdministrationRights ?? AdministrationRights.None;
314 originalUser.PermissionSet.InstanceManagerRights = model.PermissionSet.InstanceManagerRights ?? InstanceManagerRights.None;
315
316 originalUser.Group = null;
317 originalUser.GroupId = null;
318 }
319
320 var fail = CheckValidName(model, false);
321 if (fail != null)
322 return fail;
323
324 originalUser.Name = model.Name ?? originalUser.Name;
325
326 await DatabaseContext.Save(cancellationToken);
327
328 Logger.LogInformation("Updated user {userName} ({userId})", originalUser.Name, originalUser.Id);
329
330 if (userWasDisabled)
331 await permissionsUpdateNotifyee.UserDisabled(originalUser, cancellationToken);
332
333 // return id only if not a self update and cannot read users
334 var canReadBack = AuthenticationContext.User.Id == originalUser.Id
335 || callerAdministrationRights.HasFlag(AdministrationRights.ReadUsers);
336 return canReadBack
337 ? Json(originalUser.ToApi())
338 : NoContent();
339 }
