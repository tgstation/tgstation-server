125 ushort port,

127 {
128 const int MaximumTerminationSeconds = 5;
129
130 logger.LogTrace("Attempting Robust.Server graceful exit (Timeout: {seconds}s)...", MaximumTerminationSeconds);
131 var timeout = asyncDelayer.Delay(TimeSpan.FromSeconds(MaximumTerminationSeconds), cancellationToken);
132 var lifetime = process.Lifetime;
133
134 var stopwatch = Stopwatch.StartNew();
135 try
136 {
137 using var httpClient = httpClientFactory.CreateClient();
138 using var request = new HttpRequestMessage();
139 request.Headers.Add("WatchdogToken", accessIdentifier);
140 request.RequestUri = new Uri($"http://localhost:{port}/shutdown");
141 request.Content = new StringContent(
142 "{\"Reason\":\"TGS session termination\"}",
143 Encoding.UTF8,
144 new MediaTypeHeaderValue(MediaTypeNames.Application.Json));
145 request.Method = HttpMethod.Post;
146
147 var responseTask = httpClient.SendAsync(request, HttpCompletionOption.ResponseHeadersRead, cancellationToken);
148 try
149 {
150 await Task.WhenAny(timeout, lifetime, responseTask);
151 if (responseTask.IsCompleted)
152 {
153 using var response = await responseTask;
154 if (response.IsSuccessStatusCode)
155 {
156 logger.LogDebug("Robust.Server responded to the shutdown command successfully ({requestMs}ms). Waiting for exit...", stopwatch.ElapsedMilliseconds);
157 await Task.WhenAny(timeout, lifetime);
158 }
159 }
160
161 if (!lifetime.IsCompleted)
162 logger.LogWarning("Robust.Server graceful exit timed out!");
163 }
164 catch (Exception ex) when (ex is not OperationCanceledException)
165 {
166 logger.LogDebug(ex, "Unable to send graceful exit request to Robust.Server watchdog API!");
167 }
168
169 if (lifetime.IsCompleted)
170 {
171 logger.LogTrace("Robust.Server exited without termination");
172 return;
173 }
174 }
175 finally
176 {
177 logger.LogTrace("Robust.Server graceful shutdown attempt took {totalMs}ms", stopwatch.ElapsedMilliseconds);
178 }
