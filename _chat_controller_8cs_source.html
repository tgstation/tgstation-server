81 public async Task<IActionResult> Create([FromBody] ChatBotCreateRequest model, CancellationToken cancellationToken)
82 {
83 if (model == null)
84 throw new ArgumentNullException(nameof(model));
85
86 var earlyOut = StandardModelChecks(model, true);
87 if (earlyOut != null)
88 return earlyOut;
89
90 var countOfExistingBotsInInstance = await DatabaseContext
91 .ChatBots
92 .AsQueryable()
93 .Where(x => x.InstanceId == Instance.Id)
94 .CountAsync(cancellationToken)
95 .ConfigureAwait(false);
96
97 if (countOfExistingBotsInInstance >= Instance.ChatBotLimit.Value)
98 return Conflict(new ErrorMessageResponse(ErrorCode.ChatBotMax));
99
100 model.Enabled ??= false;
101 model.ReconnectionInterval ??= 1;
102
103 // try to update das db first
104 var dbModel = new ChatBot
105 {
106 Name = model.Name,
107 ConnectionString = model.ConnectionString,
108 Enabled = model.Enabled,
109 Channels = model.Channels?.Select(x => ConvertApiChatChannel(x)).ToList() ?? new List<Models.ChatChannel>(), // important that this isn't null
110 InstanceId = Instance.Id.Value,
111 Provider = model.Provider,
112 ReconnectionInterval = model.ReconnectionInterval,
113 ChannelLimit = model.ChannelLimit,
114 };
115
116 DatabaseContext.ChatBots.Add(dbModel);
117
118 await DatabaseContext.Save(cancellationToken).ConfigureAwait(false);
119 return await WithComponentInstance(
120 async instance =>
121 {
122 try
123 {
124 // try to create it
125 await instance.Chat.ChangeSettings(dbModel, cancellationToken).ConfigureAwait(false);
126
127 if (dbModel.Channels.Count > 0)
128 await instance.Chat.ChangeChannels(dbModel.Id.Value, dbModel.Channels, cancellationToken).ConfigureAwait(false);
129 }
130 catch
131 {
132 // undo the add
133 DatabaseContext.ChatBots.Remove(dbModel);
134
135 // DCTx2: Operations must always run
136 await DatabaseContext.Save(default).ConfigureAwait(false);
137 await instance.Chat.DeleteConnection(dbModel.Id.Value, default).ConfigureAwait(false);
138 throw;
139 }
140
141 return null;
142 })
143 .ConfigureAwait(false)
144 ?? StatusCode(HttpStatusCode.Created, dbModel.ToApi());
145 }

289 bool CheckModified<T>(Expression<Func<ChatBotSettings, T>> expression, ChatBotRights requiredRight)
290 {
291 var memberSelectorExpression = (MemberExpression)expression.Body;
292 var property = (PropertyInfo)memberSelectorExpression.Member;
293
294 var newVal = property.GetValue(model);
295 if (newVal == null)
296 return false;
297 if (!userRights.HasFlag(requiredRight) && property.GetValue(current) != newVal)
298 return true;
299
300 property.SetValue(current, newVal);
301 anySettingsModified = true;
302 return false;
303 }
304
305 var oldProvider = current.Provider;
306
307 if (CheckModified(x => x.ConnectionString, ChatBotRights.WriteConnectionString)
308 || CheckModified(x => x.Enabled, ChatBotRights.WriteEnabled)
309 || CheckModified(x => x.Name, ChatBotRights.WriteName)
310 || CheckModified(x => x.Provider, ChatBotRights.WriteProvider)
311 || CheckModified(x => x.ReconnectionInterval, ChatBotRights.WriteReconnectionInterval)
312 || CheckModified(x => x.ChannelLimit, ChatBotRights.WriteChannelLimit)
313 || (model.Channels != null && !userRights.HasFlag(ChatBotRights.WriteChannels)))
314 return Forbid();
315
316 var hasChannels = model.Channels != null;
317 if (hasChannels || (model.Provider.HasValue && model.Provider != oldProvider))
318 {
319 DatabaseContext.ChatChannels.RemoveRange(current.Channels);
320 if (hasChannels)
321 {
322 var dbChannels = model.Channels.Select(x => ConvertApiChatChannel(x)).ToList();
323 DatabaseContext.ChatChannels.AddRange(dbChannels);
324 current.Channels = dbChannels;
325 }

331
332 earlyOut = await WithComponentInstance(

338

364 private IActionResult StandardModelChecks(ChatBotApiBase model, bool forCreation)
365 {
366 if (model.ReconnectionInterval == 0)
367 throw new InvalidOperationException("RecconnectionInterval cannot be zero!");
368
369 if (forCreation && !model.Provider.HasValue)
370 return BadRequest(new ErrorMessageResponse(ErrorCode.ChatBotProviderMissing));
371
372 if (model.Name != null && String.IsNullOrWhiteSpace(model.Name))
373 return BadRequest(new ErrorMessageResponse(ErrorCode.ChatBotWhitespaceName));
374
375 if (model.ConnectionString != null && String.IsNullOrWhiteSpace(model.ConnectionString))
376 return BadRequest(new ErrorMessageResponse(ErrorCode.ChatBotWhitespaceConnectionString));
377
378 if (!model.ValidateProviderChannelTypes())
379 return BadRequest(new ErrorMessageResponse(ErrorCode.ChatBotWrongChannelType));
380
381 var defaultMaxChannels = (ulong)Math.Max(ChatBot.DefaultChannelLimit, model.Channels?.Count ?? 0);
382 if (defaultMaxChannels > UInt16.MaxValue)
383 return BadRequest(new ErrorMessageResponse(ErrorCode.ChatBotMaxChannels));
384
385 if (forCreation)
386 model.ChannelLimit ??= (ushort)defaultMaxChannels;
387
388 return null;
389 }
390 }
