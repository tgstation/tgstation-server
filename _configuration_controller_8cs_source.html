345 bool ForbidDueToModeConflicts(string? path, out ISystemIdentity? systemIdentityToUse)
346 {
347 if (Instance.ConfigurationType == ConfigurationType.Disallowed
348 || (Instance.ConfigurationType == ConfigurationType.SystemIdentityWrite && AuthenticationContext.SystemIdentity == null)
349 || (path != null && ioManager.PathContainsParentAccess(path)))
350 {
351 systemIdentityToUse = null;
352 return true;
353 }
354
355 systemIdentityToUse = Instance.ConfigurationType == ConfigurationType.SystemIdentityWrite ? AuthenticationContext.SystemIdentity : null;
356 return false;
