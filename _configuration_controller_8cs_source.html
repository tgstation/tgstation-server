344 bool ForbidDueToModeConflicts(string? path, out ISystemIdentity? systemIdentityToUse)
345 {
346 if (Instance.ConfigurationType == ConfigurationType.Disallowed
347 || (Instance.ConfigurationType == ConfigurationType.SystemIdentityWrite && AuthenticationContext.SystemIdentity == null)
348 || (path != null && ioManager.PathContainsParentAccess(path)))
349 {
350 systemIdentityToUse = null;
351 return true;
352 }
