138 public async Task<IActionResult> Read(CancellationToken cancellationToken)
139 {
140 try
141 {
142 Version greatestVersion = null;
143 Uri repoUrl = null;
144 try
145 {
146 var gitHubService = gitHubServiceFactory.CreateService();
147 var repositoryUrlTask = gitHubService.GetUpdatesRepositoryUrl(cancellationToken);
148 var releases = await gitHubService.GetTgsReleases(cancellationToken);
149
150 foreach (var kvp in releases)
151 {
152 var version = kvp.Key;
153 var release = kvp.Value;
154 if (version.Major > 3 // Forward/backward compatible but not before TGS4
155 && (greatestVersion == null || version > greatestVersion))
156 greatestVersion = version;
157 }
158
159 repoUrl = await repositoryUrlTask;
160 }
161 catch (NotFoundException e)
162 {
163 Logger.LogWarning(e, "Not found exception while retrieving upstream repository info!");
164 }
165
166 return Json(new AdministrationResponse
167 {
168 LatestVersion = greatestVersion,
169 TrackedRepositoryUrl = repoUrl,
170 });
171 }
172 catch (RateLimitExceededException e)
173 {
174 return RateLimit(e);
175 }
176 catch (ApiException e)
177 {
178 Logger.LogWarning(e, OctokitException);
179 return this.StatusCode(HttpStatusCode.FailedDependency, new ErrorMessageResponse(ErrorCode.RemoteApiError)
180 {
181 AdditionalData = e.Message,

199 [ProducesResponseType(typeof(ServerUpdateResponse), 202)]
200 [ProducesResponseType(typeof(ErrorMessageResponse), 410)]

204 public async Task<IActionResult> Update([FromBody] ServerUpdateRequest model, CancellationToken cancellationToken)
205 {
206 ArgumentNullException.ThrowIfNull(model);
207
208 var attemptingUpload = model.UploadZip == true;
209 if (attemptingUpload)
210 {
211 if (!AuthenticationContext.PermissionSet.AdministrationRights.Value.HasFlag(AdministrationRights.UploadVersion))
212 return Forbid();
213 }
214 else if (!AuthenticationContext.PermissionSet.AdministrationRights.Value.HasFlag(AdministrationRights.ChangeVersion))
215 return Forbid();
216
217 if (model.NewVersion == null)
218 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure)
219 {
220 AdditionalData = "newVersion is required!",
221 });
222
223 if (model.NewVersion.Major < 3)
224 return BadRequest(new ErrorMessageResponse(ErrorCode.CannotChangeServerSuite));
225
226 if (!serverControl.WatchdogPresent)
227 return UnprocessableEntity(new ErrorMessageResponse(ErrorCode.MissingHostWatchdog));
228
229 return await AttemptInitiateUpdate(model.NewVersion, attemptingUpload, cancellationToken);
230 }

373 async Task<IActionResult> AttemptInitiateUpdate(Version newVersion, bool attemptingUpload, CancellationToken cancellationToken)
374 {
375 IFileUploadTicket uploadTicket = attemptingUpload
376 ? fileTransferService.CreateUpload(FileUploadStreamKind.None)
377 : null;
378
379 ServerUpdateResult updateResult;
380 try
381 {
382 try
383 {
384 updateResult = await serverUpdateInitiator.InitiateUpdate(uploadTicket, newVersion, cancellationToken);
385 }
386 catch
387 {
388 if (attemptingUpload)
389 await uploadTicket.DisposeAsync();
390
391 throw;
392 }
393 }
394 catch (RateLimitExceededException e)
395 {
396 return RateLimit(e);
397 }
398 catch (ApiException e)
399 {
400 Logger.LogWarning(e, OctokitException);
401 return this.StatusCode(HttpStatusCode.FailedDependency, new ErrorMessageResponse(ErrorCode.RemoteApiError)
402 {
403 AdditionalData = e.Message,
404 });
405 }
406
407 return updateResult switch
408 {
409 ServerUpdateResult.Started => Accepted(new ServerUpdateResponse(newVersion, uploadTicket?.Ticket.FileTicket)),
410 ServerUpdateResult.ReleaseMissing => this.Gone(),
411 ServerUpdateResult.UpdateInProgress => BadRequest(new ErrorMessageResponse(ErrorCode.ServerUpdateInProgress)),
412 ServerUpdateResult.SwarmIntegrityCheckFailed => this.StatusCode(HttpStatusCode.FailedDependency, new ErrorMessageResponse(ErrorCode.SwarmIntegrityCheckFailed)),
413 _ => throw new InvalidOperationException($"Unexpected ServerUpdateResult: {updateResult}"),
414 };
415 }
