176 => Paginated(
177 async () =>
178 {
179 var path = fileLoggingConfigurationOptions.Value.GetFullLogDirectory(ioManager, assemblyInformationProvider, platformIdentifier);
180 try
181 {
182 var files = await ioManager.GetFiles(path, cancellationToken);
183 var tasks = files.Select(
184 async file => new LogFileResponse

186 Name = ioManager.GetFileName(file),
187 LastModified = await ioManager
188 .GetLastModified(
189 ioManager.ConcatPath(path, file),
190 cancellationToken),
191 })
192 .ToList();
193
194 await Task.WhenAll(tasks);
195
196 return new PaginatableResult<LogFileResponse>(
197 tasks
198 .AsQueryable()
199 .Select(x => x.Result)
200 .OrderByDescending(x => x.Name));
201 }
202 catch (IOException ex)

204 return new PaginatableResult<LogFileResponse>(
205 Conflict(new ErrorMessageResponse(ErrorCode.IOError)
206 {
207 AdditionalData = ex.ToString(),
208 }));
209 }

211 null,
212 page,
213 pageSize,
214 cancellationToken);
