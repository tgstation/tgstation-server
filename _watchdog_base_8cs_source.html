493 bool startMonitor,
494 bool announce,
495 bool announceFailure,
496 ReattachInformation reattachInfo,
497 CancellationToken cancellationToken)
498 {
499 Logger.LogTrace("Begin LaunchImplNoLock");
500 if (startMonitor && Status != WatchdogStatus.Offline)
501 throw new JobException(ErrorCode.WatchdogRunning);
502
503 if (reattachInfo == null && !DmbFactory.DmbAvailable)
504 throw new JobException(ErrorCode.WatchdogCompileJobCorrupted);
505
506 // this is necessary, the monitor could be in it's sleep loop trying to restart, if so cancel THAT monitor and start our own with blackjack and hookers
507 Task announceTask;
508 if (announce)
509 {
510 announceTask = Chat.QueueWatchdogMessage(
511 reattachInfo == null
512 ? "Launching..."
513 : "Reattaching...",
514 cancellationToken); // simple announce
515 if (reattachInfo == null)
516 announceTask = Task.WhenAll(
517 HandleEvent(EventType.WatchdogLaunch, Enumerable.Empty<string>(), false, cancellationToken),
518 announceTask);
519 }
520 else
521 announceTask = Task.CompletedTask; // no announce
522
523 // since neither server is running, this is safe to do
524 LastLaunchParameters = ActiveLaunchParameters;
525 heartbeatsMissed = 0;
526
527 try
528 {
529 await InitController(announceTask, reattachInfo, cancellationToken);
530 }
531 catch (OperationCanceledException ex)
532 {
533 Logger.LogTrace(ex, "Controller initialization cancelled!");
534 throw;
535 }
536 catch (Exception e)
537 {
538 Logger.LogWarning(e, "Failed to start watchdog!");
539 var originalChatTask = announceTask;
540 async Task ChainChatTaskWithErrorMessage()
541 {
542 await originalChatTask;
543 if (announceFailure)
544 await Chat.QueueWatchdogMessage("Startup failed!", cancellationToken);
545 }
546
547 announceTask = ChainChatTaskWithErrorMessage();
548 throw;
549 }
550 finally
551 {
552 // finish the chat task that's in flight
553 try
554 {
555 await announceTask;
556 }
557 catch (OperationCanceledException ex)
558 {
559 Logger.LogTrace(ex, "Announcement task canceled!");
560 }
561 }
562
563 Logger.LogInformation("Controller(s) initialized successfully");
564
565 if (startMonitor)
566 {
567 monitorCts = new CancellationTokenSource();
568 monitorTask = MonitorLifetimes(monitorCts.Token);
569 }

724 async Task MonitorRestart(CancellationToken cancellationToken)
725 {
726 Logger.LogTrace("Monitor restart!");
727
728 await DisposeAndNullControllers(cancellationToken);
729
730 var chatTask = Task.CompletedTask;
731 for (var retryAttempts = 1; ; ++retryAttempts)
732 {
733 Status = WatchdogStatus.Restoring;
734 Exception launchException;
735 using (await SemaphoreSlimContext.Lock(synchronizationSemaphore, cancellationToken))
736 try
737 {
738 // use LaunchImplNoLock without announcements or restarting the monitor
739 await LaunchNoLock(false, false, false, null, cancellationToken);
740 Status = WatchdogStatus.Online;
741 Logger.LogDebug("Relaunch successful, resuming monitor...");
742 return;
743 }
744 catch (Exception e) when (e is not OperationCanceledException)
745 {
746 launchException = e;
747 }
748 finally
749 {
750 await chatTask;
751 }
752
753 Logger.LogWarning(launchException, "Failed to automatically restart the watchdog! Attempt: {attemptNumber}", retryAttempts);
754 Status = WatchdogStatus.DelayedRestart;
755
756 var retryDelay = Math.Min(
757 Convert.ToInt32(
758 Math.Pow(2, retryAttempts)),
759 TimeSpan.FromHours(1).TotalSeconds); // max of one hour, increasing by a power of 2 each time
760
761 chatTask = Chat.QueueWatchdogMessage(
762 $"Failed to restart (Attempt: {retryAttempts}), retrying in {retryDelay}s...",
763 cancellationToken);
764
765 await Task.WhenAll(
766 AsyncDelayer.Delay(
767 TimeSpan.FromSeconds(retryDelay),
768 cancellationToken),
769 chatTask)
770 ;
771 }

802 async Task MonitorLifetimes(CancellationToken cancellationToken)
803 {
804 Logger.LogTrace("Entered MonitorLifetimes");
805 Status = WatchdogStatus.Online;
806 using var cancellationTokenLoggingRegistration = cancellationToken.Register(() => Logger.LogTrace("Monitor cancellationToken triggered"));
807
808 // this function is responsible for calling HandlerMonitorWakeup when necessary and manitaining the MonitorState
809 try
810 {
811 MonitorAction nextAction = MonitorAction.Continue;
812 Task activeServerLifetime = null,
813 activeServerReboot = null,
814 serverPrimed = null,
815 activeLaunchParametersChanged = null,
816 newDmbAvailable = null;
817 ISessionController lastController = null;
818 var ranInitialDmbCheck = false;
819 for (ulong iteration = 1; nextAction != MonitorAction.Exit; ++iteration)
820 using (LogContext.PushProperty(SerilogContextHelper.WatchdogMonitorIterationContextProperty, iteration))
821 try
822 {
823 Logger.LogTrace("Iteration {iteration} of monitor loop", iteration);
824 nextAction = MonitorAction.Continue;
825
826 var controller = GetActiveController();
827
828 void UpdateMonitoredTasks()
829 {
830 static void TryUpdateTask(ref Task oldTask, Func<Task> newTaskFactory)
831 {
832 if (oldTask?.IsCompleted == true)
833 return;
834
835 oldTask = newTaskFactory();
836 }
837
838 if (lastController == controller)
839 {
840 TryUpdateTask(ref activeServerLifetime, () => controller.Lifetime);
841 TryUpdateTask(ref activeServerReboot, () => controller.OnReboot);
842 TryUpdateTask(ref serverPrimed, () => controller.OnPrime);
843 }
844 else
845 {
846 activeServerLifetime = controller.Lifetime;
847 activeServerReboot = controller.OnReboot;
848 serverPrimed = controller.OnPrime;
849 lastController = controller;
850 }
851
852 TryUpdateTask(ref activeLaunchParametersChanged, () => ActiveParametersUpdated.Task);
853 TryUpdateTask(
854 ref newDmbAvailable,
855 () =>
856 {
857 var result = ranInitialDmbCheck
858 ? DmbFactory.OnNewerDmb
859 : InitialCheckDmbUpdated(controller.CompileJob);
860 ranInitialDmbCheck = true;
861 return result;
862 });
863 }
864
865 UpdateMonitoredTasks();
866
867 var heartbeatSeconds = ActiveLaunchParameters.HeartbeatSeconds.Value;
868 var heartbeat = heartbeatSeconds == 0
869 || !controller.DMApiAvailable
870 ? Extensions.TaskExtensions.InfiniteTask()
871 : Task.Delay(
872 TimeSpan.FromSeconds(heartbeatSeconds),
873 cancellationToken);
874
875 // cancel waiting if requested
876 var cancelTcs = new TaskCompletionSource();
877 var toWaitOn = Task.WhenAny(
878 activeServerLifetime,
879 activeServerReboot,
880 heartbeat,
881 newDmbAvailable,
882 cancelTcs.Task,
883 activeLaunchParametersChanged,
884 serverPrimed);
885
886 // wait for something to happen
887 using (cancellationToken.Register(() => cancelTcs.SetCanceled()))
888 await toWaitOn;
889
890 cancellationToken.ThrowIfCancellationRequested();
891 Logger.LogTrace("Monitor activated");
892
893 // always run HandleMonitorWakeup from the context of the semaphore lock
894 using (await SemaphoreSlimContext.Lock(synchronizationSemaphore, cancellationToken))
895 {
896 // Set this sooner so chat sends don't hold us up
897 if (activeServerLifetime.IsCompleted)
898 Status = WatchdogStatus.Restoring;
899
900 // multiple things may have happened, handle them one at a time
901 for (var moreActivationsToProcess = true; moreActivationsToProcess && (nextAction == MonitorAction.Continue || nextAction == MonitorAction.Skip);)
902 {
903 MonitorActivationReason activationReason = default; // this will always be assigned before being used
904
905 bool CheckActivationReason(ref Task task, MonitorActivationReason testActivationReason)
906 {
907 var taskCompleted = task?.IsCompleted == true;
908 task = null;
909 if (nextAction == MonitorAction.Skip)
910 nextAction = MonitorAction.Continue;
911 else if (taskCompleted)
912 {
913 activationReason = testActivationReason;
914 return true;
915 }
916
917 return false;
918 }
919
920 // process the tasks in this order and call HandlerMonitorWakup for each depending on the new monitorState
921 var anyActivation = CheckActivationReason(ref activeServerLifetime, MonitorActivationReason.ActiveServerCrashed)
922 || CheckActivationReason(ref activeServerReboot, MonitorActivationReason.ActiveServerRebooted)
923 || CheckActivationReason(ref newDmbAvailable, MonitorActivationReason.NewDmbAvailable)
924 || CheckActivationReason(ref activeLaunchParametersChanged, MonitorActivationReason.ActiveLaunchParametersUpdated)
925 || CheckActivationReason(ref heartbeat, MonitorActivationReason.Heartbeat)
926 || CheckActivationReason(ref serverPrimed, MonitorActivationReason.ActiveServerPrimed);
927
928 UpdateMonitoredTasks();
929
930 if (!anyActivation)
931 moreActivationsToProcess = false;
932 else
933 {
934 Logger.LogTrace("Reason: {activationReason}", activationReason);
935 if (activationReason == MonitorActivationReason.Heartbeat)
936 nextAction = await HandleHeartbeat(
937 cancellationToken);
938 else
939 nextAction = await HandleMonitorWakeup(
940 activationReason,
941 cancellationToken);
942 }
943 }
944 }
945
946 Logger.LogTrace("Next monitor action is to {nextAction}", nextAction);
947
948 // Restart if requested
949 if (nextAction == MonitorAction.Restart)
950 {
951 await MonitorRestart(cancellationToken);
952 nextAction = MonitorAction.Continue;
953 }
954 }
955 catch (Exception e) when (e is not OperationCanceledException)
956 {
957 // really, this should NEVER happen
958 Logger.LogError(
959 e,
960 "Monitor crashed! Iteration: {iteration}",
961 iteration);
962
963 var nextActionMessage = nextAction != MonitorAction.Exit
964 ? "Recovering"
965 : "Shutting down";
966 var chatTask = Chat.QueueWatchdogMessage(
967 $"Monitor crashed, this should NEVER happen! Please report this, full details in logs! {nextActionMessage}. Error: {e.Message}",
968 cancellationToken);
969
970 if (disposed)
971 nextAction = MonitorAction.Exit;
972 else if (nextAction != MonitorAction.Exit)
973 {
974 if (GetActiveController()?.Lifetime.IsCompleted != true)
975 await MonitorRestart(cancellationToken);
976 else
977 Logger.LogDebug("Server seems to be okay, not restarting");
978 nextAction = MonitorAction.Continue;
979 }

1055 {

1058 var response = await activeServer.SendCommand(new TopicParameters(), cancellationToken);
1059

1082 actionToTake,
1083 StringComparison.Ordinal),

1085 break;
1086 case 4:
1087 var actionTaken = shouldShutdown
1088 ? "Shutting down due to graceful termination request"
1089 : "Restarting";
1090 const string logTemplate2 = "DEFCON 1: Four heartbeats have been missed! {actionTaken}...";
1091 Logger.LogWarning(logTemplate2, actionTaken);
1092 await Chat.QueueWatchdogMessage(
1093 logTemplate2.Replace(
1094 "{actionTaken}",
1095 actionTaken,
1096 StringComparison.Ordinal),
1097 cancellationToken);

1100 {
1101 Logger.LogDebug("DumpOnHeartbeatRestart enabled.");
1102 await CreateDump(cancellationToken);
1103 }

1109 default:

1112 }
