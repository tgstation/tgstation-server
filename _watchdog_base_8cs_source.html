251  protected async Task LaunchImplNoLock(bool startMonitor, bool announce, WatchdogReattachInformation reattachInfo, CancellationToken cancellationToken)
252  {
253  Logger.LogTrace("Begin LaunchNoLock");
254 
255  if (Running)
256  throw new JobException("Watchdog already running!");
257 
258  // this is necessary, the monitor could be in it's sleep loop trying to restart, if so cancel THAT monitor and start our own with blackjack and hookers
259  Task chatTask;
260  if (startMonitor && await StopMonitor().ConfigureAwait(false))
261  chatTask = Chat.SendWatchdogMessage("Automatic retry sequence cancelled by manual launch. Restarting...", cancellationToken);
262  else if (announce)
263  chatTask = Chat.SendWatchdogMessage(reattachInfo == null ? "Starting..." : "Reattaching...", cancellationToken); // simple announce
264  else
265  chatTask = Task.CompletedTask; // no announce
266 
267  // since neither server is running, this is safe to do
268  LastLaunchParameters = ActiveLaunchParameters;
269 
270  // for when we call ourself and want to not catch thrown exceptions
271  var ignoreNestedException = false;
272  try
273  {
274  await InitControllers(() => ignoreNestedException = true, chatTask, reattachInfo, cancellationToken).ConfigureAwait(false);
275  await chatTask.ConfigureAwait(false);
276 
277  Logger.LogInformation("Launched servers successfully");
278  Running = true;
279 
280  if (startMonitor)
281  {
282  StartMonitor();
283  }
284  }
285  catch (Exception e)
286  {
287  // don't try to send chat tasks or warning logs if were suppressing exceptions or cancelled
288  if (!ignoreNestedException && !cancellationToken.IsCancellationRequested)
289  {
290  var originalChatTask = chatTask;
291  async Task ChainChatTaskWithErrorMessage()
292  {
293  await originalChatTask.ConfigureAwait(false);
294  await Chat.SendWatchdogMessage("Startup failed!", cancellationToken).ConfigureAwait(false);
295  }
296 
297  chatTask = ChainChatTaskWithErrorMessage();
298  Logger.LogWarning("Failed to start watchdog: {0}", e.ToString());
299  }
300 
301  throw;
302  }
303  finally
304  {
305  // finish the chat task that's in flight
306  try
307  {
308  await chatTask.ConfigureAwait(false);
309  }
310  catch (OperationCanceledException) { }
311  }
312  }
