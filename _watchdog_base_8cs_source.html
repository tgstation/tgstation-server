533 bool startMonitor,
534 bool announce,
535 bool announceFailure,
536 ReattachInformation? reattachInfo,
537 CancellationToken cancellationToken)
538 {
539 Logger.LogTrace("Begin LaunchImplNoLock");
540 if (startMonitor && Status != WatchdogStatus.Offline)
541 throw new JobException(ErrorCode.WatchdogRunning);
542
543 if (reattachInfo == null && !DmbFactory.DmbAvailable)
544 throw new JobException(ErrorCode.WatchdogCompileJobCorrupted);
545
546 // this is necessary, the monitor could be in it's sleep loop trying to restart, if so cancel THAT monitor and start our own with blackjack and hookers
547 var eventTask = ValueTask.CompletedTask;
548 if (announce)
549 {
550 Chat.QueueWatchdogMessage(
551 reattachInfo == null
552 ? "Launching..."
553 : "Reattaching..."); // simple announce
554 if (reattachInfo == null)
555 eventTask = HandleEventImpl(EventType.WatchdogLaunch, Enumerable.Empty<string>(), false, cancellationToken);
556 }
557
558 // since neither server is running, this is safe to do
559 LastLaunchParameters = ActiveLaunchParameters;
560 healthChecksMissed = 0;
561
562 try
563 {
564 await InitController(eventTask, reattachInfo, cancellationToken);
565 }
566 catch (OperationCanceledException ex)
567 {
568 Logger.LogTrace(ex, "Controller initialization cancelled!");
569 throw;
570 }
571 catch (Exception e)
572 {
573 Logger.LogWarning(e, "Failed to start watchdog!");
574 var originalChatTask = eventTask;
575 async ValueTask ChainEventTaskWithErrorMessage()
576 {
577 await originalChatTask;
578 if (announceFailure)
579 Chat.QueueWatchdogMessage("Startup failed!");
580 }
581
582 eventTask = ChainEventTaskWithErrorMessage();
583 throw;
584 }
585 finally
586 {
587 // finish the chat task that's in flight
588 try
589 {
590 await eventTask;
591 }
592 catch (OperationCanceledException ex)
593 {
594 Logger.LogTrace(ex, "Announcement task canceled!");
595 }
596 }
597
598 Logger.LogInformation("Controller(s) initialized successfully");
599
600 if (startMonitor)
601 {
602 monitorCts = new CancellationTokenSource();
603 monitorTask = MonitorLifetimes(monitorCts.Token);
604 }

774
775 await DisposeAndNullControllers(cancellationToken);
776
777 for (var retryAttempts = 1; ; ++retryAttempts)
778 {
779 Status = WatchdogStatus.Restoring;
780 Exception launchException;

840 async Task MonitorLifetimes(CancellationToken cancellationToken)
841 {
842 Logger.LogTrace("Entered MonitorLifetimes");
843 Status = WatchdogStatus.Online;
844 using var cancellationTokenLoggingRegistration = cancellationToken.Register(() => Logger.LogTrace("Monitor cancellationToken triggered"));
845
846 // this function is responsible for calling HandlerMonitorWakeup when necessary and manitaining the MonitorState
847 try
848 {
849 MonitorAction nextAction = MonitorAction.Continue;
850 Task? activeServerLifetime = null,
851 activeServerReboot = null,
852 activeServerStartup = null,
853 serverPrimed = null,
854 activeLaunchParametersChanged = null,
855 newDmbAvailable = null,
856 healthCheck = null;
857 ISessionController? lastController = null;
858 var ranInitialDmbCheck = false;
859 for (ulong iteration = 1; nextAction != MonitorAction.Exit; ++iteration)
860 using (LogContext.PushProperty(SerilogContextHelper.WatchdogMonitorIterationContextProperty, iteration))
861 {
862 var nextMonitorWakeupTcs = new TaskCompletionSource();
863 try
864 {
865 Logger.LogTrace("Iteration {iteration} of monitor loop", iteration);
866 nextAction = MonitorAction.Continue;
867
868 var controller = GetActiveController();
869
870 void UpdateMonitoredTasks()
871 {
872 var sameController = lastController == controller;
873 void TryUpdateTask(ref Task? oldTask, Func<Task> newTaskFactory)
874 {
875 if (sameController && oldTask?.IsCompleted == true)
876 return;
877
878 oldTask = newTaskFactory();
879 }
880
881 controller!.RebootGate = nextMonitorWakeupTcs.Task;
882
883 TryUpdateTask(ref activeServerLifetime, () => controller.Lifetime);
884 TryUpdateTask(ref activeServerReboot, () => controller.OnReboot);
885 TryUpdateTask(ref serverPrimed, () => controller.OnPrime);
886 TryUpdateTask(ref activeServerStartup, () => controller.OnStartup);
887
888 if (!sameController)
889 lastController = controller;
890
891 TryUpdateTask(ref activeLaunchParametersChanged, () => activeParametersUpdated.Task);
892 TryUpdateTask(
893 ref newDmbAvailable,
894 () =>
895 {
896 var result = ranInitialDmbCheck
897 ? DmbFactory.OnNewerDmb
898 : InitialCheckDmbUpdated(controller.CompileJob);
899 ranInitialDmbCheck = true;
900 return result;
901 });
902 }
903
904 if (controller != null)
905 {
906 UpdateMonitoredTasks();
907
908 var healthCheckSeconds = ActiveLaunchParameters.HealthCheckSeconds!.Value;
909 healthCheck = healthCheckSeconds == 0
910 || !controller.DMApiAvailable
911 ? Extensions.TaskExtensions.InfiniteTask
912 : Task.Delay(
913 TimeSpan.FromSeconds(healthCheckSeconds),
914 cancellationToken);
915
916 // cancel waiting if requested
917 var toWaitOn = Task.WhenAny(
918 activeServerLifetime!,
919 activeServerReboot!,
920 activeServerStartup!,
921 healthCheck,
922 newDmbAvailable!,
923 activeLaunchParametersChanged!,
924 serverPrimed!);
925
926 // wait for something to happen
927 await toWaitOn.WaitAsync(cancellationToken);
928 }
929 else
930 {
931 Logger.LogError("Controller was null on monitor wakeup! Attempting restart...");
932 nextAction = MonitorAction.Restart; // excuse me wtf?
933 }
934
935 cancellationToken.ThrowIfCancellationRequested();
936 Logger.LogTrace("Monitor activated");
937
938 // always run HandleMonitorWakeup from the context of the semaphore lock
939 using (await SemaphoreSlimContext.Lock(synchronizationSemaphore, cancellationToken))
940 {
941 // Set this sooner so chat sends don't hold us up
942 if (activeServerLifetime!.IsCompleted)
943 Status = WatchdogStatus.Restoring;
944
945 // multiple things may have happened, handle them one at a time
946 for (var moreActivationsToProcess = true; moreActivationsToProcess && (nextAction == MonitorAction.Continue || nextAction == MonitorAction.Skip);)
947 {
948 MonitorActivationReason activationReason = default; // this will always be assigned before being used
949
950 bool CheckActivationReason(ref Task? task, MonitorActivationReason testActivationReason)
951 {
952 var taskCompleted = task?.IsCompleted == true;
953 task = null;
954 if (nextAction == MonitorAction.Skip)
955 nextAction = MonitorAction.Continue;
956 else if (taskCompleted)
957 {
958 activationReason = testActivationReason;
959 return true;
960 }
961
962 return false;
963 }
964
965 // process the tasks in this order and call HandlerMonitorWakup for each depending on the new monitorState
966 var anyActivation = CheckActivationReason(ref activeServerLifetime, MonitorActivationReason.ActiveServerCrashed)
967 || CheckActivationReason(ref activeServerReboot, MonitorActivationReason.ActiveServerRebooted)
968 || CheckActivationReason(ref newDmbAvailable, MonitorActivationReason.NewDmbAvailable)
969 || CheckActivationReason(ref activeLaunchParametersChanged, MonitorActivationReason.ActiveLaunchParametersUpdated)
970 || CheckActivationReason(ref healthCheck, MonitorActivationReason.HealthCheck)
971 || CheckActivationReason(ref serverPrimed, MonitorActivationReason.ActiveServerPrimed)
972 || CheckActivationReason(ref activeServerStartup, MonitorActivationReason.ActiveServerStartup);
973
974 UpdateMonitoredTasks();
975
976 if (!anyActivation)
