548 Logger.LogTrace("Begin LaunchImplNoLock");

551
552 if (reattachInfo == null && !DmbFactory.DmbAvailable)
553 throw new JobException(ErrorCode.WatchdogCompileJobCorrupted);
554
555 // this is necessary, the monitor could be in it's sleep loop trying to restart, if so cancel THAT monitor and start our own with blackjack and hookers
556 var eventTask = ValueTask.CompletedTask;
557 if (announce)
558 {
559 Chat.QueueWatchdogMessage(
560 reattachInfo == null

565 }
566
567 // since neither server is running, this is safe to do
568 LastLaunchParameters = ActiveLaunchParameters;
569 healthChecksMissed = 0;
570
571 try
572 {
573 await InitController(eventTask, reattachInfo, cancellationToken);
574 }
575 catch (OperationCanceledException ex)

578 throw;
579 }
580 catch (Exception e)

582 Logger.LogWarning(e, "Failed to start watchdog!");

585 {
586 await originalChatTask;
587 if (announceFailure)
588 Chat.QueueWatchdogMessage("Startup failed!");
589 }
590
591 eventTask = ChainEventTaskWithErrorMessage();
592 throw;
593 }
594 finally
595 {
596 // finish the chat task that's in flight
597 try
598 {
599 await eventTask;
600 }

602 {

613 }

847 {

850 using var cancellationTokenLoggingRegistration = cancellationToken.Register(() => Logger.LogTrace("Monitor cancellationToken triggered"));

852 // this function is responsible for calling HandlerMonitorWakeup when necessary and manitaining the MonitorState
853 try
854 {
855 MonitorAction nextAction = MonitorAction.Continue;
856 Task? activeServerLifetime = null,
857 activeServerReboot = null,
858 activeServerStartup = null,
859 serverPrimed = null,
860 activeLaunchParametersChanged = null,
861 newDmbAvailable = null,
862 healthCheck = null;
863 ISessionController? lastController = null;
864 var ranInitialDmbCheck = false;
865 for (ulong iteration = 1; nextAction != MonitorAction.Exit; ++iteration)
866 using (LogContext.PushProperty(SerilogContextHelper.WatchdogMonitorIterationContextProperty, iteration))
867 {
868 var nextMonitorWakeupTcs = new TaskCompletionSource();
869 try
870 {
871 Logger.LogTrace("Iteration {iteration} of monitor loop", iteration);
872 nextAction = MonitorAction.Continue;
873
874 var controller = GetActiveController();
875
876 void UpdateMonitoredTasks()
877 {
878 var sameController = lastController == controller;
879 void TryUpdateTask(ref Task? oldTask, Func<Task> newTaskFactory)
880 {
881 if (sameController && oldTask?.IsCompleted == true)
882 return;
883
884 oldTask = newTaskFactory();
885 }
886
887 controller!.RebootGate = nextMonitorWakeupTcs.Task;
888
889 TryUpdateTask(ref activeServerLifetime, () => controller.Lifetime);
890 TryUpdateTask(ref activeServerReboot, () => controller.OnReboot);
891 TryUpdateTask(ref serverPrimed, () => controller.OnPrime);
892 TryUpdateTask(ref activeServerStartup, () => controller.OnStartup);
893
894 if (!sameController)
895 lastController = controller;
896
897 TryUpdateTask(ref activeLaunchParametersChanged, () => activeParametersUpdated.Task);
898 TryUpdateTask(
899 ref newDmbAvailable,
900 () =>
901 {
902 var result = ranInitialDmbCheck
903 ? DmbFactory.OnNewerDmb
904 : InitialCheckDmbUpdated(controller.CompileJob);
905 ranInitialDmbCheck = true;
906 return result;
907 });
908 }
909
910 if (controller != null)
911 {
912 UpdateMonitoredTasks();
913
914 var healthCheckSeconds = ActiveLaunchParameters.HealthCheckSeconds!.Value;
915 healthCheck = healthCheckSeconds == 0
916 || !controller.DMApiAvailable
917 ? Extensions.TaskExtensions.InfiniteTask
918 : Task.Delay(
919 TimeSpan.FromSeconds(healthCheckSeconds),
920 cancellationToken);
921
922 // cancel waiting if requested
923 var toWaitOn = Task.WhenAny(
924 activeServerLifetime!,
925 activeServerReboot!,
926 activeServerStartup!,
927 healthCheck,
928 newDmbAvailable!,
929 activeLaunchParametersChanged!,
930 serverPrimed!);
931
932 // wait for something to happen
933 await toWaitOn.WaitAsync(cancellationToken);
934 }
935 else
936 {
937 Logger.LogError("Controller was null on monitor wakeup! Attempting restart...");
938 nextAction = MonitorAction.Restart; // excuse me wtf?
939 }

941 cancellationToken.ThrowIfCancellationRequested();
942 Logger.LogTrace("Monitor activated");
943
944 // always run HandleMonitorWakeup from the context of the semaphore lock
945 using (await SemaphoreSlimContext.Lock(synchronizationSemaphore, cancellationToken))
946 {
947 // Set this sooner so chat sends don't hold us up
948 if (activeServerLifetime!.IsCompleted)
949 Status = WatchdogStatus.Restoring;
950
951 // multiple things may have happened, handle them one at a time
952 for (var moreActivationsToProcess = true; moreActivationsToProcess && (nextAction == MonitorAction.Continue || nextAction == MonitorAction.Skip);)
953 {
954 MonitorActivationReason activationReason = default; // this will always be assigned before being used
955
956 bool CheckActivationReason(ref Task? task, MonitorActivationReason testActivationReason)
957 {
958 var taskCompleted = task?.IsCompleted == true;
959 task = null;
960 if (nextAction == MonitorAction.Skip)
961 nextAction = MonitorAction.Continue;
962 else if (taskCompleted)
963 {
964 activationReason = testActivationReason;
965 return true;
966 }
967
968 return false;
969 }
970
971 // process the tasks in this order and call HandlerMonitorWakup for each depending on the new monitorState
972 var anyActivation = CheckActivationReason(ref activeServerLifetime, MonitorActivationReason.ActiveServerCrashed)
973 || CheckActivationReason(ref activeServerReboot, MonitorActivationReason.ActiveServerRebooted)
