534 bool startMonitor,
535 bool announce,
536 bool announceFailure,
537 ReattachInformation? reattachInfo,
538 CancellationToken cancellationToken)
539 {
540 Logger.LogTrace("Begin LaunchImplNoLock");
541 if (startMonitor && Status != WatchdogStatus.Offline)
542 throw new JobException(ErrorCode.WatchdogRunning);
543
544 if (reattachInfo == null && !DmbFactory.DmbAvailable)
545 throw new JobException(ErrorCode.WatchdogCompileJobCorrupted);
546
547 // this is necessary, the monitor could be in it's sleep loop trying to restart, if so cancel THAT monitor and start our own with blackjack and hookers
548 var eventTask = ValueTask.CompletedTask;
549 if (announce)
550 {
551 Chat.QueueWatchdogMessage(
552 reattachInfo == null
553 ? "Launching..."
554 : "Reattaching..."); // simple announce
555 if (reattachInfo == null)
556 eventTask = HandleEventImpl(EventType.WatchdogLaunch, Enumerable.Empty<string>(), false, cancellationToken);
557 }
558
559 // since neither server is running, this is safe to do
560 LastLaunchParameters = ActiveLaunchParameters;
561 healthChecksMissed = 0;
562
563 try
564 {
565 await InitController(eventTask, reattachInfo, cancellationToken);
566 }
567 catch (OperationCanceledException ex)
568 {
569 Logger.LogTrace(ex, "Controller initialization cancelled!");
570 throw;
571 }
572 catch (Exception e)
573 {
574 Logger.LogWarning(e, "Failed to start watchdog!");
575 var originalChatTask = eventTask;
576 async ValueTask ChainEventTaskWithErrorMessage()
577 {
578 await originalChatTask;
579 if (announceFailure)
580 Chat.QueueWatchdogMessage("Startup failed!");
581 }
582
583 eventTask = ChainEventTaskWithErrorMessage();
584 throw;
585 }
586 finally
587 {
588 // finish the chat task that's in flight
589 try
590 {
591 await eventTask;
592 }
593 catch (OperationCanceledException ex)
594 {
595 Logger.LogTrace(ex, "Announcement task canceled!");
596 }
597 }
598
599 Logger.LogInformation("Controller(s) initialized successfully");
600
601 if (startMonitor)
602 {
603 monitorCts = new CancellationTokenSource();
604 monitorTask = MonitorLifetimes(monitorCts.Token);
605 }

839 {
840 Logger.LogTrace("Entered MonitorLifetimes");
841 Status = WatchdogStatus.Online;
842 using var cancellationTokenLoggingRegistration = cancellationToken.Register(() => Logger.LogTrace("Monitor cancellationToken triggered"));
843
844 // this function is responsible for calling HandlerMonitorWakeup when necessary and manitaining the MonitorState
845 try
846 {
847 MonitorAction nextAction = MonitorAction.Continue;
848 Task? activeServerLifetime = null,
849 activeServerReboot = null,
850 activeServerStartup = null,
851 serverPrimed = null,
852 activeLaunchParametersChanged = null,
853 newDmbAvailable = null,
854 healthCheck = null;
855 ISessionController? lastController = null;
856 var ranInitialDmbCheck = false;
857 for (ulong iteration = 1; nextAction != MonitorAction.Exit; ++iteration)
858 using (LogContext.PushProperty(SerilogContextHelper.WatchdogMonitorIterationContextProperty, iteration))
859 {
860 var nextMonitorWakeupTcs = new TaskCompletionSource();
861 try
862 {
863 Logger.LogTrace("Iteration {iteration} of monitor loop", iteration);
864 nextAction = MonitorAction.Continue;
865
866 var controller = GetActiveController();
867
868 void UpdateMonitoredTasks()
869 {
870 var sameController = lastController == controller;
871 void TryUpdateTask(ref Task? oldTask, Func<Task> newTaskFactory)
872 {
873 if (sameController && oldTask?.IsCompleted == true)
874 return;
875
876 oldTask = newTaskFactory();
877 }
878
879 controller!.RebootGate = nextMonitorWakeupTcs.Task;
880
881 TryUpdateTask(ref activeServerLifetime, () => controller.Lifetime);
882 TryUpdateTask(ref activeServerReboot, () => controller.OnReboot);
883 TryUpdateTask(ref serverPrimed, () => controller.OnPrime);
884 TryUpdateTask(ref activeServerStartup, () => controller.OnStartup);
885
886 if (!sameController)
887 lastController = controller;
888
889 TryUpdateTask(ref activeLaunchParametersChanged, () => activeParametersUpdated.Task);
890 TryUpdateTask(
891 ref newDmbAvailable,
892 () =>
893 {
894 var result = ranInitialDmbCheck
895 ? DmbFactory.OnNewerDmb
896 : InitialCheckDmbUpdated(controller.CompileJob);
897 ranInitialDmbCheck = true;
898 return result;
899 });
900 }
901
902 if (controller != null)
903 {
904 UpdateMonitoredTasks();
905
906 var healthCheckSeconds = ActiveLaunchParameters.HealthCheckSeconds!.Value;
907 healthCheck = healthCheckSeconds == 0
908 || !controller.DMApiAvailable
909 ? Extensions.TaskExtensions.InfiniteTask
910 : Task.Delay(
911 TimeSpan.FromSeconds(healthCheckSeconds),
912 cancellationToken);
913
914 // cancel waiting if requested
915 var toWaitOn = Task.WhenAny(
916 activeServerLifetime!,
917 activeServerReboot!,
918 activeServerStartup!,
919 healthCheck,
920 newDmbAvailable!,
921 activeLaunchParametersChanged!,
922 serverPrimed!);
923
924 // wait for something to happen
925 await toWaitOn.WaitAsync(cancellationToken);
926 }
927 else
928 {
929 Logger.LogError("Controller was null on monitor wakeup! Attempting restart...");
930 nextAction = MonitorAction.Restart; // excuse me wtf?
931 }

933 cancellationToken.ThrowIfCancellationRequested();
934 Logger.LogTrace("Monitor activated");
935
936 // always run HandleMonitorWakeup from the context of the semaphore lock
937 using (await SemaphoreSlimContext.Lock(synchronizationSemaphore, cancellationToken))
938 {
939 // Set this sooner so chat sends don't hold us up
940 if (activeServerLifetime!.IsCompleted)
941 Status = WatchdogStatus.Restoring;
942
943 // multiple things may have happened, handle them one at a time
944 for (var moreActivationsToProcess = true; moreActivationsToProcess && (nextAction == MonitorAction.Continue || nextAction == MonitorAction.Skip);)
945 {
946 MonitorActivationReason activationReason = default; // this will always be assigned before being used
947
948 bool CheckActivationReason(ref Task? task, MonitorActivationReason testActivationReason)
949 {
950 var taskCompleted = task?.IsCompleted == true;
951 task = null;
952 if (nextAction == MonitorAction.Skip)
953 nextAction = MonitorAction.Continue;
954 else if (taskCompleted)
955 {
956 activationReason = testActivationReason;
957 return true;
958 }
959
960 return false;
961 }
962
963 // process the tasks in this order and call HandlerMonitorWakup for each depending on the new monitorState
964 var anyActivation = CheckActivationReason(ref activeServerLifetime, MonitorActivationReason.ActiveServerCrashed)
965 || CheckActivationReason(ref activeServerReboot, MonitorActivationReason.ActiveServerRebooted)
966 || CheckActivationReason(ref newDmbAvailable, MonitorActivationReason.NewDmbAvailable)
967 || CheckActivationReason(ref activeLaunchParametersChanged, MonitorActivationReason.ActiveLaunchParametersUpdated)
968 || CheckActivationReason(ref healthCheck, MonitorActivationReason.HealthCheck)
969 || CheckActivationReason(ref serverPrimed, MonitorActivationReason.ActiveServerPrimed)
970 || CheckActivationReason(ref activeServerStartup, MonitorActivationReason.ActiveServerStartup);
971
972 UpdateMonitoredTasks();
