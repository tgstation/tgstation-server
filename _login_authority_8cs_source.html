134 public async ValueTask<AuthorityResponse<LoginPayload>> AttemptLogin(CancellationToken cancellationToken)
135 {
136 var headers = apiHeadersProvider.ApiHeaders;
137 if (headers == null)
138 return GenerateHeadersExceptionResponse(apiHeadersProvider.HeadersException!);
139
140 if (headers.IsTokenAuthentication)
141 return BadRequest<LoginPayload>(ErrorCode.TokenWithToken);
142
143 var oAuthLogin = headers.OAuthProvider.HasValue;
144
145 ISystemIdentity? systemIdentity = null;
146 if (!oAuthLogin)
147 try
148 {
149 // trust the system over the database because a user's name can change while still having the same SID
150 systemIdentity = await systemIdentityFactory.CreateSystemIdentity(headers.Username!, headers.Password!, cancellationToken);
151 }
152 catch (NotImplementedException)
153 {
154 // Intentionally suppressed
155 }
156
157 using (systemIdentity)
158 {
159 // Get the user from the database
160 IQueryable<User> query = DatabaseContext.Users.AsQueryable();
161 if (oAuthLogin)
162 {
163 var oAuthProvider = headers.OAuthProvider!.Value;
164 string? externalUserId;
165 try
166 {
167 var validator = oAuthProviders
168 .GetValidator(oAuthProvider);
169
170 if (validator == null)
171 return BadRequest<LoginPayload>(ErrorCode.OAuthProviderDisabled);
172
173 externalUserId = await validator
174 .ValidateResponseCode(headers.OAuthCode!, cancellationToken);
175
176 Logger.LogTrace("External {oAuthProvider} UID: {externalUserId}", oAuthProvider, externalUserId);
177 }
178 catch (Octokit.RateLimitExceededException ex)
179 {
180 return RateLimit<LoginPayload>(ex);
181 }
182
183 if (externalUserId == null)
184 return Unauthorized<LoginPayload>();
185
186 query = query.Where(
187 x => x.OAuthConnections!.Any(
188 y => y.Provider == oAuthProvider
189 && y.ExternalUserId == externalUserId));
190 }
191 else
192 {
193 var canonicalUserName = User.CanonicalizeName(headers.Username!);
194 if (canonicalUserName == User.CanonicalizeName(User.TgsSystemUserName))
195 return Unauthorized<LoginPayload>();
196
197 if (systemIdentity == null)
198 query = query.Where(x => x.CanonicalName == canonicalUserName);
199 else
200 query = query.Where(x => x.CanonicalName == canonicalUserName || x.SystemIdentifier == systemIdentity.Uid);
201 }
202
203 var user = await SelectUserInfoFromQuery(query, cancellationToken);
204
205 // No user? You're not allowed
206 if (user == null)
207 return Unauthorized<LoginPayload>();
208
209 // A system user may have had their name AND password changed to one in our DB...
210 // Or a DB user was created that had the same user/pass as a system user
211 // Dumb admins...
212 // FALLBACK TO THE DB USER HERE, DO NOT REVEAL A SYSTEM LOGIN!!!
213 // This of course, allows system users to discover TGS users in this (HIGHLY IMPROBABLE) case but that is not our fault
214 var originalHash = user.PasswordHash;
215 var isLikelyDbUser = originalHash != null;
216 var usingSystemIdentity = systemIdentity != null && !isLikelyDbUser;
217 if (!oAuthLogin)
218 if (!usingSystemIdentity)
219 {
220 // DB User password check and update
221 if (!isLikelyDbUser || !cryptographySuite.CheckUserPassword(user, headers.Password!))
222 return Unauthorized<LoginPayload>();
223 if (user.PasswordHash != originalHash)
224 {
225 Logger.LogDebug("User ID {userId}'s password hash needs a refresh, updating database.", user.Id);
226 var updatedUser = new User
227 {
228 Id = user.Id,
229 };
230 DatabaseContext.Users.Attach(updatedUser);
231 updatedUser.PasswordHash = user.PasswordHash;
232 await DatabaseContext.Save(cancellationToken);
233 }
234 }
235 else
236 {
237 var usernameMismatch = systemIdentity!.Username != user.Name;
238 if (isLikelyDbUser || usernameMismatch)
239 {
240 DatabaseContext.Users.Attach(user);
241 if (usernameMismatch)
242 {
243 // System identity username change update
244 Logger.LogDebug("User ID {userId}'s system identity needs a refresh, updating database.", user.Id);
245 user.Name = systemIdentity.Username;
246 user.CanonicalName = User.CanonicalizeName(user.Name);
247 }
248
249 if (isLikelyDbUser)
250 {
251 // cleanup from https://github.com/tgstation/tgstation-server/issues/1528
252 Logger.LogDebug("System user ID {userId}'s PasswordHash is polluted, updating database.", user.Id);
253 user.PasswordHash = null;
254 sessionInvalidationTracker.UserModifiedInvalidateSessions(user);
255 }
256
257 await DatabaseContext.Save(cancellationToken);
258 }
259 }
260
261 // Now that the bookeeping is done, tell them to fuck off if necessary
262 if (!user.Enabled!.Value)
263 {
264 Logger.LogTrace("Not logging in disabled user {userId}.", user.Id);
265 return Forbid<LoginPayload>();
266 }
267
268 var token = tokenFactory.CreateToken(user, oAuthLogin);
269 var payload = new LoginPayload
270 {
271 Bearer = token,
272 User = ((IApiTransformable<User, GraphQL.Types.User, UserGraphQLTransformer>)user).ToApi(),
273 };
274
275 if (usingSystemIdentity)
276 await CacheSystemIdentity(systemIdentity!, user, payload);
277
278 Logger.LogDebug("Successfully logged in user {userId}!", user.Id);
279
280 return new AuthorityResponse<LoginPayload>(payload);
281 }
282 }

291 private async ValueTask CacheSystemIdentity(ISystemIdentity systemIdentity, User user, LoginPayload loginPayload)
292 {
293 // expire the identity slightly after the auth token in case of lag

295 identExpiry += tokenFactory.ValidationParameters.ClockSkew;
