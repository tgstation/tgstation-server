220 public async ValueTask ScheduleAutoUpdate(uint newInterval, string? newCron)
221 {
222 if (newInterval > 0 && !String.IsNullOrWhiteSpace(newCron))
223 throw new ArgumentException("Only one of newInterval and newCron may be set!");
224
225 Task toWait;
226 lock (timerLock)
227 if (timerTask != null)
228 {
229 logger.LogTrace("Cancelling auto-update task");
230 timerCts!.Cancel();
231 timerCts.Dispose();
232 toWait = timerTask;
233 timerTask = null;
234 timerCts = null;
235 }
236 else
237 toWait = Task.CompletedTask;
238
239 await toWait;
240 if (newInterval == 0 && String.IsNullOrWhiteSpace(newCron))
241 {
242 logger.LogTrace("Auto-update disabled 0. Not starting task.");
243 return;
244 }
245
246 lock (timerLock)
247 {
248 // race condition, just quit
249 if (timerTask != null)
250 {
251 logger.LogWarning("Aborting auto-update scheduling change due to race condition!");
252 return;
253 }
254
255 timerCts = new CancellationTokenSource();
256 timerTask = TimerLoop(newInterval, newCron, timerCts.Token);
257 }

496 async Task TimerLoop(uint minutes, string? cron, CancellationToken cancellationToken)
497 {
498 logger.LogDebug("Entering auto-update loop");
499 while (true)
500 try
501 {
502 TimeSpan delay;
503 if (!String.IsNullOrWhiteSpace(cron))
504 {
505 logger.LogTrace("Using cron schedule: {cron}", cron);
506 var schedule = CrontabSchedule.Parse(
507 cron,
508 new CrontabSchedule.ParseOptions
509 {
510 IncludingSeconds = true,
511 });
512 var now = DateTime.UtcNow;
513 var nextOccurrence = schedule.GetNextOccurrence(now);
514 delay = nextOccurrence - now;
515 }
516 else
517 {
518 logger.LogTrace("Using interval: {interval}m", minutes);
519
520 delay = TimeSpan.FromMinutes(minutes);
521 }
522
523 logger.LogInformation("Next auto-update will occur at {time}", DateTimeOffset.UtcNow + delay);
524
525 // https://learn.microsoft.com/en-us/dotnet/api/system.threading.tasks.task.delay?view=net-8.0#system-threading-tasks-task-delay(system-timespan)
526 const uint DelayMinutesLimit = UInt32.MaxValue - 1;
527 Debug.Assert(DelayMinutesLimit == 4294967294, "Delay limit assertion failure!");
528
529 var maxDelayIterations = 0UL;
530 if (delay.TotalMilliseconds >= UInt32.MaxValue)
531 {
532 maxDelayIterations = (ulong)Math.Floor(delay.TotalMilliseconds / DelayMinutesLimit);
533 logger.LogDebug("Breaking interval into {iterationCount} iterations", maxDelayIterations + 1);
534 delay = TimeSpan.FromMilliseconds(delay.TotalMilliseconds - (maxDelayIterations * DelayMinutesLimit));
535 }
536
537 if (maxDelayIterations > 0)
538 {
539 var longDelayTimeSpan = TimeSpan.FromMilliseconds(DelayMinutesLimit);
540 for (var i = 0UL; i < maxDelayIterations; ++i)
541 {
542 logger.LogTrace("Long delay #{iteration}...", i + 1);
543 await asyncDelayer.Delay(longDelayTimeSpan, cancellationToken);
544 }
545
546 logger.LogTrace("Final delay iteration #{iteration}...", maxDelayIterations + 1);
547 }
548
549 await asyncDelayer.Delay(delay, cancellationToken);
550 logger.LogInformation("Beginning auto update...");
551 await eventConsumer.HandleEvent(EventType.InstanceAutoUpdateStart, Enumerable.Empty<string>(), true, cancellationToken);
552
553 var repositoryUpdateJob = Job.Create(Api.Models.JobCode.RepositoryAutoUpdate, null, metadata, RepositoryRights.CancelPendingChanges);
554 await jobManager.RegisterOperation(
555 repositoryUpdateJob,
556 RepositoryAutoUpdateJob,
557 cancellationToken);
558
559 var repoUpdateJobResult = await jobManager.WaitForJobCompletion(repositoryUpdateJob, null, cancellationToken, cancellationToken);
560 if (repoUpdateJobResult == false)
561 {
562 logger.LogWarning("Aborting auto-update due to repository update error!");
563 continue;
564 }
565
566 Job compileProcessJob;
567 using (var repo = await RepositoryManager.LoadRepository(cancellationToken))
568 {
569 if (repo == null)
570 throw new JobException(Api.Models.ErrorCode.RepoMissing);
571
572 var deploySha = repo.Head;
573 if (deploySha == null)
574 {
575 logger.LogTrace("Aborting auto update, repository error!");
576 continue;
577 }
578
579 if (deploySha == (await LatestCompileJob())?.RevisionInformation.CommitSha)
580 {
581 logger.LogTrace("Aborting auto update, same revision as latest CompileJob");
582 continue;
583 }
584
585 // finally set up the job
586 compileProcessJob = Job.Create(Api.Models.JobCode.AutomaticDeployment, null, metadata, DreamMakerRights.CancelCompile);
587 await jobManager.RegisterOperation(
588 compileProcessJob,
589 (core, databaseContextFactory, job, progressReporter, jobCancellationToken) =>
590 {
591 if (core != this)
592 throw new InvalidOperationException(DifferentCoreExceptionMessage);
593 return DreamMaker.DeploymentProcess(
594 job,
595 databaseContextFactory,
596 progressReporter,
597 jobCancellationToken);
598 },
599 cancellationToken);
600 }
601
602 await jobManager.WaitForJobCompletion(compileProcessJob, null, default, cancellationToken);
603 }
604 catch (OperationCanceledException)
605 {
606 logger.LogDebug("Cancelled auto update loop!");
607 break;
608 }
609 catch (Exception e)
610 {
611 logger.LogError(e, "Error in auto update loop!");
612 continue;
613 }
614
615 logger.LogTrace("Leaving auto update loop...");
616 }
