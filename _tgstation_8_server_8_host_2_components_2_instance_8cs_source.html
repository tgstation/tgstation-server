216 public async ValueTask SetAutoUpdateInterval(uint newInterval)
217 {
218 Task toWait;
219 lock (timerLock)
220 {
221 if (timerTask != null)
222 {
223 logger.LogTrace("Cancelling auto-update task");
224 timerCts!.Cancel();
225 timerCts.Dispose();
226 toWait = timerTask;
227 timerTask = null;
228 timerCts = null;
229 }
230 else
231 toWait = Task.CompletedTask;
232 }
233
234 await toWait;
235 if (newInterval == 0)
236 {
237 logger.LogTrace("New auto-update interval is 0. Not starting task.");
238 return;
239 }
240
241 lock (timerLock)
242 {
243 // race condition, just quit
244 if (timerTask != null)
245 {
246 logger.LogWarning("Aborting auto update interval change due to race condition!");
247 return;
248 }
249
250 timerCts = new CancellationTokenSource();
251 timerTask = TimerLoop(newInterval, timerCts.Token);
252 }

321 {

325 currentRevInfo = await databaseContext

327 .AsQueryable()
328 .Where(x => x.CommitSha == startSha && x.InstanceId == metadata.Id)
329 .Include(x => x.ActiveTestMerges!)
330 .ThenInclude(x => x.TestMerge)
331 .FirstOrDefaultAsync(cancellationToken);
332 }
333
334 if (currentRevInfo == default)
335 {
336 logger.LogInformation(Repository.Repository.OriginTrackingErrorTemplate, currentHead);
337 onOrigin = true;
338 }
339 else if (currentRevInfo.CommitSha == currentHead)
340 {
341 logger.LogTrace("Not updating rev-info, already in DB.");
342 return;
343 }
344
345 if (attachedInstance == null)
346 {
347 attachedInstance = new Models.Instance

355 currentRevInfo = new RevisionInformation

357 CommitSha = currentHead,
358 Timestamp = await repo.TimestampCommit(currentHead, cancellationToken),

361 : await repo.GetOriginSha(cancellationToken),
362 Instance = attachedInstance,
363 };
364
365 if (!onOrigin)

380 await UpdateRevInfo(repo.Head, false, null);

382 var preserveTestMerges = repositorySettings.AutoUpdatesKeepTestMerges!.Value;

396 repositorySettings.CommitterEmail!,
397 true,
398 cancellationToken);
399
400 // take appropriate auto update actions
401 var shouldSyncTracked = false;
402 if (result.HasValue)
403 {
404 if (updatedTestMerges.Count == 0)

406 logger.LogTrace("All test merges have been merged on remote");

420 shouldSyncTracked = stillOnOrigin;

422 }

434 repositorySettings.AccessToken,

436 true,
437 cancellationToken);
438
439 var currentHead = repo.Head;
440
441 currentRevInfo = await databaseContext.RevisionInformations
442 .AsQueryable()
443 .Where(x => x.CommitSha == currentHead && x.InstanceId == metadata.Id)
444 .FirstOrDefaultAsync(cancellationToken);
445
446 if (currentHead != startSha && currentRevInfo == default)
447 await UpdateRevInfo(currentHead, true, null);

450 }

452 // synch if necessary
453 if (repositorySettings.AutoUpdatesSynchronize!.Value && startSha != repo.Head && (shouldSyncTracked || repositorySettings.PushTestMergeCommits!.Value))

462 true,
463 cancellationToken);
464 var currentHead = repo.Head;
465 if (currentHead != currentRevInfo!.CommitSha)
466 await UpdateRevInfo(currentHead, pushedOrigin, null);
467 }
468
469 if (hasDbChanges)
470 try
471 {
472 await databaseContext.Save(cancellationToken);
473 }
474 catch

490 async Task TimerLoop(uint minutes, CancellationToken cancellationToken)
491 {
492 logger.LogDebug("Entering auto-update loop");
493 while (true)
494 try
495 {
496 await asyncDelayer.Delay(TimeSpan.FromMinutes(minutes > Int32.MaxValue ? Int32.MaxValue : minutes), cancellationToken);
497 logger.LogInformation("Beginning auto update...");
498 await eventConsumer.HandleEvent(EventType.InstanceAutoUpdateStart, Enumerable.Empty<string>(), true, cancellationToken);
499 try
500 {
501 var repositoryUpdateJob = Job.Create(Api.Models.JobCode.RepositoryAutoUpdate, null, metadata, RepositoryRights.CancelPendingChanges);
502 await jobManager.RegisterOperation(
503 repositoryUpdateJob,
504 RepositoryAutoUpdateJob,
505 cancellationToken);
506
507 var repoUpdateJobResult = await jobManager.WaitForJobCompletion(repositoryUpdateJob, null, cancellationToken, cancellationToken);
508 if (repoUpdateJobResult == false)
509 {
510 logger.LogWarning("Aborting auto-update due to repository update error!");
511 continue;
512 }
513
514 Job compileProcessJob;
515 using (var repo = await RepositoryManager.LoadRepository(cancellationToken))
516 {
517 if (repo == null)
518 throw new JobException(Api.Models.ErrorCode.RepoMissing);
519
520 var deploySha = repo.Head;
521 if (deploySha == null)
522 {
523 logger.LogTrace("Aborting auto update, repository error!");
524 continue;
525 }
526
527 if (deploySha == LatestCompileJob()?.RevisionInformation.CommitSha)
528 {
529 logger.LogTrace("Aborting auto update, same revision as latest CompileJob");
530 continue;
531 }
532
533 // finally set up the job
534 compileProcessJob = Job.Create(Api.Models.JobCode.AutomaticDeployment, null, metadata, DreamMakerRights.CancelCompile);
535 await jobManager.RegisterOperation(
536 compileProcessJob,
537 (core, databaseContextFactory, job, progressReporter, jobCancellationToken) =>
538 {
539 if (core != this)
540 throw new InvalidOperationException(DifferentCoreExceptionMessage);
541 return DreamMaker.DeploymentProcess(
542 job,
543 databaseContextFactory,
544 progressReporter,
545 jobCancellationToken);
546 },
547 cancellationToken);
548 }
549
550 await jobManager.WaitForJobCompletion(compileProcessJob, null, default, cancellationToken);
551 }
552 catch (Exception e) when (e is not OperationCanceledException)
553 {
554 logger.LogWarning(e, "Error in auto update loop!");
555 continue;
556 }
557 }
558 catch (OperationCanceledException)
559 {
560 logger.LogDebug("Cancelled auto update loop!");
561 break;
562 }
563
564 logger.LogTrace("Leaving auto update loop...");
565 }
