214 public async Task SetAutoUpdateInterval(uint newInterval)
215 {
216 Task toWait;
217 lock (timerLock)
218 {
219 if (timerTask != null)
220 {
221 logger.LogTrace("Cancelling auto-update task");
222 timerCts.Cancel();
223 timerCts.Dispose();
224 toWait = timerTask;
225 timerTask = null;
226 timerCts = null;
227 }
228 else
229 toWait = Task.CompletedTask;
230 }
231
232 await toWait;
233 if (newInterval == 0)
234 {
235 logger.LogTrace("New auto-update interval is 0. Not starting task.");
236 return;
237 }
238
239 lock (timerLock)
240 {
241 // race condition, just quit
242 if (timerTask != null)
243 {
244 logger.LogWarning("Aborting auto update interval change due to race condition!");
245 return;
246 }
247
248 timerCts = new CancellationTokenSource();
249 timerTask = TimerLoop(newInterval, timerCts.Token);
250 }

301 logger.LogTrace("Aborting repo update, active ref not tracking any remote branch!");
302 return;
303 }
304

306

318 async Task UpdateRevInfo(string currentHead, bool onOrigin, IEnumerable<TestMerge> updatedTestMerges)
319 {
320 if (currentRevInfo == null)
321 {
322 logger.LogTrace("Loading revision info for commit {sha}...", startSha[..7]);
323 currentRevInfo = await databaseContext
324 .RevisionInformations
325 .AsQueryable()
326 .Where(x => x.CommitSha == startSha && x.Instance.Id == metadata.Id)
327 .Include(x => x.ActiveTestMerges)
328 .ThenInclude(x => x.TestMerge)
329 .FirstOrDefaultAsync(cancellationToken);
330 }
331
332 if (currentRevInfo == default)
333 {
334 logger.LogInformation(Repository.Repository.OriginTrackingErrorTemplate, currentHead);
335 onOrigin = true;
336 }
337 else if (currentRevInfo.CommitSha == currentHead)
338 {
339 logger.LogTrace("Not updating rev-info, already in DB.");
340 return;
341 }
342
343 if (attachedInstance == null)
344 {
345 attachedInstance = new Models.Instance
346 {
347 Id = metadata.Id,
348 };
349 databaseContext.Instances.Attach(attachedInstance);
350 }
351
352 var oldRevInfo = currentRevInfo;
353 currentRevInfo = new RevisionInformation
354 {
355 CommitSha = currentHead,
356 Timestamp = await repo.TimestampCommit(currentHead, cancellationToken),
357 OriginCommitSha = onOrigin
358 ? currentHead
359 : await repo.GetOriginSha(cancellationToken),
360 Instance = attachedInstance,
361 };
362
363 if (!onOrigin)
364 {
365 var testMerges = updatedTestMerges ?? oldRevInfo.ActiveTestMerges.Select(x => x.TestMerge);
366 var revInfoTestMerges = testMerges.Select(
367 testMerge => new RevInfoTestMerge

369 TestMerge = testMerge,
370 })

372
373 currentRevInfo.ActiveTestMerges = revInfoTestMerges;
374 }
375
376 databaseContext.RevisionInformations.Add(currentRevInfo);

378 }
379
380 // build current commit data if it's missing
381 await UpdateRevInfo(repo.Head, false, null);
382
383 var result = await repo.MergeOrigin(
384 NextProgressReporter("Merge Origin"),
385 repositorySettings.CommitterName,
386 repositorySettings.CommitterEmail,
387 true,
388 cancellationToken);
389
390 var preserveTestMerges = repositorySettings.AutoUpdatesKeepTestMerges.Value;
391 var remoteDeploymentManager = remoteDeploymentManagerFactory.CreateRemoteDeploymentManager(
392 metadata,
393 repo.RemoteGitProvider.Value);
394
395 // take appropriate auto update actions
396 var shouldSyncTracked = false;
397 if (result.HasValue)
398 {
399 var updatedTestMerges = await remoteDeploymentManager.RemoveMergedTestMerges(
400 repo,
401 repositorySettings,
402 currentRevInfo,
403 cancellationToken);
404
405 if (updatedTestMerges.Count == 0)
406 {
407 logger.LogTrace("All test merges have been merged on remote");

409 }
410 else
411 {
412 var lastRevInfoWasOriginCommit =
413 currentRevInfo == default
414 || currentRevInfo.CommitSha == currentRevInfo.OriginCommitSha;
415 var stillOnOrigin = result.Value && lastRevInfoWasOriginCommit;
416
417 var currentHead = repo.Head;
418 if (currentHead != startSha)
419 {
420 await UpdateRevInfo(currentHead, stillOnOrigin, updatedTestMerges);
421 shouldSyncTracked = stillOnOrigin;
422 }
423 }
424 }
425 else if (preserveTestMerges)
426 throw new JobException(Api.Models.ErrorCode.InstanceUpdateTestMergeConflict);
427
428 if (!preserveTestMerges)
429 {
430 const string StageName = "Resetting to origin...";
431 logger.LogTrace(StageName);
432 await repo.ResetToOrigin(
433 NextProgressReporter(StageName),
434 repositorySettings.AccessUser,
435 repositorySettings.AccessToken,
436 repositorySettings.UpdateSubmodules.Value,
437 true,
438 cancellationToken);
439
440 var currentHead = repo.Head;
441
442 currentRevInfo = await databaseContext.RevisionInformations
443 .AsQueryable()
444 .Where(x => x.CommitSha == currentHead && x.Instance.Id == metadata.Id)
445 .FirstOrDefaultAsync(cancellationToken);
446
447 if (currentHead != startSha && currentRevInfo == default)
448 await UpdateRevInfo(currentHead, true, null);
449
450 shouldSyncTracked = true;
451 }
452
453 // synch if necessary
454 if (repositorySettings.AutoUpdatesSynchronize.Value && startSha != repo.Head && (shouldSyncTracked || repositorySettings.PushTestMergeCommits.Value))
455 {
456 var pushedOrigin = await repo.Sychronize(
457 NextProgressReporter("Synchronize"),
458 repositorySettings.AccessUser,
459 repositorySettings.AccessToken,
460 repositorySettings.CommitterName,
461 repositorySettings.CommitterEmail,
462 shouldSyncTracked,
463 true,
464 cancellationToken);
465 var currentHead = repo.Head;
466 if (currentHead != currentRevInfo.CommitSha)
467 await UpdateRevInfo(currentHead, pushedOrigin, null);
468 }
469
470 if (hasDbChanges)
471 try
472 {
473 await databaseContext.Save(cancellationToken);
474 }
475 catch

491 async Task TimerLoop(uint minutes, CancellationToken cancellationToken)
492 {
493 logger.LogDebug("Entering auto-update loop");
494 while (true)
495 try
496 {
497 await asyncDelayer.Delay(TimeSpan.FromMinutes(minutes > Int32.MaxValue ? Int32.MaxValue : minutes), cancellationToken);
498 logger.LogInformation("Beginning auto update...");
499 await eventConsumer.HandleEvent(EventType.InstanceAutoUpdateStart, Enumerable.Empty<string>(), true, cancellationToken);
500 try
501 {
502 var repositoryUpdateJob = new Job
503 {
504 Instance = new Models.Instance
505 {
506 Id = metadata.Id,
507 },
508 Description = "Scheduled repository update",
509 CancelRightsType = RightsType.Repository,
510 CancelRight = (ulong)RepositoryRights.CancelPendingChanges,
511 };
512
513 await jobManager.RegisterOperation(
514 repositoryUpdateJob,
515 RepositoryAutoUpdateJob,
516 cancellationToken);
517
518 await jobManager.WaitForJobCompletion(repositoryUpdateJob, null, cancellationToken, cancellationToken);
519
520 Job compileProcessJob;
521 using (var repo = await RepositoryManager.LoadRepository(cancellationToken))
522 {
523 var deploySha = repo.Head;
524 if (deploySha == null)
525 {
526 logger.LogTrace("Aborting auto update, repository error!");
527 continue;
528 }
529
530 if (deploySha == LatestCompileJob()?.RevisionInformation.CommitSha)
531 {
532 logger.LogTrace("Aborting auto update, same revision as latest CompileJob");
533 continue;
534 }
535
536 // finally set up the job
537 compileProcessJob = new Job
538 {
539 Instance = repositoryUpdateJob.Instance,
540 Description = "Scheduled code deployment",
541 CancelRightsType = RightsType.DreamMaker,
542 CancelRight = (ulong)DreamMakerRights.CancelCompile,
543 };
544
545 await jobManager.RegisterOperation(
546 compileProcessJob,
547 (core, databaseContextFactory, job, progressReporter, jobCancellationToken) =>
548 {
549 if (core != this)
550 throw new InvalidOperationException(DifferentCoreExceptionMessage);
551 return DreamMaker.DeploymentProcess(
552 job,
553 databaseContextFactory,
554 progressReporter,
555 jobCancellationToken);
556 },
557 cancellationToken);
558 }
559
560 await jobManager.WaitForJobCompletion(compileProcessJob, null, default, cancellationToken);
561 }
562 catch (Exception e) when (e is not OperationCanceledException)
563 {
564 logger.LogWarning(e, "Error in auto update loop!");
565 continue;
566 }
567 }
568 catch (OperationCanceledException)
569 {
570 logger.LogDebug("Cancelled auto update loop!");
571 break;
572 }
573
574 logger.LogTrace("Leaving auto update loop...");
575 }
