480  async Task TimerLoop(uint minutes, CancellationToken cancellationToken)
481  {
482  logger.LogDebug("Entering auto-update loop");
483  while (true)
484  try
485  {
486  await Task.Delay(TimeSpan.FromMinutes(minutes > Int32.MaxValue ? Int32.MaxValue : minutes), cancellationToken).ConfigureAwait(false);
487  logger.LogInformation("Beginning auto update...");
488  await eventConsumer.HandleEvent(EventType.InstanceAutoUpdateStart, Enumerable.Empty<string>(), cancellationToken).ConfigureAwait(false);
489  try
490  {
491  var repositoryUpdateJob = new Job
492  {
493  Instance = new Models.Instance
494  {
495  Id = metadata.Id,
496  },
497  Description = "Scheduled repository update",
498  CancelRightsType = RightsType.Repository,
499  CancelRight = (ulong)RepositoryRights.CancelPendingChanges,
500  };
501 
502  await jobManager.RegisterOperation(
503  repositoryUpdateJob,
504  RepositoryAutoUpdateJob,
505  cancellationToken)
506  .ConfigureAwait(false);
507 
508  // DCT: First token will cancel the job, second is for cancelling the cancellation, unwanted
509  await jobManager.WaitForJobCompletion(repositoryUpdateJob, null, cancellationToken, default).ConfigureAwait(false);
510 
511  Job compileProcessJob;
512  using (var repo = await RepositoryManager.LoadRepository(cancellationToken).ConfigureAwait(false))
513  {
514  var deploySha = repo.Head;
515  if (deploySha == null)
516  {
517  logger.LogTrace("Aborting auto update, repository error!");
518  continue;
519  }
520 
521  if (deploySha == LatestCompileJob()?.RevisionInformation.CommitSha)
522  {
523  logger.LogTrace("Aborting auto update, same revision as latest CompileJob");
524  continue;
525  }
526 
527  // finally set up the job
528  compileProcessJob = new Job
529  {
530  Instance = repositoryUpdateJob.Instance,
531  Description = "Scheduled code deployment",
532  CancelRightsType = RightsType.DreamMaker,
533  CancelRight = (ulong)DreamMakerRights.CancelCompile,
534  };
535 
536  await jobManager.RegisterOperation(
537  compileProcessJob,
538  (core, databaseContextFactory, job, progressReporter, jobCancellationToken) =>
539  {
540  if (core != this)
541  throw new InvalidOperationException(DifferentCoreExceptionMessage);
542  return DreamMaker.DeploymentProcess(
543  job,
544  databaseContextFactory,
545  progressReporter,
546  jobCancellationToken);
547  },
548  cancellationToken)
549  .ConfigureAwait(false);
550  }
551 
552  await jobManager.WaitForJobCompletion(compileProcessJob, null, default, cancellationToken).ConfigureAwait(false);
553  }
554  catch (Exception e) when (!(e is OperationCanceledException))
555  {
556  logger.LogWarning(e, "Error in auto update loop!");
557  continue;
558  }
559  }
560  catch (OperationCanceledException)
561  {
562  logger.LogDebug("Cancelled auto update loop!");
563  break;
564  }
