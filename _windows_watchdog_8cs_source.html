148 protected override async Task HandleNewDmbAvailable(CancellationToken cancellationToken)
149 {
150 IDmbProvider compileJobProvider = DmbFactory.LockNextDmb(1);
151 bool canSeamlesslySwap = true;
152
153 if (compileJobProvider.CompileJob.ByondVersion != ActiveCompileJob.ByondVersion)
154 {
155 // have to do a graceful restart
156 Logger.LogDebug(
157 "Not swapping to new compile job {0} as it uses a different BYOND version ({1}) than what is currently active {2}. Queueing graceful restart instead...",
158 compileJobProvider.CompileJob.Id,
159 compileJobProvider.CompileJob.ByondVersion,
160 ActiveCompileJob.ByondVersion);
161 canSeamlesslySwap = false;
162 }
163
164 if (compileJobProvider.CompileJob.DmeName != ActiveCompileJob.DmeName)
165 {
166 Logger.LogDebug(
167 "Not swapping to new compile job {0} as it uses a different .dmb name ({1}) than what is currently active {2}. Queueing graceful restart instead...",
168 compileJobProvider.CompileJob.Id,
169 compileJobProvider.CompileJob.DmeName,
170 ActiveCompileJob.DmeName);
171 canSeamlesslySwap = false;
172 }
173
174 if (!canSeamlesslySwap)
175 {
176 compileJobProvider.Dispose();
177 await base.HandleNewDmbAvailable(cancellationToken);
178 return;
179 }
180
181 SwappableDmbProvider windowsProvider = null;
182 bool suspended = false;
183 try
184 {
185 windowsProvider = new SwappableDmbProvider(compileJobProvider, GameIOManager, symlinkFactory);
186
187 Logger.LogDebug("Swapping to compile job {0}...", windowsProvider.CompileJob.Id);
188 try
189 {
190 Server.Suspend();
191 suspended = true;
192 }
193 catch (Exception ex)
194 {
195 Logger.LogWarning(ex, "Exception while suspending server!");
196 }
197
198 await windowsProvider.MakeActive(cancellationToken);
199 }
200 catch (Exception ex)

202 Logger.LogError(ex, "Exception while swapping");
203 IDmbProvider providerToDispose = windowsProvider ?? compileJobProvider;
204 providerToDispose.Dispose();
205 throw;
206 }
207
208 // Let this throw hard if it fails
209 if (suspended)
210 Server.Resume();
211
212 pendingSwappable?.Dispose();
213 pendingSwappable = windowsProvider;

217 protected sealed override async Task<IDmbProvider> PrepServerForLaunch(IDmbProvider dmbToUse, CancellationToken cancellationToken)
218 {
219 if (ActiveSwappable != null)
220 throw new InvalidOperationException("Expected activeSwappable to be null!");
221 if (startupDmbProvider != null)
222 throw new InvalidOperationException("Expected startupDmbProvider to be null!");
223
224 Logger.LogTrace("Prep for server launch. pendingSwappable is {0}available", pendingSwappable == null ? "not " : String.Empty);
225
226 // Add another lock to the startup DMB because it'll be used throughout the lifetime of the watchdog
227 startupDmbProvider = await DmbFactory.FromCompileJob(dmbToUse.CompileJob, cancellationToken);
228
229 pendingSwappable ??= new SwappableDmbProvider(dmbToUse, GameIOManager, symlinkFactory);
230 ActiveSwappable = pendingSwappable;
231 pendingSwappable = null;
232
233 try
234 {
235 await InitialLink(cancellationToken);
236 }
237 catch (Exception ex)
238 {
239 // We won't worry about disposing activeSwappable here as we can't dispose dmbToUse here.
240 Logger.LogTrace(ex, "Initial link error, nulling ActiveSwappable");
241 ActiveSwappable = null;
242 throw;
243 }
244
245 return ActiveSwappable;
