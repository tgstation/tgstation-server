219 public async Task<ISessionController> LaunchNew(
220 IDmbProvider dmbProvider,
221 IByondExecutableLock currentByondLock,
222 DreamDaemonLaunchParameters launchParameters,
223 bool apiValidate,
224 CancellationToken cancellationToken)
225 {
226 logger.LogTrace("Begin session launch...");
227 if (!launchParameters.Port.HasValue)
228 throw new InvalidOperationException("Given port is null!");
229 switch (dmbProvider.CompileJob.MinimumSecurityLevel)
230 {
231 case DreamDaemonSecurity.Ultrasafe:
232 break;
233 case DreamDaemonSecurity.Safe:
234 if (launchParameters.SecurityLevel == DreamDaemonSecurity.Ultrasafe)
235 {
236 logger.LogTrace("Boosting security level to minimum of Safe");
237 launchParameters.SecurityLevel = DreamDaemonSecurity.Safe;
238 }
239
240 break;
241 case DreamDaemonSecurity.Trusted:
242 if (launchParameters.SecurityLevel != DreamDaemonSecurity.Trusted)
243 logger.LogTrace("Boosting security level to minimum of Trusted");
244
245 launchParameters.SecurityLevel = DreamDaemonSecurity.Trusted;
246 break;
247 default:
248 throw new InvalidOperationException(String.Format(CultureInfo.InvariantCulture, "Invalid DreamDaemonSecurity value: {0}", dmbProvider.CompileJob.MinimumSecurityLevel));
249 }
250
251 var chatTrackingContext = chat.CreateTrackingContext();
252 try
253 {
254 // get the byond lock
255 var byondLock = currentByondLock ?? await byond.UseExecutables(Version.Parse(dmbProvider.CompileJob.ByondVersion), cancellationToken);
256 try
257 {
258 logger.LogDebug(
259 "Launching session with CompileJob {0}...",
260 dmbProvider.CompileJob.Id);
261
262 if (launchParameters.SecurityLevel == DreamDaemonSecurity.Trusted)
263 await byondLock.TrustDmbPath(ioManager.ConcatPath(dmbProvider.Directory, dmbProvider.DmbName), cancellationToken);
264
265 PortBindTest(launchParameters.Port.Value);
266 await CheckPagerIsNotRunning(cancellationToken);
267
268 var accessIdentifier = cryptographySuite.GetSecureString();
269
270 var byondTopicSender = topicClientFactory.CreateTopicClient(
271 TimeSpan.FromMilliseconds(
272 launchParameters.TopicRequestTimeout.Value));
273
274 // set command line options
275 // more sanitization here cause it uses the same scheme
276 var parameters = $"{DMApiConstants.ParamApiVersion}={byondTopicSender.SanitizeString(DMApiConstants.InteropVersion.Semver().ToString())}&{byondTopicSender.SanitizeString(DMApiConstants.ParamServerPort)}={serverPortProvider.HttpApiPort}&{byondTopicSender.SanitizeString(DMApiConstants.ParamAccessIdentifier)}={byondTopicSender.SanitizeString(accessIdentifier)}";
277
278 if (!String.IsNullOrEmpty(launchParameters.AdditionalParameters))
279 parameters = $"{parameters}&{launchParameters.AdditionalParameters}";
280
281 // important to run on all ports to allow port changing
282 Guid? logFileGuid = null;
283 var arguments = String.Format(
284 CultureInfo.InvariantCulture,
285 "{0} -port {1} -ports 1-65535 {2}-close -verbose -logself -{3} -{4}{5}{6} -params \"{7}\"",
286 dmbProvider.DmbName,
287 launchParameters.Port.Value,
288 launchParameters.AllowWebClient.Value ? "-webclient " : String.Empty,
289 SecurityWord(launchParameters.SecurityLevel.Value),
290 VisibilityWord(launchParameters.Visibility.Value),
291 platformIdentifier.IsWindows
292 ? $" -log {logFileGuid = Guid.NewGuid()}"
293 : String.Empty, // Just use stdout on linux
294 launchParameters.StartProfiler.Value
295 ? " -profile"
296 : String.Empty,
297 parameters);
298
299 // See https://github.com/tgstation/tgstation-server/issues/719
300 var noShellExecute = !platformIdentifier.IsWindows;
301
302 if (!apiValidate && dmbProvider.CompileJob.DMApiVersion == null)
303 logger.LogDebug("Session will have no DMAPI support!");
304
305 // launch dd
306 var process = processExecutor.LaunchProcess(
307 byondLock.DreamDaemonPath,
308 dmbProvider.Directory,
309 arguments,
310 noShellExecute,
311 noShellExecute,
312 noShellExecute: noShellExecute);

442 public async Task<ISessionController> Reattach(
443 ReattachInformation reattachInformation,
444 CancellationToken cancellationToken)
445 {
446 if (reattachInformation == null)
447 throw new ArgumentNullException(nameof(reattachInformation));
448
449 logger.LogTrace("Begin session reattach...");
450 var byondTopicSender = topicClientFactory.CreateTopicClient(reattachInformation.TopicRequestTimeout);
451 var chatTrackingContext = chat.CreateTrackingContext();
452 try
453 {
454 var byondLock = await byond.UseExecutables(Version.Parse(reattachInformation.Dmb.CompileJob.ByondVersion), cancellationToken);
455
456 try
457 {
458 logger.LogDebug(
459 "Attaching to session PID: {0}, CompileJob: {1}...",
460 reattachInformation.ProcessId,
461 reattachInformation.Dmb.CompileJob.Id);
462
463 var process = processExecutor.GetProcess(reattachInformation.ProcessId);
464 if (process == null)
465 return null;
466
467 try
468 {
469 networkPromptReaper.RegisterProcess(process);
470 var runtimeInformation = CreateRuntimeInformation(
471 reattachInformation.Dmb,
472 chatTrackingContext,
473 null,
474 null,
475 false);
476 reattachInformation.SetRuntimeInformation(runtimeInformation);
477
478 var controller = new SessionController(
479 reattachInformation,
480 instance,
481 process,
482 byondLock,
483 byondTopicSender,
484 chatTrackingContext,
485 bridgeRegistrar,
486 chat,
487 assemblyInformationProvider,
488 loggerFactory.CreateLogger<SessionController>(),
489 () => Task.CompletedTask,
490 null,
491 true,
492 false);
493
494 process = null;
495 byondLock = null;
496 chatTrackingContext = null;
497
498 return controller;
499 }
500 finally
501 {
502 process?.Dispose();
503 }
504 }
505 finally
506 {
507 byondLock?.Dispose();
508 }
509 }
510 finally
511 {
512 chatTrackingContext?.Dispose();
513 }
514 }

546 async Task CheckPagerIsNotRunning(CancellationToken cancellationToken)
547 {
548 if (!platformIdentifier.IsWindows)
549 return;
550
551 using var otherProcess = processExecutor.GetProcessByName("byond");
552 if (otherProcess == null)
553 return;
554
555 var otherUsernameTask = otherProcess.GetExecutingUsername(cancellationToken);
556 using var ourProcess = processExecutor.GetCurrentProcess();
557 var ourUserName = await ourProcess.GetExecutingUsername(cancellationToken);
558 var otherUserName = await otherUsernameTask;
559
560 if (otherUserName.Equals(ourUserName, StringComparison.Ordinal))
561 throw new JobException(ErrorCode.DeploymentPagerRunning);
562 }
