234 public async Task MoveInstance(Models.Instance instance, string oldPath, CancellationToken cancellationToken)
235 {
236 if (oldPath == null)
237 throw new ArgumentNullException(nameof(oldPath));
238
239 using var lockContext = await SemaphoreSlimContext.Lock(instanceStateChangeSemaphore, cancellationToken);
240 using var instanceReferenceCheck = GetInstanceReference(instance);
241 if (instanceReferenceCheck != null)
242 throw new InvalidOperationException("Cannot move an online instance!");
243 var newPath = instance.Path;
244 try
245 {
246 await ioManager.MoveDirectory(oldPath, newPath, cancellationToken);
247
248 // Delete the Game directory to clear out broken symlinks
249 var instanceGameIOManager = instanceFactory.CreateGameIOManager(instance);
250 await instanceGameIOManager.DeleteDirectory(DefaultIOManager.CurrentDirectory, cancellationToken);
251 }
252 catch (Exception ex)
253 {
254 logger.LogError(
255 ex,
256 "Error moving instance {instanceId}!",
257 instance.Id);
258 try
259 {
260 logger.LogDebug("Reverting instance {instanceId}'s path to {oldPath} in the DB...", instance.Id, oldPath);
261
262 // DCT: Operation must always run
263 await databaseContextFactory.UseContext(db =>
264 {
265 var targetInstance = new Models.Instance

267 Id = instance.Id,
268 };
269 db.Instances.Attach(targetInstance);
270 targetInstance.Path = oldPath;
271 return db.Save(default);
272 });
273 }
274 catch (Exception innerEx)
275 {
276 logger.LogCritical(
277 innerEx,
278 "Error reverting database after failing to move instance {instanceId}! Attempting to detach...",
279 instance.Id);
280

282 {
283 // DCT: Operation must always run

294 "Okay, what gamma radiation are you under? Failed to write instance attach file!");

307 public async Task OfflineInstance(Models.Instance metadata, Models.User user, CancellationToken cancellationToken)
308 {
309 if (metadata == null)
310 throw new ArgumentNullException(nameof(metadata));
311
312 using (await SemaphoreSlimContext.Lock(instanceStateChangeSemaphore, cancellationToken))
313 {
314 ReferenceCountingContainer<IInstance, InstanceWrapper> container;
315 lock (instances)
316 {
317 if (!instances.TryGetValue(metadata.Id.Value, out container))
318 {
319 logger.LogDebug("Not offlining removed instance {instanceId}", metadata.Id);
320 return;
321 }
322
323 instances.Remove(metadata.Id.Value);
324 }
325
326 logger.LogInformation("Offlining instance ID {instanceId}", metadata.Id);
327
328 try
329 {
330 await container.OnZeroReferences.WithToken(cancellationToken);
331
332 // we are the one responsible for cancelling his jobs
333 var tasks = new List<Task>();
334 await databaseContextFactory.UseContext(
335 async db =>
336 {
337 var jobs = await db
338 .Jobs
339 .AsQueryable()
340 .Where(x => x.Instance.Id == metadata.Id && !x.StoppedAt.HasValue)
341 .Select(x => new Models.Job
342 {
343 Id = x.Id,
344 })
345 .ToListAsync(cancellationToken);
346 foreach (var job in jobs)
347 tasks.Add(jobManager.CancelJob(job, user, true, cancellationToken));
348 });
349
350 await Task.WhenAll(tasks);
351 }
352 catch
353 {
354 // not too late to change your mind
355 lock (instances)
356 instances.Add(metadata.Id.Value, container);
357
358 throw;
359 }
360
361 try
362 {
363 // at this point we can't really stop offlining the instance just because the request was cancelled
364 await container.Instance.StopAsync(shutdownCancellationTokenSource.Token);
365 }
366 finally
367 {
368 await container.Instance.DisposeAsync();
369 }
370 }
371 }

473 public async Task<BridgeResponse> ProcessBridgeRequest(BridgeParameters parameters, CancellationToken cancellationToken)
474 {
475 if (parameters == null)
476 throw new ArgumentNullException(nameof(parameters));
477
478 IBridgeHandler bridgeHandler = null;
479 for (var i = 0; bridgeHandler == null && i < 30; ++i)
480 {
481 // There's a miniscule time period where we could potentially receive a bridge request and not have the registration ready when we launch DD
482 // This is a stopgap
483 Task delayTask = Task.CompletedTask;
484 lock (bridgeHandlers)
485 if (!bridgeHandlers.TryGetValue(parameters.AccessIdentifier, out bridgeHandler))
486 delayTask = asyncDelayer.Delay(TimeSpan.FromMilliseconds(100), cancellationToken);
487
488 await delayTask;
489 }
490
491 if (bridgeHandler == null)
492 lock (bridgeHandlers)
493 if (!bridgeHandlers.TryGetValue(parameters.AccessIdentifier, out bridgeHandler))
494 {
495 logger.LogWarning("Recieved invalid bridge request with access identifier: {accessIdentifier}", parameters.AccessIdentifier);
496 return null;
497 }
498
499 return await bridgeHandler.ProcessBridgeRequest(parameters, cancellationToken);

540 async Task Initialize(CancellationToken cancellationToken)
541 {
542 try
543 {
544 logger.LogInformation("{versionString}", assemblyInformationProvider.VersionString);
545
546 CheckSystemCompatibility();
547
548 // To let the web server startup immediately before we do any intense work
549 await Task.Yield();
550
551 await InitializeSwarm(cancellationToken);
552
553 List<Models.Instance> dbInstances = null;
554 var instanceEnumeration = databaseContextFactory.UseContext(
555 async databaseContext => dbInstances = await databaseContext
556 .Instances
557 .AsQueryable()
558 .Where(x => x.Online.Value && x.SwarmIdentifer == swarmConfiguration.Identifier)
559 .Include(x => x.RepositorySettings)
560 .Include(x => x.ChatSettings)
561 .ThenInclude(x => x.Channels)
562 .Include(x => x.DreamDaemonSettings)
563 .ToListAsync(cancellationToken));
564
565 var factoryStartup = instanceFactory.StartAsync(cancellationToken);
566 var jobManagerStartup = jobManager.StartAsync(cancellationToken);
567
568 await Task.WhenAll(instanceEnumeration, factoryStartup, jobManagerStartup);
569
570 var instanceOnliningTasks = dbInstances.Select(
571 async metadata =>
572 {
573 try
574 {
575 await OnlineInstance(metadata, cancellationToken);
576 }
577 catch (Exception ex)
578 {
579 logger.LogError(ex, "Failed to online instance {instanceId}!", metadata.Id);
580 }
581 });
582
583 await Task.WhenAll(instanceOnliningTasks);
584
585 jobManager.Activate(this);
586
587 logger.LogInformation("Server ready!");
588 readyTcs.SetResult();
589 }
590 catch (OperationCanceledException ex)
591 {
592 logger.LogInformation(ex, "Cancelled instance manager initialization!");
593 }
594 catch (Exception e)
595 {
596 logger.LogCritical(e, "Instance manager startup error!");
597 try
598 {
599 await serverControl.Die(e);
600 return;
601 }
602 catch (Exception e2)
603 {
604 logger.LogCritical(e2, "Failed to kill server!");
605 }
606 }
607 }

632 async Task InitializeSwarm(CancellationToken cancellationToken)
633 {
634 SwarmRegistrationResult registrationResult;
635 do
636 {
637 registrationResult = await swarmServiceController.Initialize(cancellationToken);
638
639 if (registrationResult == SwarmRegistrationResult.Unauthorized)
640 throw new InvalidOperationException("Swarm private key does not match the swarm controller's!");
641
642 if (registrationResult == SwarmRegistrationResult.VersionMismatch)
643 throw new InvalidOperationException("Swarm controller's TGS version does not match our own!");
644
645 if (registrationResult != SwarmRegistrationResult.Success)
646 await asyncDelayer.Delay(TimeSpan.FromSeconds(5), cancellationToken);
647 }
648 while (registrationResult != SwarmRegistrationResult.Success && !cancellationToken.IsCancellationRequested);
649 }
