178 {
179 if (model == null)

181
182 if (model.ApiValidationPort == 0)
183 throw new InvalidOperationException("ApiValidationPort cannot be 0!");
184
185 var hostModel = await DatabaseContext
186 .DreamMakerSettings
187 .AsQueryable()
188 .Where(x => x.InstanceId == Instance.Id)
189 .FirstOrDefaultAsync(cancellationToken);
190 if (hostModel == null)
191 return Gone();
192
193 if (model.ProjectName != null)
194 {
195 if (!AuthenticationContext.InstancePermissionSet.DreamMakerRights.Value.HasFlag(DreamMakerRights.SetDme))
196 return Forbid();
197 if (model.ProjectName.Length == 0)
198 hostModel.ProjectName = null;
199 else
200 hostModel.ProjectName = model.ProjectName;
201 }
202
203 if (model.ApiValidationPort.HasValue)
204 {
205 if (!AuthenticationContext.InstancePermissionSet.DreamMakerRights.Value.HasFlag(DreamMakerRights.SetApiValidationPort))
206 return Forbid();
207
208 if (model.ApiValidationPort.Value != hostModel.ApiValidationPort.Value)
209 {
210 var verifiedPort = await portAllocator
211 .GetAvailablePort(
212 model.ApiValidationPort.Value,
213 true,
214 cancellationToken);
215 if (verifiedPort != model.ApiValidationPort)
216 return Conflict(new ErrorMessageResponse(ErrorCode.PortNotAvailable));
217
218 hostModel.ApiValidationPort = model.ApiValidationPort;
219 }
220 }
221
222 if (model.ApiValidationSecurityLevel.HasValue)
223 {
224 if (!AuthenticationContext.InstancePermissionSet.DreamMakerRights.Value.HasFlag(DreamMakerRights.SetSecurityLevel))
225 return Forbid();
226 hostModel.ApiValidationSecurityLevel = model.ApiValidationSecurityLevel;
227 }
228
229 if (model.RequireDMApiValidation.HasValue)
230 {
231 if (!AuthenticationContext.InstancePermissionSet.DreamMakerRights.Value.HasFlag(DreamMakerRights.SetApiValidationRequirement))
232 return Forbid();
233 hostModel.RequireDMApiValidation = model.RequireDMApiValidation;
234 }
235
236 if (model.Timeout.HasValue)
237 {
238 if (!AuthenticationContext.InstancePermissionSet.DreamMakerRights.Value.HasFlag(DreamMakerRights.SetTimeout))
239 return Forbid();
240 hostModel.Timeout = model.Timeout;
241 }
242
243 await DatabaseContext.Save(cancellationToken);
244
245 if ((AuthenticationContext.GetRight(RightsType.DreamMaker) & (ulong)DreamMakerRights.Read) == 0)
246 return NoContent();
