249 static bool BadCreateRequestChecks(
250 UserCreateRequest createRequest,
251 bool? needZeroLengthPasswordWithOAuthConnections,
252 [NotNullWhen(true)] out AuthorityResponse<UpdatedUser>? failResponse)
253 {
254 if (createRequest.OAuthConnections?.Any(x => x == null) == true)
255 {
256 failResponse = BadRequest<UpdatedUser>(ErrorCode.ModelValidationFailure);
257 return true;
258 }
259
260 var hasNonNullPassword = createRequest.Password != null;
261 var hasNonNullSystemIdentifier = createRequest.SystemIdentifier != null;
262 var hasOAuthConnections = (createRequest.OAuthConnections?.Count > 0) == true;
263 if ((hasNonNullPassword && hasNonNullSystemIdentifier)
264 || (!hasNonNullPassword && !hasNonNullSystemIdentifier && !hasOAuthConnections))
265 {
266 failResponse = BadRequest<UpdatedUser>(ErrorCode.UserMismatchPasswordSid);
267 return true;
268 }
269
270 var hasZeroLengthPassword = createRequest.Password?.Length == 0;
271 if (needZeroLengthPasswordWithOAuthConnections.HasValue)
272 {
273 if (needZeroLengthPasswordWithOAuthConnections.Value)
274 {
275 if (createRequest.OAuthConnections == null)
276 throw new InvalidOperationException($"Expected {nameof(UserCreateRequest.OAuthConnections)} to be set here!");
277
278 if (createRequest.OAuthConnections.Count == 0)
279 {
280 failResponse = BadRequest<UpdatedUser>(ErrorCode.ModelValidationFailure);
281 return true;
282 }
283 }
284 else if (hasZeroLengthPassword)
285 {
286 failResponse = BadRequest<UpdatedUser>(ErrorCode.ModelValidationFailure);
287 return true;
288 }
289 }
290
291 if (createRequest.Group != null && createRequest.PermissionSet != null)
292 {
293 failResponse = BadRequest<UpdatedUser>(ErrorCode.UserGroupAndPermissionSet);
294 return true;
295 }
296
297 createRequest.Name = createRequest.Name?.Trim();
298 if (createRequest.Name?.Length == 0)
299 createRequest.Name = null;
300
301 if (!(createRequest.Name == null ^ createRequest.SystemIdentifier == null))
302 {
303 failResponse = BadRequest<UpdatedUser>(ErrorCode.UserMismatchNameSid);
304 return true;
305 }
306
307 failResponse = CheckValidName(createRequest, true);
308 return failResponse != null;
309 }

434#pragma warning restore CA1502
435#pragma warning restore CA1506
436 => new(
437 () =>
438 {
439 RightsConditional<AdministrationRights>? conditional = null;
440
441 // Ensure they are only trying to edit things they have perms for (system identity change will trigger a bad request)
442 if (model.OidcConnections != null || model.OAuthConnections != null)
443 conditional = Flag(AdministrationRights.EditOwnServiceConnections);
444
445 if (model.Password != null && model.Id == claimsPrincipalAccessor.User.GetTgsUserId())
446 {
447 var newFlag = Flag(AdministrationRights.EditOwnPassword);
448 if (conditional != null)
449 conditional = And(conditional, newFlag);
450 else
451 conditional = newFlag;
452 }
453
454 if (conditional != null)
455 conditional = Or(conditional, Flag(AdministrationRights.WriteUsers));
456 else if (model.Enabled.HasValue
457 || model.Group != null
458 || model.Name != null
459 || model.PermissionSet != null)
460 conditional = Flag(AdministrationRights.WriteUsers);
461
462 return conditional;
463 },
464 async authorizationService =>
465 {
466 ArgumentNullException.ThrowIfNull(model);
467
468 if (!model.Id.HasValue || model.OAuthConnections?.Any(x => x == null) == true)
469 return BadRequest<UpdatedUser>(ErrorCode.ModelValidationFailure);
470
471 if (model.Group != null && model.PermissionSet != null)
472 return BadRequest<UpdatedUser>(ErrorCode.UserGroupAndPermissionSet);
473
474 var userQuery = DatabaseContext
475 .Users
476 .AsQueryable()
477 .Where(x => x.Id == model.Id)
478 .Include(x => x.CreatedBy)
479 .Include(x => x.OAuthConnections)
480 .Include(x => x.OidcConnections)
481 .Include(x => x.Group!)
482 .ThenInclude(x => x.PermissionSet)
483 .Include(x => x.PermissionSet)
484 .FirstOrDefaultAsync(cancellationToken);
485
486 var originalUser = await userQuery;
487
488 if (originalUser == default)
489 return NotFound<UpdatedUser>();
490
491 if (originalUser.CanonicalName == User.CanonicalizeName(User.TgsSystemUserName))
492 return Forbid<UpdatedUser>();
493
494 var originalUserHasSid = originalUser.SystemIdentifier != null;
495 var invalidateSessions = false;
496 if (originalUserHasSid && originalUser.PasswordHash != null)
497 {
498 // cleanup from https://github.com/tgstation/tgstation-server/issues/1528
499 Logger.LogDebug("System user ID {userId}'s PasswordHash is polluted, updating database.", originalUser.Id);
500 originalUser.PasswordHash = null;
501
502 invalidateSessions = true;
503 }
504
505 if (model.SystemIdentifier != null && model.SystemIdentifier != originalUser.SystemIdentifier)
506 return BadRequest<UpdatedUser>(ErrorCode.UserSidChange);
507
508 if (model.Password != null)
509 {
510 if (originalUserHasSid)
511 return BadRequest<UpdatedUser>(ErrorCode.UserMismatchPasswordSid);
512
513 var result = TrySetPassword(originalUser, model.Password, false);
514 if (result != null)
515 return result;
516
517 invalidateSessions = true;
518 }
519
520 if (model.Name != null && User.CanonicalizeName(model.Name) != originalUser.CanonicalName)
521 return BadRequest<UpdatedUser>(ErrorCode.UserNameChange);
522
523 if (model.OAuthConnections != null
524 && (model.OAuthConnections.Count != originalUser.OAuthConnections!.Count
525 || !model.OAuthConnections.All(x => originalUser.OAuthConnections.Any(y => y.Provider == x.Provider && y.ExternalUserId == x.ExternalUserId))))
526 {
527 if (securityConfigurationOptions.Value.OidcStrictMode)
528 return BadRequest<UpdatedUser>(ErrorCode.BadUserEditDueToOidcStrictMode);
529
530 if (originalUser.CanonicalName == User.CanonicalizeName(DefaultCredentials.AdminUserName))
531 return BadRequest<UpdatedUser>(ErrorCode.AdminUserCannotHaveServiceConnection);
532
533 if (model.OAuthConnections.Count == 0 && originalUser.PasswordHash == null && originalUser.SystemIdentifier == null)
534 return BadRequest<UpdatedUser>(ErrorCode.CannotRemoveLastAuthenticationOption);
535
536 DatabaseContext.OAuthConnections.RemoveRange(originalUser.OAuthConnections);
537 originalUser.OAuthConnections.Clear();
538

540 originalUser.OAuthConnections.Add(new Models.OAuthConnection
541 {
542 Provider = updatedConnection.Provider,
543 ExternalUserId = updatedConnection.ExternalUserId,
544 });
545 }
546
547 if (model.OidcConnections != null
548 && (model.OidcConnections.Count != originalUser.OidcConnections!.Count
549 || !model.OidcConnections.All(x => originalUser.OidcConnections.Any(y => y.SchemeKey == x.SchemeKey && y.ExternalUserId == x.ExternalUserId))))
550 {
551 if (securityConfigurationOptions.Value.OidcStrictMode)
552 return BadRequest<UpdatedUser>(ErrorCode.BadUserEditDueToOidcStrictMode);
553
554 if (originalUser.CanonicalName == User.CanonicalizeName(DefaultCredentials.AdminUserName))
555 return BadRequest<UpdatedUser>(ErrorCode.AdminUserCannotHaveServiceConnection);
556
557 if (model.OidcConnections.Count == 0 && originalUser.PasswordHash == null && originalUser.SystemIdentifier == null)
558 return BadRequest<UpdatedUser>(ErrorCode.CannotRemoveLastAuthenticationOption);
559
560 DatabaseContext.OidcConnections.RemoveRange(originalUser.OidcConnections);
561 originalUser.OidcConnections.Clear();
562 foreach (var updatedConnection in model.OidcConnections)

573 return BadRequest<UpdatedUser>(ErrorCode.BadUserEditDueToOidcStrictMode);
574

576 .Groups
577 .AsQueryable()

608 originalUser.GroupId = null;
609 }
610
611 var fail = CheckValidName(model, false);
612 if (fail != null)
613 return fail;
614
615 originalUser.Name = model.Name ?? originalUser.Name;
616
617 if (model.Enabled.HasValue)
618 {
619 if (securityConfigurationOptions.Value.OidcStrictMode)
620 return BadRequest<UpdatedUser>(ErrorCode.BadUserEditDueToOidcStrictMode);
621
622 invalidateSessions = originalUser.Require(x => x.Enabled) && !model.Enabled.Value;

642
643 return await responseTask;
