1020 async Task CheckRunWizard(CancellationToken cancellationToken)
1021 {

1023 if (setupWizardMode == SetupWizardMode.Never)
1024 return;
1025
1026 var forceRun = setupWizardMode == SetupWizardMode.Force || setupWizardMode == SetupWizardMode.Only;
1027 if (!console.Available)
1028 {
1029 if (forceRun)
1030 throw new InvalidOperationException("Asked to run setup wizard with no console avaliable!");
1031 return;
1032 }
1033
1034 var userConfigFileName = String.Format(CultureInfo.InvariantCulture, "appsettings.{0}.yml", hostingEnvironment.EnvironmentName);

1036 async Task HandleSetupCancel()
1037 {
1038 // DCTx2: Operation should always run
1039 await console.WriteAsync(String.Empty, true, default);
1040 await console.WriteAsync("Aborting setup!", true, default);
1041 }
1042
1043 // Link passed cancellationToken with cancel key press
1044 Task finalTask = Task.CompletedTask;
1045 using (var cts = CancellationTokenSource.CreateLinkedTokenSource(cancellationToken, console.CancelKeyPress))
1046 using ((cancellationToken = cts.Token).Register(() => finalTask = HandleSetupCancel()))
1047 try
1048 {
1049 var exists = await ioManager.FileExists(userConfigFileName, cancellationToken);
1050 if (!exists)
1051 {
1052 var legacyJsonFileName = $"appsettings.{hostingEnvironment.EnvironmentName}.json";
1053 exists = await ioManager.FileExists(legacyJsonFileName, cancellationToken);
1054 if (exists)
1055 userConfigFileName = legacyJsonFileName;
1056 }
1057
1058 bool shouldRunBasedOnAutodetect;
1059 if (exists)
1060 {
1061 var bytes = await ioManager.ReadAllBytes(userConfigFileName, cancellationToken);
1062 var contents = Encoding.UTF8.GetString(bytes);
1063 var existingConfigIsEmpty = String.IsNullOrWhiteSpace(contents) || contents.Trim() == "{}";
1064 shouldRunBasedOnAutodetect = existingConfigIsEmpty;
1065 }
1066 else
1067 shouldRunBasedOnAutodetect = true;
1068
1069 if (!shouldRunBasedOnAutodetect)
1070 {
1071 if (forceRun)
1072 {
1073 await console.WriteAsync(String.Format(CultureInfo.InvariantCulture, "The configuration settings are requesting the setup wizard be run, but you already appear to have a configuration file ({0})!", userConfigFileName), true, cancellationToken);
1074
1075 forceRun = await PromptYesNo("Continue running setup wizard? (y/n): ", cancellationToken);
1076 }
1077
1078 if (!forceRun)
1079 return;
1080 }
1081
1082 // flush the logs to prevent console conflicts
1083 await asyncDelayer.Delay(TimeSpan.FromSeconds(1), cancellationToken);
1084
1085 await RunWizard(userConfigFileName, cancellationToken);
1086 }
