52 var deployedRevisionInformation = compileJob.RevisionInformation;

55 return;
56
57 previousRevisionInformation ??= new RevisionInformation();
58 previousRevisionInformation.ActiveTestMerges ??= new List<RevInfoTestMerge>();
59
60 deployedRevisionInformation.ActiveTestMerges ??= new List<RevInfoTestMerge>();
61 var tasks = new List<Task>();
62
63 // added prs
64 var addedTestMerges = deployedRevisionInformation
65 .ActiveTestMerges
66 .Select(x => x.TestMerge)
67 .Where(x => !previousRevisionInformation
68 .ActiveTestMerges
69 .Any(y => y.TestMerge.Number == x.Number))
70 .ToList();
71 var removedTestMerges = previousRevisionInformation
72 .ActiveTestMerges
73 .Select(x => x.TestMerge)
74 .Where(x => !deployedRevisionInformation
75 .ActiveTestMerges
76 .Any(y => y.TestMerge.Number == x.Number))
77 .ToList();
78 var updatedTestMerges = deployedRevisionInformation
79 .ActiveTestMerges
80 .Select(x => x.TestMerge)
81 .Where(x => previousRevisionInformation
82 .ActiveTestMerges
83 .Any(y => y.TestMerge.Number == x.Number))
84 .ToList();
85
86 if (!addedTestMerges.Any() && !removedTestMerges.Any() && !updatedTestMerges.Any())
87 return;
88
89 Logger.LogTrace(
90 "Commenting on {0} added, {1} removed, and {2} updated test merge sources...",
91 addedTestMerges.Count,
92 removedTestMerges.Count,
93 updatedTestMerges.Count);
94 foreach (var addedTestMerge in addedTestMerges)
95 tasks.Add(
96 CommentOnTestMergeSource(
97 repositorySettings,
98 repoOwner,
99 repoName,
100 FormatTestMerge(
101 repositorySettings,
102 compileJob,
103 addedTestMerge,
104 repoOwner,
105 repoName,
106 false),
107 addedTestMerge.Number,
108 cancellationToken));
109
110 foreach (var removedTestMerge in removedTestMerges)
111 tasks.Add(
112 CommentOnTestMergeSource(
113 repositorySettings,
114 repoOwner,
115 repoName,
116 "#### Test Merge Removed",
117 removedTestMerge.Number,
118 cancellationToken));
119
120 foreach (var updatedTestMerge in updatedTestMerges)

123 repositorySettings,
124 repoOwner,

130 repoOwner,
131 repoName,
132 true),

134 cancellationToken));
135
136 if (tasks.Any())

138 }
