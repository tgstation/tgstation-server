344 BadRequestObjectResult? ValidateEngineVersion(EngineVersion? version)
345 {
346 if (version == null || !version.Engine.HasValue)
347 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure));
348
349 var isByond = version.Engine.Value == EngineType.Byond;
350 var validSha = version.SourceSHA?.Length == Limits.MaximumCommitShaLength;
351 if ((isByond
352 && (version.Version == null
353 || validSha))
354 || (version.Engine.Value == EngineType.OpenDream &&
355 ((version.SourceSHA == null && version.Version == null)
356 || (version.Version != null && (version.Version.Revision != -1 || validSha)))))
357 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure));
358
359 if (isByond)
360 {
361 version.Version = NormalizeByondVersion(version.Version!);
362 if (version.Version.Build != -1)
363 return BadRequest(new ErrorMessageResponse(ErrorCode.ModelValidationFailure));
364 }
