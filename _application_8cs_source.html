391 public void Configure(
392 IApplicationBuilder applicationBuilder,
393 IServerControl serverControl,
394 ITokenFactory tokenFactory,
395 IInstanceManager instanceManager,
396 IServerPortProvider serverPortProvider,
397 IAssemblyInformationProvider assemblyInformationProvider,
398 IOptions<ControlPanelConfiguration> controlPanelConfigurationOptions,
399 IOptions<GeneralConfiguration> generalConfigurationOptions,
400 ILogger<Application> logger)
401 {
402 if (applicationBuilder == null)
403 throw new ArgumentNullException(nameof(applicationBuilder));
404 if (serverControl == null)
405 throw new ArgumentNullException(nameof(serverControl));
406
407 this.tokenFactory = tokenFactory ?? throw new ArgumentNullException(nameof(tokenFactory));
408
409 if (instanceManager == null)
410 throw new ArgumentNullException(nameof(instanceManager));
411 if (serverPortProvider == null)
412 throw new ArgumentNullException(nameof(serverPortProvider));
413 if (assemblyInformationProvider == null)
414 throw new ArgumentNullException(nameof(assemblyInformationProvider));
415
416 var controlPanelConfiguration = controlPanelConfigurationOptions?.Value ?? throw new ArgumentNullException(nameof(controlPanelConfigurationOptions));
417 var generalConfiguration = generalConfigurationOptions?.Value ?? throw new ArgumentNullException(nameof(generalConfigurationOptions));
418
419 if (logger == null)
420 throw new ArgumentNullException(nameof(logger));
421
422 logger.LogDebug("Content Root: {contentRoot}", hostingEnvironment.ContentRootPath);
423 logger.LogTrace("Web Root: {webRoot}", hostingEnvironment.WebRootPath);
424
425 // attempt to restart the server if the configuration changes
426 if (serverControl.WatchdogPresent)
427 ChangeToken.OnChange(Configuration.GetReloadToken, () =>
428 {
429 logger.LogInformation("Configuration change detected");
430 serverControl.Restart();
431 });
432
433 // setup the HTTP request pipeline
434 // Final point where we wrap exceptions in a 500 (ErrorMessage) response
435 applicationBuilder.UseServerErrorHandling();
436
437 // Add the X-Powered-By response header
438 applicationBuilder.UseServerBranding(assemblyInformationProvider);
439
440 // 503 requests made while the application is starting
441 applicationBuilder.UseAsyncInitialization(async (cancellationToken) =>
442 {
443 await instanceManager.Ready.WithToken(cancellationToken);
444 });
445
446 // suppress OperationCancelledExceptions, they are just aborted HTTP requests
447 applicationBuilder.UseCancelledRequestSuppression();
448
449 if (generalConfiguration.HostApiDocumentation)
450 {
451 applicationBuilder.UseSwagger();
452 applicationBuilder.UseSwaggerUI(c => c.SwaggerEndpoint("/swagger/v1/swagger.json", "TGS API"));
453 logger.LogTrace("Swagger API generation enabled");
454 }
455
456 // Set up CORS based on configuration if necessary
457 Action<CorsPolicyBuilder> corsBuilder = null;
458 if (controlPanelConfiguration.AllowAnyOrigin)
459 {
460 logger.LogTrace("Access-Control-Allow-Origin: *");
461 corsBuilder = builder => builder.AllowAnyOrigin();
462 }
463 else if (controlPanelConfiguration.AllowedOrigins?.Count > 0)
464 {
465 logger.LogTrace("Access-Control-Allow-Origin: {allowedOrigins}", String.Join(',', controlPanelConfiguration.AllowedOrigins));
466 corsBuilder = builder => builder.WithOrigins(controlPanelConfiguration.AllowedOrigins.ToArray());
467 }
468
469 var originalBuilder = corsBuilder;
470 corsBuilder = builder =>
471 {
472 builder
473 .AllowAnyHeader()
474 .AllowAnyMethod()
475 .SetPreflightMaxAge(TimeSpan.FromDays(1));
476 originalBuilder?.Invoke(builder);
477 };
478 applicationBuilder.UseCors(corsBuilder);
479
480 // spa loading if necessary
481 if (controlPanelConfiguration.Enable)
482 {
483 logger.LogInformation("Web control panel enabled.");
484 applicationBuilder.UseFileServer(new FileServerOptions
485 {
486 RequestPath = ControlPanelController.ControlPanelRoute,
487 EnableDefaultFiles = true,
488 EnableDirectoryBrowsing = false,
489 });
490 }
491 else
492 logger.LogTrace("Web control panel disabled!");
493
494 // Do not cache a single thing beyond this point, it's all API
495 applicationBuilder.UseDisabledClientCache();
496
497 // authenticate JWT tokens using our security pipeline if present, returns 401 if bad
498 applicationBuilder.UseAuthentication();
499
500 // suppress and log database exceptions
501 applicationBuilder.UseDbConflictHandling();
502
503 // majority of handling is done in the controllers
504 applicationBuilder.UseMvc();
505
506 // 404 anything that gets this far
507 // End of request pipeline setup
508 logger.LogTrace("Configuration version: {configVersion}", GeneralConfiguration.CurrentConfigVersion);
509 logger.LogTrace("DMAPI Interop version: {interopVersion}", DMApiConstants.InteropVersion);
510 if (controlPanelConfiguration.Enable)
511 logger.LogTrace("Web control panel version: {webCPVersion}", MasterVersionsAttribute.Instance.RawControlPanelVersion);
512
513 logger.LogDebug("Starting hosting on port {httpApiPort}...", serverPortProvider.HttpApiPort);
514 }
