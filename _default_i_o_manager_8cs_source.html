59 static void NormalizeAndDelete(IDirectoryInfo dir, CancellationToken cancellationToken)
60 {
61 cancellationToken.ThrowIfCancellationRequested();
62
63 // check if we are a symbolic link
64 if (!dir.Attributes.HasFlag(FileAttributes.Directory) || dir.Attributes.HasFlag(FileAttributes.ReparsePoint))
65 {
66 dir.Delete();
67 return;
68 }
69
70 List<Exception>? exceptions = null;
71 foreach (var subDir in dir.EnumerateDirectories())
72 try
73 {
74 NormalizeAndDelete(subDir, cancellationToken);
75 }
76 catch (AggregateException ex)
77 {
78 exceptions ??= new List<Exception>();
79 exceptions.AddRange(ex.InnerExceptions);
80 }
81
82 foreach (var file in dir.EnumerateFiles())
83 {
84 cancellationToken.ThrowIfCancellationRequested();
85 try
86 {
87 file.Attributes = FileAttributes.Normal;
88 file.Delete();
89 }
90 catch (Exception ex)
91 {
92 exceptions ??= new List<Exception>();
93 exceptions.Add(ex);
94 }
95 }
96
97 cancellationToken.ThrowIfCancellationRequested();
98 try
99 {
100 dir.Delete(true);
101 }
102 catch (Exception ex)
103 {
104 exceptions ??= new List<Exception>();
105 exceptions.Add(ex);
106 }
107
108 if (exceptions != null)
109 throw new AggregateException(exceptions);
110 }

440 IEnumerable<Task> CopyDirectoryImpl(
441 string src,
442 string dest,
443 IEnumerable<string>? ignore,
444 Func<string, string, ValueTask>? postCopyCallback,
445 SemaphoreSlim? semaphore,
446 CancellationToken cancellationToken)
447 {
448 var dir = fileSystem.DirectoryInfo.New(src);
449 Task? subdirCreationTask = null;
450 foreach (var subDirectory in dir.EnumerateDirectories())
451 {
452 if (ignore != null && ignore.Contains(subDirectory.Name))
453 continue;
454
455 var checkingSubdirCreationTask = true;
456 foreach (var copyTask in CopyDirectoryImpl(subDirectory.FullName, fileSystem.Path.Combine(dest, subDirectory.Name), null, postCopyCallback, semaphore, cancellationToken))
457 {
458 if (subdirCreationTask == null)
459 {
460 subdirCreationTask = copyTask;
461 yield return subdirCreationTask;
462 }
463 else if (!checkingSubdirCreationTask)
464 yield return copyTask;
465
466 checkingSubdirCreationTask = false;
467 }
468 }
469
470 foreach (var fileInfo in dir.EnumerateFiles())
471 {
472 if (subdirCreationTask == null)
473 {
474 subdirCreationTask = CreateDirectory(dest, cancellationToken);
475 yield return subdirCreationTask;
476 }
477
478 if (ignore != null && ignore.Contains(fileInfo.Name))
479 continue;
480
481 var sourceFile = fileInfo.FullName;
482 var destFile = ConcatPath(dest, fileInfo.Name);
483
484 async Task CopyThisFile()
485 {
486 await subdirCreationTask.WaitAsync(cancellationToken);
487 using var lockContext = semaphore != null
488 ? await SemaphoreSlimContext.Lock(semaphore, cancellationToken)
489 : null;
490 await CopyFile(sourceFile, destFile, cancellationToken);
491 if (postCopyCallback != null)
492 await postCopyCallback(sourceFile, destFile);
493 }
494
495 yield return CopyThisFile();
496 }
497 }
