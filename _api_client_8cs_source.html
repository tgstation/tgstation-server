307 protected virtual async Task<TResult> RunRequest<TResult>(
308 string route,
309 HttpContent? content,
310 HttpMethod method,
311 long? instanceId,
312 bool tokenRefresh,
313 CancellationToken cancellationToken)
314 {
315 if (route == null)
316 throw new ArgumentNullException(nameof(route));
317 if (method == null)
318 throw new ArgumentNullException(nameof(method));
319 if (content == null && (method == HttpMethod.Post || method == HttpMethod.Put))
320 throw new InvalidOperationException("content cannot be null for POST or PUT!");
321
322 HttpResponseMessage response;
323 var fullUri = new Uri(Url, route);
324 var serializerSettings = GetSerializerSettings();
325 var fileDownload = typeof(TResult) == typeof(Stream);
326 using (var request = new HttpRequestMessage(method, fullUri))
327 {
328 if (content != null)
329 request.Content = content;
330
331 try
332 {
333 var headersToUse = tokenRefresh ? tokenRefreshHeaders! : headers;
334 headersToUse.SetRequestHeaders(request.Headers, instanceId);
335
336 if (authless)
337 request.Headers.Remove(HeaderNames.Authorization);
338
339 if (fileDownload)
340 request.Headers.Accept.Add(new MediaTypeWithQualityHeaderValue(MediaTypeNames.Application.Octet));
341
342 await Task.WhenAll(requestLoggers.Select(x => x.LogRequest(request, cancellationToken))).ConfigureAwait(false);
343
344 response = await httpClient.SendAsync(request, HttpCompletionOption.ResponseHeadersRead, cancellationToken).ConfigureAwait(false);
345 }
346 finally
347 {
348 // prevent content param from getting disposed
349 request.Content = null;
350 }
351 }
352
353 try
354 {
355 await Task.WhenAll(requestLoggers.Select(x => x.LogResponse(response, cancellationToken))).ConfigureAwait(false);
356
357 // just stream
358 if (fileDownload && response.IsSuccessStatusCode)
359 return (TResult)(object)await CachedResponseStream.Create(response).ConfigureAwait(false);
360 }
361 catch

365 }

400 async Task<bool> RefreshToken(CancellationToken cancellationToken)
401 {
402 if (tokenRefreshHeaders == null)
403 return false;
404
405 var startingToken = headers.Token;
406 await semaphoreSlim.WaitAsync(cancellationToken).ConfigureAwait(false);
407 try
408 {
409 if (startingToken != headers.Token)
410 return true;
411
412 var token = await RunRequest<object, TokenResponse>(Routes.Root, new object(), HttpMethod.Post, null, true, cancellationToken).ConfigureAwait(false);
413 headers = new ApiHeaders(headers.UserAgent!, token.Bearer!);
414 }
415 catch (ClientException)
416 {
417 return false;
418 }
419 finally
420 {
421 semaphoreSlim.Release();
422 }
423
424 return true;
425 }
