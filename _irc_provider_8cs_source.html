167 public override async ValueTask SendMessage(Message? replyTo, MessageContent message, ulong channelId, CancellationToken cancellationToken)
168 {
169 ArgumentNullException.ThrowIfNull(message);
170
171 await Task.Factory.StartNew(
172 () =>
173 {
174 // IRC doesn't allow newlines
175 // Explicitly ignore embeds
176 var messageText = message.Text;
177 messageText ??= $"Embed Only: {JsonConvert.SerializeObject(message.Embed)}";
178
179 messageText = String.Concat(
180 messageText
181 .Where(x => x != '\r')
182 .Select(x => x == '\n' ? '|' : x));
183
184 var channelName = channelIdMap[channelId];
185 SendType sendType;
186 if (channelName == null)
187 {
188 channelName = queryChannelIdMap[channelId];
189 sendType = SendType.Notice;
190 }
191 else
192 sendType = SendType.Message;
193
194 var messageSize = Encoding.UTF8.GetByteCount(messageText) + Encoding.UTF8.GetByteCount(channelName) + PreambleMessageLength;
195 var messageTooLong = messageSize > MessageBytesLimit;
196 if (messageTooLong)
197 messageText = $"TGS: Could not send message to IRC. Line write exceeded protocol limit of {MessageBytesLimit}B.";
198
199 try
200 {
201 client.SendMessage(sendType, channelName, messageText);
202 }
203 catch (Exception e)
204 {
205 Logger.LogWarning(e, "Unable to send to channel {channelName}!", channelName);
206 return;
207 }
208
209 if (messageTooLong)
210 Logger.LogWarning(
211 "Failed to send to channel {channelId}: Message size ({messageSize}B) exceeds IRC limit of 512B",
212 channelId,
213 messageSize);
214 },
215 cancellationToken,
216 DefaultIOManager.BlockingTaskCreationOptions,
217 TaskScheduler.Current);
