314  async void UpdateServer()
315  {
316  try
317  {
318  UseWaitCursor = true;
319  Enabled = false;
320  try
321  {
322  string res = null;
323  var repo = Instance.Repository;
324  var pulls = await Task.Run(() => repo.MergedPullRequests(out res));
325 
326  if (pulls == null)
327  {
328  MessageBox.Show(res);
329  return;
330  }
331 
332  List<Task<PullRequest>> pullsRequests = null;
333  if (Program.GetRepositoryRemote(repo, out string remoteOwner, out string remoteName))
334  {
335  //find out which of the PRs have been merged
336  pullsRequests = new List<Task<PullRequest>>();
337  foreach (var I in pulls)
338  pullsRequests.Add(ghclient.PullRequest.Get(remoteOwner, remoteName, I.Number));
339  }
340 
341  res = await Task.Run(() => repo.Update(true));
342 
343  if (res != null)
344  {
345  MessageBox.Show(res, "Error updating repository");
346  return;
347  }
348 
349  await Task.Run(() => repo.GenerateChangelog(out res));
350 
351  if (res != null)
352  MessageBox.Show(res, "Error generating changelog");
353 
354  res = await Task.Run(() => repo.SynchronizePush());
355 
356  if (res != null)
357  MessageBox.Show(res, "Error synchronizing commits");
358 
359  if (pullsRequests != null)
360  Task.WaitAll(pullsRequests.ToArray());
361 
362  foreach (var I in pullsRequests)
363  if (I.Result.Merged)
364  pulls.RemoveAll(x => x.Number == I.Result.Number);
365 
366  var mergeResults = await Task.Run(() => repo.MergePullRequests(pulls, true));
367  var compileStartResult = await Task.Run(() => Instance.Compiler.Compile(true));
368 
369  //Show any errors
370  for (var I = 0; I < mergeResults.Count(); ++I)
371  {
372  var err = mergeResults.ElementAt(I);
373  if (err != null)
374  MessageBox.Show(err, String.Format("Error re-merging PR #{0}", pulls[I].Number));
375  }
376 
377  if (!compileStartResult)
378  MessageBox.Show(res, "Error starting compile!");
379  }
380  finally
381  {
382  UseWaitCursor = false;
383  Enabled = true;
384  }
385  }
386  catch (ForbiddenException)
387  {
388  if (ghclient.Credentials.AuthenticationType == AuthenticationType.Anonymous)
389  {
390  if (Program.RateLimitPrompt(ghclient))
391  UpdateServer();
392  return;
393  }
394  else
395  throw;
396  }
397  LoadServerPage();
398  }
