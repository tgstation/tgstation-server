313  async void UpdateServer()
314  {
315  try
316  {
317  UseWaitCursor = true;
318  Enabled = false;
319  try
320  {
321  string res = null;
322  var repo = Interface.GetComponent<ITGRepository>();
323  var pulls = await Task.Factory.StartNew(() => repo.MergedPullRequests(out res));
324 
325  if (pulls == null)
326  {
327  MessageBox.Show(res);
328  return;
329  }
330 
331  List<Task<PullRequest>> pullsRequests = null;
332  if(Program.GetRepositoryRemote(repo, out string remoteOwner, out string remoteName)) {
333  //find out which of the PRs have been merged
334  pullsRequests = new List<Task<PullRequest>>();
335  foreach (var I in pulls)
336  pullsRequests.Add(ghclient.PullRequest.Get(remoteOwner, remoteName, I.Number));
337  }
338 
339  res = await Task.Factory.StartNew(() => repo.Update(true));
340 
341  if(res != null)
342  {
343  MessageBox.Show(res, "Error updating repository");
344  return;
345  }
346 
347  await Task.Factory.StartNew(() => repo.GenerateChangelog(out res));
348 
349  if (res != null)
350  MessageBox.Show(res, "Error generating changelog");
351 
352  res = await Task.Factory.StartNew(() => repo.SynchronizePush());
353 
354  if (res != null)
355  MessageBox.Show(res, "Error synchronizing commits");
356 
357  if (pullsRequests != null)
358  Task.WaitAll(pullsRequests.ToArray());
359 
360  foreach (var I in pullsRequests)
361  if (I.Result.Merged)
362  pulls.RemoveAll(x => x.Number == I.Result.Number);
363 
364  var results = new List<string>();
365  foreach (var I in pulls) {
366  retry:
367  await Task.Factory.StartNew(() => res = repo.MergePullRequest(I.Number, I.Sha));
368  if (res != null)
369  switch(MessageBox.Show(res, "Error Re-merging Pull Request", MessageBoxButtons.AbortRetryIgnore))
370  {
371  case DialogResult.Abort:
372  return;
373  case DialogResult.Retry:
374  goto retry;
375  }
376  }
377 
378  await Task.Factory.StartNew(() => Interface.GetComponent<ITGCompiler>().Compile(pulls.Count != 1));
379  if (res != null)
380  MessageBox.Show(res, "Error starting compile!");
381  }
382  finally
383  {
384  UseWaitCursor = false;
385  Enabled = true;
386  }
387  }
388  catch (ForbiddenException)
389  {
390  if (ghclient.Credentials.AuthenticationType == AuthenticationType.Anonymous)
391  {
392  if (Program.RateLimitPrompt(ghclient))
393  UpdateServer();
394  return;
395  }
396  else
397  throw;
398  }

400  }
