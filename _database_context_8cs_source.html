348  public async Task SchemaDowngradeForServerVersion(Version version, CancellationToken cancellationToken)
349  {
350  if (version == null)
351  throw new ArgumentNullException(nameof(version));
352  if (version < new Version(4, 0))
353  throw new ArgumentOutOfRangeException(nameof(version), version, "Not a valid V4 version!");
354 
355  // Update this with new migrations as they are made
356  string targetMigration = null;
357  if (version < new Version(4, 1, 0))
358  throw new NotSupportedException("Cannot migrate below version 4.1.0!");
359  else if (version < new Version(4, 2, 0))
360  targetMigration = DatabaseType == DatabaseType.Sqlite ? nameof(SLRebuild) : nameof(MSFixCascadingDelete);
361 
362  if (targetMigration == null)
363  {
364  Logger.LogDebug("No down migration required.");
365  return;
366  }
367 
368  string migrationSubstitution;
369  switch (DatabaseType)
370  {
371  case DatabaseType.SqlServer:
372  // already setup
373  migrationSubstitution = null;
374  break;
375  case DatabaseType.MySql:
376  case DatabaseType.MariaDB:
377  migrationSubstitution = "MY{0}";
378  break;
379  case DatabaseType.Sqlite:
380  migrationSubstitution = "SL{0}";
381  break;
382  default:
383  throw new InvalidOperationException($"Invalid DatabaseType: {DatabaseType}");
384  }
385 
386  if (migrationSubstitution != null)
387  targetMigration = String.Format(CultureInfo.InvariantCulture, migrationSubstitution, targetMigration.Substring(2));
388 
389  // even though it clearly implements it in the DatabaseFacade definition this won't work without casting ()
390  var dbServiceProvider = ((IInfrastructure<IServiceProvider>)Database).Instance;
391  var migrator = dbServiceProvider.GetRequiredService<IMigrator>();
392 
393  Logger.LogInformation("Migrating down to version {0}. Target: {1}", version, targetMigration);
394  try
395  {
396  await migrator.MigrateAsync(targetMigration, cancellationToken).ConfigureAwait(false);
397  }
398  catch (Exception e)
399  {
400  Logger.LogCritical("Failed to migrate! Exception: {0}", e);
401  }
402  }
