286  public async Task<bool> Migrate(ILogger<DatabaseContext> logger, CancellationToken cancellationToken)
287  {
288  if (logger == null)
289  throw new ArgumentNullException(nameof(logger));
290  var migrations = await Database.GetAppliedMigrationsAsync(cancellationToken).ConfigureAwait(false);
291  var wasEmpty = !migrations.Any();
292 
293  if (wasEmpty || (await Database.GetPendingMigrationsAsync(cancellationToken).ConfigureAwait(false)).Any())
294  {
295  logger.LogInformation("Migrating database...");
296  await Database.MigrateAsync(cancellationToken).ConfigureAwait(false);
297  }
298  else
299  logger.LogDebug("No migrations to apply");
300 
301  wasEmpty |= (await Users.AsQueryable().CountAsync(cancellationToken).ConfigureAwait(false)) == 0;
302 
303  return wasEmpty;
304  }
