159 public async Task<IActionResult> Update([FromBody] DreamDaemonRequest model, CancellationToken cancellationToken)
160 {
161 if (model == null)
162 throw new ArgumentNullException(nameof(model));
163
164 if (model.SoftShutdown == true && model.SoftRestart == true)
165 return BadRequest(new ErrorMessageResponse(ErrorCode.DreamDaemonDoubleSoft));
166
167 // alias for changing DD settings
168 var current = await DatabaseContext
169 .Instances
170 .AsQueryable()
171 .Where(x => x.Id == Instance.Id)
172 .Select(x => x.DreamDaemonSettings)
173 .FirstOrDefaultAsync(cancellationToken)
174 ;
175
176 if (current == default)
177 return Gone();
178
179 if (model.Port.HasValue && model.Port.Value != current.Port.Value)
180 {
181 var verifiedPort = await portAllocator
182 .GetAvailablePort(
183 model.Port.Value,
184 true,
185 cancellationToken)
186 ;
187 if (verifiedPort != model.Port)
188 return Conflict(new ErrorMessageResponse(ErrorCode.PortNotAvailable));
189 }
190
191 var userRights = (DreamDaemonRights)AuthenticationContext.GetRight(RightsType.DreamDaemon);
192
193 bool CheckModified<T>(Expression<Func<Api.Models.Internal.DreamDaemonSettings, T>> expression, DreamDaemonRights requiredRight)
194 {
195 var memberSelectorExpression = (MemberExpression)expression.Body;
196 var property = (PropertyInfo)memberSelectorExpression.Member;
197
198 var newVal = property.GetValue(model);
199 if (newVal == null)
200 return false;
201 if (!userRights.HasFlag(requiredRight) && property.GetValue(current) != newVal)
202 return true;
203
204 property.SetValue(current, newVal);
205 return false;
206 }
207
208 return await WithComponentInstance(
209 async instance =>
210 {
211 var watchdog = instance.Watchdog;
212 var rebootState = watchdog.RebootState;
213 var oldSoftRestart = rebootState == RebootState.Restart;
214 var oldSoftShutdown = rebootState == RebootState.Shutdown;
215
216 if (CheckModified(x => x.AllowWebClient, DreamDaemonRights.SetWebClient)
217 || CheckModified(x => x.AutoStart, DreamDaemonRights.SetAutoStart)
218 || CheckModified(x => x.Port, DreamDaemonRights.SetPort)
219 || CheckModified(x => x.SecurityLevel, DreamDaemonRights.SetSecurity)
220 || CheckModified(x => x.Visibility, DreamDaemonRights.SetVisibility)
221 || (model.SoftRestart.HasValue && !AuthenticationContext.InstancePermissionSet.DreamDaemonRights.Value.HasFlag(DreamDaemonRights.SoftRestart))
222 || (model.SoftShutdown.HasValue && !AuthenticationContext.InstancePermissionSet.DreamDaemonRights.Value.HasFlag(DreamDaemonRights.SoftShutdown))
223 || CheckModified(x => x.StartupTimeout, DreamDaemonRights.SetStartupTimeout)
224 || CheckModified(x => x.HeartbeatSeconds, DreamDaemonRights.SetHeartbeatInterval)
225 || CheckModified(x => x.DumpOnHeartbeatRestart, DreamDaemonRights.CreateDump)
226 || CheckModified(x => x.TopicRequestTimeout, DreamDaemonRights.SetTopicTimeout)
227 || CheckModified(x => x.AdditionalParameters, DreamDaemonRights.SetAdditionalParameters)
228 || CheckModified(x => x.StartProfiler, DreamDaemonRights.SetProfiler)
229 || CheckModified(x => x.LogOutput, DreamDaemonRights.SetLogOutput))
230 return Forbid();
231
232 await DatabaseContext.Save(cancellationToken);
233
234 // run this second because current may be modified by it
235 await watchdog.ChangeSettings(current, cancellationToken);
236
237 if (!oldSoftRestart && model.SoftRestart == true && watchdog.Status == WatchdogStatus.Online)
238 await watchdog.Restart(true, cancellationToken);
239 else if (!oldSoftShutdown && model.SoftShutdown == true)
240 await watchdog.Terminate(true, cancellationToken);
241 else if ((oldSoftRestart && model.SoftRestart == false) || (oldSoftShutdown && model.SoftShutdown == false))
242 await watchdog.ResetRebootState(cancellationToken);
243
244 return await ReadImpl(current, cancellationToken);
245 })
246 ;
247 }

325 Task<IActionResult> ReadImpl(DreamDaemonSettings settings, CancellationToken cancellationToken)
326 => WithComponentInstance(async instance =>
327 {
328 var dd = instance.Watchdog;
329
330 var metadata = (AuthenticationContext.GetRight(RightsType.DreamDaemon) & (ulong)DreamDaemonRights.ReadMetadata) != 0;
331 var revision = (AuthenticationContext.GetRight(RightsType.DreamDaemon) & (ulong)DreamDaemonRights.ReadRevision) != 0;
332
333 if (settings == null)
334 {
335 settings = await DatabaseContext
336 .Instances
337 .AsQueryable()
338 .Where(x => x.Id == Instance.Id)
339 .Select(x => x.DreamDaemonSettings)
340 .FirstOrDefaultAsync(cancellationToken)
341 ;
342 if (settings == default)
343 return Gone();
344 }
345
346 var result = new DreamDaemonResponse();
347 if (metadata)
348 {
349 var alphaActive = dd.AlphaIsActive;
350 var llp = dd.LastLaunchParameters;
351 var rstate = dd.RebootState;
352 result.AutoStart = settings.AutoStart.Value;
353 result.CurrentPort = llp?.Port.Value;
354 result.CurrentSecurity = llp?.SecurityLevel.Value;
355 result.CurrentVisibility = llp?.Visibility.Value;
356 result.CurrentAllowWebclient = llp?.AllowWebClient.Value;
357 result.Port = settings.Port.Value;
358 result.AllowWebClient = settings.AllowWebClient.Value;
359 result.Status = dd.Status;
360 result.SecurityLevel = settings.SecurityLevel.Value;
361 result.Visibility = settings.Visibility.Value;
362 result.SoftRestart = rstate == RebootState.Restart;
363 result.SoftShutdown = rstate == RebootState.Shutdown;
364 result.StartupTimeout = settings.StartupTimeout.Value;
365 result.HeartbeatSeconds = settings.HeartbeatSeconds.Value;
366 result.DumpOnHeartbeatRestart = settings.DumpOnHeartbeatRestart.Value;
367 result.TopicRequestTimeout = settings.TopicRequestTimeout.Value;
368 result.AdditionalParameters = settings.AdditionalParameters;
369 result.StartProfiler = settings.StartProfiler;
370 result.LogOutput = settings.LogOutput;
371 }
372
373 if (revision)
374 {
375 var latestCompileJob = instance.LatestCompileJob();
376 result.ActiveCompileJob = ((instance.Watchdog.Status != WatchdogStatus.Offline
377 ? dd.ActiveCompileJob
378 : latestCompileJob) ?? latestCompileJob)
379 ?.ToApi();
380 if (latestCompileJob?.Id != result.ActiveCompileJob?.Id)
381 result.StagedCompileJob = latestCompileJob?.ToApi();
382 }
383
384 return Json(result);
