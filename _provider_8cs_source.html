265 async Task ReconnectionLoop(uint reconnectInterval, bool connectNow, CancellationToken cancellationToken)
266 {
267 do
268 {
269 try
270 {
271 if (!connectNow)
272 await AsyncDelayer.Delay(TimeSpan.FromMinutes(reconnectInterval), cancellationToken);
273 else
274 connectNow = false;
275 if (!Connected)
276 {
277 var job = new Job
278 {
279 Description = $"Reconnect chat bot: {ChatBot.Name}",
280 CancelRight = (ulong)ChatBotRights.WriteEnabled,
281 CancelRightsType = RightsType.ChatBots,
282 Instance = ChatBot.Instance,
283 };
284
285 await jobManager.RegisterOperation(
286 job,
287 async (core, databaseContextFactory, paramJob, progressReporter, jobCancellationToken) =>
288 {
289 try
290 {
291 if (Connected)
292 {
293 Logger.LogTrace("Disconnecting...");
294 await DisconnectImpl(jobCancellationToken);
295 }
296 else
297 Logger.LogTrace("Already disconnected not doing disconnection attempt!");
298
299 Logger.LogTrace("Connecting...");
300 await Connect(jobCancellationToken);
301 Logger.LogTrace("Connected successfully");
302 EnqueueMessage(null);
303 }
304 catch
305 {
306 // we set this here because otherwise there could be stuff waiting on to connect us forever
307 initialConnectionTcs.TrySetResult();
308 throw;
309 }
310 },
311 cancellationToken);
312
313 await jobManager.WaitForJobCompletion(job, null, cancellationToken, cancellationToken);
314 }
315 }
316 catch (OperationCanceledException e)

319 }
320 catch (Exception e)

323 }
