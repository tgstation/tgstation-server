110 {
111 ArgumentNullException.ThrowIfNull(executeAction);

113 // validate the headers
114 if (ApiHeaders == null)
115 {
116 if (requireHeaders)
117 return HeadersIssue(ApiHeadersProvider.HeadersException!);
118 }
119
120 var errorCase = await ValidateRequest(cancellationToken);
121 if (errorCase != null)
122 return errorCase;
123
124 if (ModelState?.IsValid == false)
125 {
126 var errorMessages = ModelState
127 .SelectMany(x => x.Value!.Errors)
128 .Select(x => x.ErrorMessage)
129
130 // We use RequiredAttributes purely for preventing properties from becoming nullable in the databases
131 // We validate missing required fields in controllers
132 // Unfortunately, we can't remove the whole validator for that as it checks other things like StringLength
133 // This is the best way to deal with it unfortunately
134 .Where(x => !x.EndsWith(" field is required.", StringComparison.Ordinal));
135
136 if (errorMessages.Any())
137 return BadRequest(
138 new ErrorMessageResponse(ErrorCode.ModelValidationFailure)
139 {
140 AdditionalData = String.Join(Environment.NewLine, errorMessages),
141 });
142
143 ModelState.Clear();
144 }
145
146 using (ApiHeaders?.InstanceId != null
147 ? LogContext.PushProperty(SerilogContextHelper.InstanceIdContextProperty, ApiHeaders.InstanceId)
148 : null)
149 using (AuthenticationContext.Valid
150 ? LogContext.PushProperty(SerilogContextHelper.UserIdContextProperty, AuthenticationContext.User.Id)
151 : null)
152 using (LogContext.PushProperty(SerilogContextHelper.RequestPathContextProperty, $"{Request.Method} {Request.Path}"))
153 {
154 if (ApiHeaders != null)
155 {
156 var isGet = HttpMethods.IsGet(Request.Method);
157 Logger.Log(
158 isGet
159 ? LogLevel.Trace
160 : LogLevel.Debug,
161 "Starting API request: Version: {clientApiVersion}. {userAgentHeaderName}: {clientUserAgent}",
162 ApiHeaders.ApiVersion.Semver(),
163 HeaderNames.UserAgent,
164 ApiHeaders.RawUserAgent);
165 }
166 else if (Request.Headers.TryGetValue(HeaderNames.UserAgent, out var userAgents))
167 Logger.LogDebug(
168 "Starting unauthorized API request. {userAgentHeaderName}: {allUserAgents}",
169 HeaderNames.UserAgent,
170 userAgents);
171 else
172 Logger.LogDebug(
173 "Starting unauthorized API request. No {userAgentHeaderName}!",
174 HeaderNames.UserAgent);
175
176 await executeAction();
177 }
178
179 return null;
180 }

323 async ValueTask<IActionResult> PaginatedImpl<TModel, TResultModel>(
324 Func<ValueTask<PaginatableResult<TModel>>> queryGenerator,
325 Func<TResultModel, ValueTask>? resultTransformer,
326 int? pageQuery,
327 int? pageSizeQuery,
328 CancellationToken cancellationToken)
329 {
330 ArgumentNullException.ThrowIfNull(queryGenerator);
331
332 if (pageQuery <= 0 || pageSizeQuery <= 0)
333 return BadRequest(new ErrorMessageResponse(ErrorCode.ApiInvalidPageOrPageSize));
334
335 var pageSize = pageSizeQuery ?? DefaultPageSize;
336 if (pageSize > MaximumPageSize)
337 return BadRequest(new ErrorMessageResponse(ErrorCode.ApiPageTooLarge)
338 {
339 AdditionalData = $"Maximum page size: {MaximumPageSize}",
340 });
341
342 var page = pageQuery ?? 1;
343
344 var paginationResult = await queryGenerator();
345 if (!paginationResult.Valid)
346 return paginationResult.EarlyOut;
347
348 var queriedResults = paginationResult
349 .Results
350 .Skip((page - 1) * pageSize)
351 .Take(pageSize);
352
353 int totalResults;
354 List<TModel> pagedResults;
355 if (queriedResults.Provider is IAsyncQueryProvider)
356 {
357 totalResults = await paginationResult.Results.CountAsync(cancellationToken);
358 pagedResults = await queriedResults
359 .ToListAsync(cancellationToken);
360 }
361 else
362 {
363 totalResults = paginationResult.Results.Count();
364 pagedResults = queriedResults.ToList();
365 }
366
367 ICollection<TResultModel> finalResults;
368 if (typeof(TModel) == typeof(TResultModel))
369 finalResults = (List<TResultModel>)(object)pagedResults; // clearly a safe cast
370 else
371 finalResults = pagedResults
372 .OfType<IApiTransformable<TResultModel>>()
373 .Select(x => x.ToApi())
374 .ToList();
375
376 if (resultTransformer != null)
377 foreach (var finalResult in finalResults)
378 await resultTransformer(finalResult);
379
380 var carryTheOne = totalResults % pageSize != 0
381 ? 1
382 : 0;
383 return Json(
384 new PaginatedResponse<TResultModel>
385 {
386 Content = finalResults,
387 PageSize = pageSize,
388 TotalPages = (ushort)(totalResults / pageSize) + carryTheOne,
389 TotalItems = totalResults,
390 });
391 }
