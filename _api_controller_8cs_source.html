100 public override async Task OnActionExecutionAsync(ActionExecutingContext context, ActionExecutionDelegate next)
101 {
102 if (context == null)
103 throw new ArgumentNullException(nameof(context));
104
105 // ALL valid token and login requests that match a route go through this function
106 // 404 is returned before
107 if (AuthenticationContext != null && AuthenticationContext.User == null)
108 {
109 // valid token, expired password
110 await Unauthorized().ExecuteResultAsync(context);
111 return;
112 }
113
114 // validate the headers
115 try
116 {
117 ApiHeaders = new ApiHeaders(Request.GetTypedHeaders());
118
119 if (!ApiHeaders.Compatible())
120 {
121 await StatusCode(
122 HttpStatusCode.UpgradeRequired,
123 new ErrorMessageResponse(ErrorCode.ApiMismatch))
124 .ExecuteResultAsync(context)
125 ;
126 return;
127 }
128
129 var errorCase = await ValidateRequest(context.HttpContext.RequestAborted);
130 if (errorCase != null)
131 {
132 await errorCase.ExecuteResultAsync(context);
133 return;
134 }
135 }
136 catch (HeadersException)
137 {
138 if (requireHeaders)
139 {
140 await HeadersIssue(false)
141 .ExecuteResultAsync(context)
142 ;
143 return;
144 }
145 }
146
147 if (ModelState?.IsValid == false)
148 {
149 var errorMessages = ModelState
150 .SelectMany(x => x.Value.Errors)
151 .Select(x => x.ErrorMessage)
152
153 // We use RequiredAttributes purely for preventing properties from becoming nullable in the databases
154 // We validate missing required fields in controllers
155 // Unfortunately, we can't remove the whole validator for that as it checks other things like StringLength
156 // This is the best way to deal with it unfortunately
157 .Where(x => !x.EndsWith(" field is required.", StringComparison.Ordinal));
158
159 if (errorMessages.Any())
160 {
161 await BadRequest(
162 new ErrorMessageResponse(ErrorCode.ModelValidationFailure)
163 {
164 AdditionalData = String.Join(Environment.NewLine, errorMessages),
165 })
166 .ExecuteResultAsync(context);
167 return;
168 }
169
170 ModelState.Clear();
171 }
172
173 using (ApiHeaders?.InstanceId != null
174 ? LogContext.PushProperty("Instance", ApiHeaders.InstanceId)
175 : null)
176 using (AuthenticationContext != null
177 ? LogContext.PushProperty("User", AuthenticationContext.User.Id)
178 : null)
179 using (LogContext.PushProperty("Request", $"{Request.Method} {Request.Path}"))
180 {
181 if (ApiHeaders != null)
182 Logger.LogDebug(
183 "Starting API request: Version: {clientApiVersion}. {userAgentHeaderName}: {clientUserAgent}",
184 ApiHeaders.ApiVersion.Semver(),
185 HeaderNames.UserAgent,
186 ApiHeaders.RawUserAgent);
187 else if (Request.Headers.TryGetValue(HeaderNames.UserAgent, out var userAgents))
188 Logger.LogDebug(
189 "Starting unauthorized API request. {userAgentHeaderName}: {allUserAgents}",
190 HeaderNames.UserAgent,
191 userAgents);
192 else
193 Logger.LogDebug(
194 "Starting unauthorized API request. No {userAgentHeaderName}!",
195 HeaderNames.UserAgent);
196 await base.OnActionExecutionAsync(context, next);
197 }

352 async Task<IActionResult> PaginatedImpl<TModel, TResultModel>(
353 Func<Task<PaginatableResult<TModel>>> queryGenerator,
354 Func<TResultModel, Task> resultTransformer,
355 int? pageQuery,
356 int? pageSizeQuery,
357 CancellationToken cancellationToken)
358 {
359 if (queryGenerator == null)
360 throw new ArgumentNullException(nameof(queryGenerator));
361
362 if (pageQuery <= 0 || pageSizeQuery <= 0)
363 return BadRequest(new ErrorMessageResponse(ErrorCode.ApiInvalidPageOrPageSize));
364
365 var pageSize = pageSizeQuery ?? DefaultPageSize;
366 if (pageSize > MaximumPageSize)
367 return BadRequest(new ErrorMessageResponse(ErrorCode.ApiPageTooLarge)
368 {
369 AdditionalData = $"Maximum page size: {MaximumPageSize}",
370 });
371
372 var page = pageQuery ?? 1;
373
374 var paginationResult = await queryGenerator();
375 if (paginationResult.EarlyOut != null)
376 return paginationResult.EarlyOut;
377
378 var queriedResults = paginationResult
379 .Results
380 .Skip((page - 1) * pageSize)
381 .Take(pageSize);
382
383 int totalResults;
384 List<TModel> pagedResults;
385 if (queriedResults.Provider is IAsyncQueryProvider)
386 {
387 totalResults = await paginationResult.Results.CountAsync(cancellationToken);
388 pagedResults = await queriedResults
389 .ToListAsync(cancellationToken)
390 ;
391 }
392 else
393 {
394 totalResults = paginationResult.Results.Count();
395 pagedResults = queriedResults.ToList();
396 }
397
398 ICollection<TResultModel> finalResults;
399 if (typeof(TModel) == typeof(TResultModel))
400 finalResults = (List<TResultModel>)(object)pagedResults; // clearly a safe cast
401 else
402 finalResults = pagedResults
403 .OfType<IApiTransformable<TResultModel>>()
404 .Select(x => x.ToApi())
405 .ToList();
406
407 if (resultTransformer != null)
408 foreach (var finalResult in finalResults)
409 await resultTransformer(finalResult);
410
411 var carryTheOne = totalResults % pageSize != 0
412 ? 1
413 : 0;
414 return Json(
415 new PaginatedResponse<TResultModel>
416 {
417 Content = finalResults,
418 PageSize = pageSize,
419 TotalPages = (ushort)(totalResults / pageSize) + carryTheOne,
420 TotalItems = totalResults,
421 });
422 }
