166 public override async ValueTask TrustDmbPath(EngineVersion version, string fullDmbPath, CancellationToken cancellationToken)
167 {
168 ArgumentNullException.ThrowIfNull(version);
169 ArgumentNullException.ThrowIfNull(fullDmbPath);
170
171 var byondDir = PathToUserFolder;
172 var cfgDir = IOManager.ConcatPath(
173 byondDir,
174 CfgDirectoryName);
175 var trustedFilePath = IOManager.ConcatPath(
176 cfgDir,
177 TrustedDmbFileName);
178
179 Logger.LogDebug("Adding .dmb ({dmbPath}) to {trustedFilePath}", fullDmbPath, trustedFilePath);
180
181 using (await SemaphoreSlimContext.Lock(UserFilesSemaphore, cancellationToken))
182 {
183 string trustedFileText;
184 var filePreviouslyExisted = await IOManager.FileExists(trustedFilePath, cancellationToken);
185 if (filePreviouslyExisted)
186 {
187 var trustedFileBytes = await IOManager.ReadAllBytes(trustedFilePath, cancellationToken);
188 trustedFileText = Encoding.UTF8.GetString(trustedFileBytes);
189 trustedFileText = $"{trustedFileText.Trim()}{Environment.NewLine}";
190 }
191 else
192 trustedFileText = String.Empty;
193
194 if (trustedFileText.Contains(fullDmbPath, StringComparison.Ordinal))
195 return;
196
197 trustedFileText = $"{trustedFileText}{fullDmbPath}{Environment.NewLine}";
198
199 var newTrustedFileBytes = Encoding.UTF8.GetBytes(trustedFileText);
200
201 if (!filePreviouslyExisted)
202 await IOManager.CreateDirectory(cfgDir, cancellationToken);
203
204 await IOManager.WriteAllBytes(trustedFilePath, newTrustedFileBytes, cancellationToken);
205 }
