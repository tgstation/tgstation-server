930 async Task UpdateSubmodules(JobProgressReporter progressReporter, string username, string password, CancellationToken cancellationToken)
931 {
932 var submoduleCount = libGitRepo.Submodules.Count();
933 if (submoduleCount == 0)
934 {
935 logger.LogTrace("No submodules, skipping update");
936 return;
937 }
938
939 logger.LogTrace("Updating submodules with{0} credentials...", username == null ? "out" : String.Empty);
940
941 var iteration = 0;
942 var factor = 100 / submoduleCount;
943 foreach (var submodule in libGitRepo.Submodules)
944 {
945 void LocalProgressReporter(string stage, int? percentage) => progressReporter(stage, percentage.HasValue ? (int?)((iteration * factor) + (percentage.Value / submoduleCount)) : null);
946 var submoduleUpdateOptions = new SubmoduleUpdateOptions
947 {
948 Init = true,
949 OnTransferProgress = TransferProgressHandler(
950 (stage, progress) => LocalProgressReporter(stage, progress.Value / 2),
951 $"Fetch submodule {submodule.Name}",
952 cancellationToken),
953 OnProgress = output => !cancellationToken.IsCancellationRequested,
954 OnUpdateTips = (a, b, c) => !cancellationToken.IsCancellationRequested,
955 CredentialsProvider = credentialsProvider.GenerateCredentialsHandler(username, password),
956 OnCheckoutProgress = CheckoutProgressHandler(
957 (stage, progress) => LocalProgressReporter(stage, progress.HasValue ? (int?)(50 + (progress.Value / 2)) : null),
958 $"Checkout submodule {submodule.Name}"),
959 };
960
961 logger.LogDebug("Updating submodule {0}...", submodule.Name);
962 Task RawSubModuleUpdate() => Task.Factory.StartNew(
963 () => libGitRepo.Submodules.Update(submodule.Name, submoduleUpdateOptions),
964 cancellationToken,
965 DefaultIOManager.BlockingTaskCreationOptions,
966 TaskScheduler.Current);
967 try
968 {
969 await RawSubModuleUpdate().ConfigureAwait(false);
970 }
971 catch (LibGit2SharpException ex)
972 {
973 // workaround for https://github.com/libgit2/libgit2/issues/3820
974 // kill off the modules/ folder in .git and try again
975 credentialsProvider.CheckBadCredentialsException(ex);
976 logger.LogWarning(ex, "Initial update of submodule {0} failed. Deleting submodule directories and re-attempting...", submodule.Name);
977
978 await Task.WhenAll(
979 ioMananger.DeleteDirectory($".git/modules/{submodule.Path}", cancellationToken),
980 ioMananger.DeleteDirectory(submodule.Path, cancellationToken))
981 .ConfigureAwait(false);
982
983 logger.LogTrace("Second update attempt for submodule {0}...", submodule.Name);
984 try
985 {
986 await RawSubModuleUpdate().ConfigureAwait(false);
987 }
988 catch (UserCancelledException)
989 {
990 cancellationToken.ThrowIfCancellationRequested();
991 }
992 catch (LibGit2SharpException ex2)
993 {
994 credentialsProvider.CheckBadCredentialsException(ex2);
995 logger.LogTrace(ex2, "Retried update of submodule {0} failed!", submodule.Name);
996 throw new AggregateException(ex, ex2);
997 }
998 }
999
1000 await eventConsumer.HandleEvent(EventType.RepoSubmoduleUpdate, new List<string> { submodule.Name }, cancellationToken).ConfigureAwait(false);
