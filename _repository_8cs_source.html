338 PrettifyMessage = true,
339 }),

577 trackedBranch.FriendlyName,

581 {
582 CommitOnSuccess = true,
583 FailOnConflict = true,

587 });
588

590
591 if (result.Status == MergeStatus.Conflicts)
592 {
593 logger.LogDebug("Merge conflict, aborting and reverting to {oldHeadFriendlyName}", oldHead.FriendlyName);
594 progressReporter.ReportProgress(0);
595 libGitRepo.Reset(ResetMode.Hard, oldTip, new CheckoutOptions
596 {
597 OnCheckoutProgress = CheckoutProgressHandler(progressReporter.CreateSection($"Hard Reset to {oldHead.FriendlyName}", 1.0)),
598 });
599 cancellationToken.ThrowIfCancellationRequested();
600 }
601
602 libGitRepo.RemoveUntrackedFiles();
603 },
604 cancellationToken,
605 DefaultIOManager.BlockingTaskCreationOptions,
606 TaskScheduler.Current);
607
608 if (result!.Status == MergeStatus.Conflicts)
609 {
610 await eventConsumer.HandleEvent(
611 EventType.RepoMergeConflict,
612 new List<string>
613 {
614 oldTip.Sha,
615 trackedBranch!.Tip.Sha,
616 oldHead.FriendlyName ?? UnknownReference,
617 trackedBranch.FriendlyName,
618 },
619 deploymentPipeline,
620 cancellationToken);

622 }

1045 {

1047 if (submoduleCount == 0)
1048 {
1049 logger.LogTrace("No submodules, skipping update");
1050 return;
1051 }

1053 var factor = 1.0 / submoduleCount / 3;
1054 foreach (var submodule in parentRepository.Submodules)
1055 {
1056 logger.LogTrace("Entering submodule {name} ({path}) for recursive updates...", submodule.Name, submodule.Path);
1057 var submoduleUpdateOptions = new SubmoduleUpdateOptions
1058 {
1059 Init = true,
1060 OnCheckoutNotify = (_, _) => !cancellationToken.IsCancellationRequested,
1061 };
1062
1063 submoduleUpdateOptions.FetchOptions.Hydrate(
1064 logger,
1065 currentProgressReporter?.CreateSection($"Fetch submodule {submodule.Name}", factor),
1066 credentialsProvider.GenerateCredentialsHandler(username, password),
1067 cancellationToken);
1068
1069 if (currentProgressReporter != null)
1070 submoduleUpdateOptions.OnCheckoutProgress = CheckoutProgressHandler(
1071 currentProgressReporter.CreateSection($"Checkout submodule {submodule.Name}", factor));
1072
1073 logger.LogDebug("Updating submodule {submoduleName}...", submodule.Name);
1074 Task RawSubModuleUpdate() => Task.Factory.StartNew(
1075 () => parentRepository.Submodules.Update(submodule.Name, submoduleUpdateOptions),
1076 cancellationToken,
1077 DefaultIOManager.BlockingTaskCreationOptions,

1081 {
1082 await RawSubModuleUpdate();
1083 }

1087 // kill off the modules/ folder in .git and try again

1089 credentialsProvider.CheckBadCredentialsException(ex);

1091

1093 ioManager.DeleteDirectory($".git/modules/{submodule.Path}", cancellationToken),
1094 ioManager.DeleteDirectory(submodule.Path, cancellationToken));

1098 {
1099 await RawSubModuleUpdate();

1114 EventType.RepoSubmoduleUpdate,
1115 new List<string> { submodule.Name },
1116 deploymentPipeline,
1117 cancellationToken);

1120 ioManager.ConcatPath(
1121 parentGitDirectory,
1122 submodule.Path));
1123
1124 using var submoduleRepo = await submoduleFactory.CreateFromPath(
1125 submodulePath,
1126 cancellationToken);
1127 await RecursiveUpdateSubmodules(
1128 submoduleRepo,
1129 currentProgressReporter?.CreateSection($"Entering submodule \"{submodule.Name}\"...", factor),
1130 submodulePath);
1131 }

1142 CheckoutProgressHandler CheckoutProgressHandler(JobProgressReporter progressReporter) => (a, completedSteps, totalSteps) =>
1143 {
1144 double? percentage;
1145
1146 // short circuit initialization where totalSteps is 0
1147 if (completedSteps == 0)
1148 percentage = 0;
1149 else if (totalSteps < completedSteps || totalSteps == 0)
1150 percentage = null;
1151 else
1152 {
1153 percentage = ((double)completedSteps) / totalSteps;
1154 if (percentage < 0)
1155 percentage = null;
1156 }
1157
1158 if (percentage == null)
1159 logger.LogDebug(
1160 "Bad checkout progress values (Please tell Dominion)! Completeds: {completed}, Total: {total}",
1161 completedSteps,
1162 totalSteps);
