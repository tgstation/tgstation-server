98 {
99 if (telemetryConfiguration.DisableVersionReporting)

101 logger.LogDebug("Version telemetry disabled");
102 return;
103 }
104
105 if (!telemetryConfiguration.VersionReportingRepositoryId.HasValue)
106 {
107 logger.LogError("Version reporting repository is misconfigured. Telemetry cannot be sent!");
108 return;
109 }
110
111 var attribute = TelemetryAppSerializedKeyAttribute.Instance;
112 if (attribute == null)
113 {
114 logger.LogDebug("TGS build configuration does not allow for version telemetry");
115 return;
116 }
117
118 logger.LogDebug("Starting...");
119
120 try
121 {
122 var telemetryIdFile = ioManager.ResolvePath(
123 ioManager.ConcatPath(
124 ioManager.GetPathInLocalDirectory(assemblyInformationProvider),
125 "telemetry.id"));
126
127 Guid telemetryId;
128 if (!await ioManager.FileExists(telemetryIdFile, stoppingToken))
129 {
130 telemetryId = Guid.NewGuid();
131 await ioManager.WriteAllBytes(telemetryIdFile, Encoding.UTF8.GetBytes(telemetryId.ToString()), stoppingToken);
132 logger.LogInformation("Generated telemetry ID {telemetryId} and wrote to {file}", telemetryId, telemetryIdFile);
133 }
134 else
135 {
136 var contents = await ioManager.ReadAllBytes(telemetryIdFile, stoppingToken);
137
138 string guidStr;
139 try
140 {
141 guidStr = Encoding.UTF8.GetString(contents);
142 }
143 catch (Exception ex)
144 {
145 logger.LogError(ex, "Cannot decode telemetry ID from installation file ({path}). Telemetry will not be sent!", telemetryIdFile);
146 return;
147 }
148
149 if (!Guid.TryParse(guidStr, out telemetryId))
150 {
151 logger.LogError("Cannot parse telemetry ID from installation file ({path}). Telemetry will not be sent!", telemetryIdFile);
152 return;
153 }
154 }
155
156 try
157 {
158 while (!stoppingToken.IsCancellationRequested)
159 {
160 var nextDelayHours = await TryReportVersion(
161 telemetryId,
162 attribute.SerializedKey,
163 telemetryConfiguration.VersionReportingRepositoryId.Value,
164 false,
165 stoppingToken)
166 ? 24
167 : 1;
168
169 logger.LogDebug("Next version report in {hours} hours", nextDelayHours);
170 await asyncDelayer.Delay(TimeSpan.FromHours(nextDelayHours), stoppingToken);
171 }
172 }
173 catch (OperationCanceledException ex)
174 {
175 logger.LogTrace(ex, "Inner cancellation");
176 }
177
178 shutdownCancellationToken.ThrowIfCancellationRequested();
179
180 logger.LogDebug("Sending shutdown telemetry");
181 await TryReportVersion(
182 telemetryId,
183 attribute.SerializedKey,
184 telemetryConfiguration.VersionReportingRepositoryId.Value,
185 true,
186 shutdownCancellationToken);
187 }
188 catch (OperationCanceledException ex)
189 {
190 logger.LogTrace(ex, "Exiting due to outer cancellation...");
191 }
192 catch (Exception ex)

194 logger.LogError(ex, "Crashed!");
195 }

207 async ValueTask<bool> TryReportVersion(Guid telemetryId, string serializedPem, long repositoryId, bool shutdown, CancellationToken cancellationToken)
208 {
209 logger.LogDebug("Sending version telemetry...");
210
211 var serverFriendlyName = telemetryConfiguration.ServerFriendlyName;
212 if (String.IsNullOrWhiteSpace(serverFriendlyName))
213 serverFriendlyName = null;
214
215 logger.LogTrace(
216 "Repository ID: {repoId}, Server friendly name: {friendlyName}",
217 repositoryId,
218 serverFriendlyName == null
219 ? "(null)"
220 : $"\"{serverFriendlyName}\"");
221 try
222 {
223 var gitHubClient = await gitHubClientFactory.CreateInstallationClient(
224 serializedPem,
225 repositoryId,
226 cancellationToken);
227
228 if (gitHubClient == null)
229 {
230 logger.LogWarning("Could not create GitHub client to connect to repository ID {repoId}!", repositoryId);
231 return false;
232 }
233
234 // remove this lookup once https://github.com/octokit/octokit.net/pull/2960 is merged and released
235 var repository = await gitHubClient.Repository.Get(repositoryId);
236
237 logger.LogTrace("Repository ID {id} resolved to {owner}/{name}", repositoryId, repository.Owner.Name, repository.Name);
238
239 var inputs = new Dictionary<string, object>
240 {
241 { "telemetry_id", telemetryId.ToString() },
242 { "tgs_semver", assemblyInformationProvider.Version.Semver().ToString() },
243 { "shutdown", shutdown ? "true" : "false" },
244 };
245
246 if (serverFriendlyName != null)
247 inputs.Add("server_friendly_name", serverFriendlyName);
248
249 await gitHubClient.Actions.Workflows.CreateDispatch(
250 repository.Owner.Login,
251 repository.Name,
252 ".github/workflows/tgs_deployments_telemetry.yml",
253 new CreateWorkflowDispatch("main")
254 {
255 Inputs = inputs,
256 });
257
258 logger.LogTrace("Telemetry sent successfully");
259
260 return true;
261 }
262 catch (Exception ex)
263 {
264 logger.LogWarning(ex, "Failed to report version!");
265 return false;
266 }
267 }
