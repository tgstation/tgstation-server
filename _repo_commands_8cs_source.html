75  public RepoStatusCommand()
76  {
77  Keyword = "status";
78  }
79  protected override ExitCode Run(IList<string> parameters)
80  {
81  var Repo = Interface.GetComponent<ITGRepository>();
82  var busy = Repo.OperationInProgress();
83  if (!busy)
84  {
85  OutputProc("Repo: Idle");
86  var head = Repo.GetHead(false, out string error);
87  if (head == null)
88  head = "Error: " + error;
89  var remotehead = Repo.GetHead(true, out error);
90  if (remotehead == null)
91  remotehead = "Error: " + error;
92  var branch = Repo.GetBranch(out error);
93  if (branch == null)
94  branch = "Error: " + error;
95  var remote = Repo.GetRemote(out error);
96  if (remote == null)
97  remote = "Error: " + error;
98  OutputProc("Remote: " + remote + " (" + remotehead + ")");
99  OutputProc("Branch: " + branch);
100  OutputProc("HEAD: " + head);
101  OutputProc(String.Format("Committer Identity: {0} ({1})", Repo.GetCommitterName(), Repo.GetCommitterEmail()));
102  }
103  else
104  {
105  OutputProc("Repo: Busy");
106  var progress = Repo.CheckoutProgress();
107  if (progress != -1)
108  {
109  var eqs = "";
110  for (var I = 0; I < progress / 10; ++I)
111  eqs += "=";
112  var dshs = "";
113  for (var I = 0; I < 10 - (progress / 10); ++I)
114  eqs += "-";
115  OutputProc(String.Format("Progress: [{0}{1}] {2}%", eqs, dshs, progress));
116  }
117  }
118  return ExitCode.Normal;
119  }
120  public override string GetHelpText()
121  {
122  return "Shows the busy status of the repo, remote, branch, and HEAD information";
123  }
