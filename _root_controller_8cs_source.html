70 static Tuple<string, string?> GetControlPanelActionLink(Expression<Func<ControlPanelController, IActionResult>> actionExpression)
71 {
72 var memberSelectorExpression = (MethodCallExpression)actionExpression.Body;
73 var method = memberSelectorExpression.Method;
74
75 var controllerName = typeof(ControlPanelController).Name;
76
77 const string ControllerSuffix = nameof(Controller);
78 if (controllerName.EndsWith(ControllerSuffix, StringComparison.Ordinal))
79 controllerName = controllerName.Substring(0, controllerName.Length - ControllerSuffix.Length);

81 var actionName = method.Name;
82
83 return Tuple.Create<string, string?>(controllerName, actionName);
