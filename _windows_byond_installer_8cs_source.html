186 {

188 if (installedDirectX)

190 Logger.LogTrace("DirectX already installed.");

195
196 // always install it, it's pretty fast and will do better redundancy checking than us
197 var rbdx = IOManager.ConcatPath(path, ByondDXDir);
198
199 try
200 {
201 // noShellExecute because we aren't doing runas shennanigans
202 await using var directXInstaller = processExecutor.LaunchProcess(

207

230 async ValueTask AddDreamDaemonToFirewall(Version version, string path, CancellationToken cancellationToken)
231 {
232 var dreamDaemonName = GetDreamDaemonName(version, out var usesDDExe, out var _);
233
234 var dreamDaemonPath = IOManager.ResolvePath(
235 IOManager.ConcatPath(
236 path,
237 ByondManager.BinPath,
238 dreamDaemonName));
239
240 Logger.LogInformation("Adding Windows Firewall exception for {path}...", dreamDaemonPath);
241 try
242 {
243 // I really wish we could add the instance name here but
244 // 1. It'd make IByondInstaller need to be transient per-instance and WindowsByondInstaller relys on being a singleton for its DX installer call
245 // 2. The instance could be renamed, so it'd have to be an unfriendly ID anyway.
246 var arguments = $"advfirewall firewall add rule name=\"TGS DreamDaemon {version}\" program=\"{dreamDaemonPath}\" protocol=tcp dir=in enable=yes action=allow";
247 await using var netshProcess = processExecutor.LaunchProcess(
248 "netsh.exe",
249 IOManager.ResolvePath(),
250 arguments,
251 readStandardHandles: true,
252 noShellExecute: true);
253
254 int exitCode;
255 using (cancellationToken.Register(() => netshProcess.Terminate()))
256 exitCode = (await netshProcess.Lifetime).Value;
257 cancellationToken.ThrowIfCancellationRequested();
258
259 Logger.LogDebug(
260 "netsh.exe output:{newLine}{output}",
261 Environment.NewLine,
262 await netshProcess.GetCombinedOutput(cancellationToken));
263
264 if (exitCode != 0)
265 throw new JobException(ErrorCode.ByondDreamDaemonFirewallFail, new JobException($"Invalid exit code: {exitCode}"));
266
267 if (usesDDExe)
268 await IOManager.WriteAllBytes(
269 IOManager.ConcatPath(path, TgsFirewalledDDFile),
270 Array.Empty<byte>(),
271 cancellationToken);
272 }
273 catch (Exception ex)
274 {
275 throw new JobException(ErrorCode.ByondDreamDaemonFirewallFail, ex);
276 }
