132 public override async Task UpgradeInstallation(Version version, string path, CancellationToken cancellationToken)

134 if (version == null)

178 async Task InstallDirectX(string path, CancellationToken cancellationToken)
179 {
180 using var lockContext = await SemaphoreSlimContext.Lock(semaphore, cancellationToken);
181 if (installedDirectX)
182 {
183 Logger.LogTrace("DirectX already installed.");
184 return;
185 }
186
187 Logger.LogTrace("Installing DirectX redistributable...");
188
189 // always install it, it's pretty fast and will do better redundancy checking than us
190 var rbdx = IOManager.ConcatPath(path, ByondDXDir);
191
192 try
193 {
194 // noShellExecute because we aren't doing runas shennanigans
195 await using var directXInstaller = await processExecutor.LaunchProcess(
196 IOManager.ConcatPath(rbdx, "DXSETUP.exe"),
197 rbdx,
198 "/silent",
199 noShellExecute: true);
200
201 int exitCode;
202 using (cancellationToken.Register(() => directXInstaller.Terminate()))
203 exitCode = await directXInstaller.Lifetime;
204 cancellationToken.ThrowIfCancellationRequested();
205
206 if (exitCode != 0)
207 throw new JobException(ErrorCode.ByondDirectXInstallFail, new JobException($"Invalid exit code: {exitCode}"));
208 installedDirectX = true;
209 }
210 catch (Exception e)
211 {
212 throw new JobException(ErrorCode.ByondDirectXInstallFail, e);
213 }

223 async Task AddDreamDaemonToFirewall(Version version, string path, CancellationToken cancellationToken)
224 {
225 var dreamDaemonName = GetDreamDaemonName(version, out var supportsCli);
226
227 var dreamDaemonPath = IOManager.ResolvePath(
228 IOManager.ConcatPath(
229 path,
230 ByondManager.BinPath,
231 dreamDaemonName));
232
233 Logger.LogInformation("Adding Windows Firewall exception for {path}...", dreamDaemonPath);
234 try
235 {
236 await using var netshProcess = await processExecutor.LaunchProcess(
237 "netsh.exe",
238 IOManager.ResolvePath(),
239 $"advfirewall firewall add rule name=\"TGS DreamDaemon\" program=\"{dreamDaemonPath}\" protocol=tcp dir=in enable=yes action=allow",
240 readStandardHandles: true,
241 noShellExecute: true);
242
243 int exitCode;
244 using (cancellationToken.Register(() => netshProcess.Terminate()))
245 exitCode = await netshProcess.Lifetime;
246 cancellationToken.ThrowIfCancellationRequested();
247
248 Logger.LogDebug(
249 "netsh.exe output:{newLine}{output}",
250 Environment.NewLine,
251 await netshProcess.GetCombinedOutput(cancellationToken));
252
253 if (exitCode != 0)
254 throw new JobException(ErrorCode.ByondDreamDaemonFirewallFail, new JobException($"Invalid exit code: {exitCode}"));
255
256 if (supportsCli)
257 await IOManager.WriteAllBytes(
258 IOManager.ConcatPath(path, TgsFirewalledDDFile),
259 Array.Empty<byte>(),
260 cancellationToken);
261 }
262 catch (Exception ex)
263 {
264 throw new JobException(ErrorCode.ByondDreamDaemonFirewallFail, ex);
265 }
