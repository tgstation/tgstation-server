577#pragma warning restore CA1502
578 {
579 if (!provider.Connected)
580 {
581 logger.LogTrace("Abort message processing because provider is disconnected!");
582 return;
583 }
584
585 // provider reconnected, remap channels.
586 if (message == null)
587 {
588 await RemapProvider(provider, cancellationToken).ConfigureAwait(false);
589 return;
590 }
591
592 // map the channel if it's private and we haven't seen it
593 var providerChannelId = message.User.Channel.RealId;
594 KeyValuePair<ulong, ChannelMapping>? mappedChannel;
595 long providerId;
596 lock (providers)
597 {
598 // important, otherwise we could end up processing during shutdown
599 cancellationToken.ThrowIfCancellationRequested();
600
601 providerId = providers
602 .Where(x => x.Value == provider)
603 .Select(x => x.Key)
604 .First();
605 mappedChannel = mappedChannels
606 .Where(x => x.Value.ProviderId == providerId && x.Value.ProviderChannelId == providerChannelId)
607 .Select(x => (KeyValuePair<ulong, ChannelMapping>?)x)
608 .FirstOrDefault();
609 }
610
611 if (message.User.Channel.IsPrivateChannel)
612 lock (mappedChannels)
613 if (!mappedChannel.HasValue)
614 {
615 ulong newId;
616 lock (synchronizationLock)
617 newId = channelIdCounter++;
618 logger.LogTrace(
619 "Mapping private channel {0}:{1} as {2}",
620 message.User.Channel.ConnectionName,
621 message.User.FriendlyName,
622 newId);
623 mappedChannels.Add(newId, new ChannelMapping
624 {
625 IsWatchdogChannel = false,
626 ProviderChannelId = message.User.Channel.RealId,
627 ProviderId = providerId,
628 Channel = message.User.Channel,
629 });
630 message.User.Channel.RealId = newId;
631 }
632 else
633 message.User.Channel.RealId = mappedChannel.Value.Key;
634 else
635 {
636 if (!mappedChannel.HasValue)
637 {
638 logger.LogError(
639 "Error mapping message: Provider ID: {providerId}, Channel Real ID: {realId}",
640 providerId,
641 message.User.Channel.RealId);
642 await SendMessage(
643 "Processing error, check logs!",
644 new List<ulong>
645 {
646 message.User.Channel.RealId,
647 },
648 cancellationToken)
649 .ConfigureAwait(false);
650 return;
651 }
652
653 var mappingChannelRepresentation = mappedChannel.Value.Value.Channel;
654
655 message.User.Channel.Id = mappingChannelRepresentation.Id;
656 message.User.Channel.Tag = mappingChannelRepresentation.Tag;
657 message.User.Channel.IsAdminChannel = mappingChannelRepresentation.IsAdminChannel;
658 }
659
660 var splits = new List<string>(message.Content.Trim().Split(' '));
661 var address = splits[0];
662 if (address.Length > 1 && (address.Last() == ':' || address.Last() == ','))
663 address = address[0..^1];
664
665 address = address.ToUpperInvariant();
666
667 var addressed =
668 address == CommonMention.ToUpperInvariant()
669 || address == provider.BotMention.ToUpperInvariant();
670
671 // no mention
672 if (!addressed && !message.User.Channel.IsPrivateChannel)
673 return;
674
675 logger.LogTrace(
676 "Start processing command: {0}. User (True provider Id): {1}",
677 message.Content,
678 JsonConvert.SerializeObject(message.User));
679 try
680 {
681 if (addressed)
682 splits.RemoveAt(0);
683
684 if (splits.Count == 0)
685 {
686 // just a mention
687 await SendMessage("Hi!", new List<ulong> { message.User.Channel.RealId }, cancellationToken).ConfigureAwait(false);
688 return;
689 }

692 splits.RemoveAt(0);
693 var arguments = String.Join(" ", splits);

696 {
697 if (!builtinCommands.TryGetValue(commandName, out var handler))
698 {
699 handler = trackingContexts
700 .Where(x => x.CustomCommands != null)
701 .SelectMany(x => x.CustomCommands)
702 .Where(x => x.Name.ToUpperInvariant() == commandName)
703 .FirstOrDefault();
704 }
705
706 return handler;
707 }
708
709 const string UnknownCommandMessage = "Unknown command! Type '?' or 'help' for available commands.";
710
711 if (command == "HELP" || command == "?")
712 {
713 string helpText;
714 if (splits.Count == 0)
715 {
716 var allCommands = builtinCommands.Select(x => x.Value).ToList();
717 allCommands.AddRange(
718 trackingContexts
719 .Where(x => x.CustomCommands != null)
720 .SelectMany(
721 x => x.CustomCommands));
722 helpText = String.Format(CultureInfo.InvariantCulture, "Available commands (Type '?' or 'help' and then a command name for more details): {0}", String.Join(", ", allCommands.Select(x => x.Name)));
723 }
724 else
725 {
726 var helpHandler = GetCommand(splits[0].ToUpperInvariant());
727 if (helpHandler != default)
728 helpText = String.Format(CultureInfo.InvariantCulture, "{0}: {1}{2}", helpHandler.Name, helpHandler.HelpText, helpHandler.AdminOnly ? " - May only be used in admin channels" : String.Empty);
729 else
730 helpText = UnknownCommandMessage;
731 }
732
733 await SendMessage(helpText, new List<ulong> { message.User.Channel.RealId }, cancellationToken).ConfigureAwait(false);
734 return;
735 }
736
737 var commandHandler = GetCommand(command);
738
739 if (commandHandler == default)
740 {
741 await SendMessage(UnknownCommandMessage, new List<ulong> { message.User.Channel.RealId }, cancellationToken).ConfigureAwait(false);
742 return;
743 }
744
745 if (commandHandler.AdminOnly && !message.User.Channel.IsAdminChannel)
746 {
747 await SendMessage("Use this command in an admin channel!", new List<ulong> { message.User.Channel.RealId }, cancellationToken).ConfigureAwait(false);
748 return;
749 }
750
751 var result = await commandHandler.Invoke(arguments, message.User, cancellationToken).ConfigureAwait(false);
752 if (result != null)
753 await SendMessage(result, new List<ulong> { message.User.Channel.RealId }, cancellationToken).ConfigureAwait(false);
754 }
755 catch (OperationCanceledException ex)
756 {
757 logger.LogTrace(ex, "Command processing canceled!");
758 throw;
759 }
760 catch (Exception e)
761 {
762 // error bc custom commands should reply about why it failed
763 logger.LogError(e, "Error processing chat command");

765 "TGS: Internal error processing command! Check server logs!",
766 new List<ulong> { message.User.Channel.RealId },
767 cancellationToken)
768 .ConfigureAwait(false);
769 }
770 finally
771 {
772 logger.LogTrace("Done processing command.");
773 }

781 async Task MonitorMessages(CancellationToken cancellationToken)
782 {
783 logger.LogTrace("Starting processing loop...");
784 var messageTasks = new Dictionary<IProvider, Task<Message>>();
785 Task activeProcessingTask = Task.CompletedTask;
786 try
787 {
788 Task updatedTask = null;
789 while (!cancellationToken.IsCancellationRequested)
790 {
791 if (updatedTask?.IsCompleted != false)
792 lock (synchronizationLock)
793 updatedTask = connectionsUpdated.Task;
794
795 // prune disconnected providers
796 foreach (var disposedProviderMessageTaskKvp in messageTasks.Where(x => x.Key.Disposed).ToList())
797 messageTasks.Remove(disposedProviderMessageTaskKvp.Key);
798
799 // add new ones
800 lock (providers)
801 foreach (var providerKvp in providers)
802 if (!messageTasks.ContainsKey(providerKvp.Value))
803 messageTasks.Add(providerKvp.Value, providerKvp.Value.NextMessage(cancellationToken));
804
805 if (messageTasks.Count == 0)
806 {
807 logger.LogTrace("No providers active, pausing messsage monitoring...");
808 await updatedTask.WithToken(cancellationToken).ConfigureAwait(false);
809 logger.LogTrace("Resuming message monitoring...");
810 continue;
811 }
812
813 // wait for a message
814 await Task.WhenAny(updatedTask, Task.WhenAny(messageTasks.Select(x => x.Value))).ConfigureAwait(false);
815
816 // process completed ones
817 foreach (var completedMessageTaskKvp in messageTasks.Where(x => x.Value.IsCompleted).ToList())
818 {
819 var message = await completedMessageTaskKvp.Value.ConfigureAwait(false);
820 var messageNumber = Interlocked.Increment(ref messagesProcessed);
821
822 async Task WrapProcessMessage()
823 {
824 var localActiveProcessingTask = activeProcessingTask;
825 using (LogContext.PushProperty("ChatMessage", messageNumber))
826 try
827 {
828 await ProcessMessage(completedMessageTaskKvp.Key, message, cancellationToken).ConfigureAwait(false);
829 }
830 catch (Exception ex)
831 {
832 logger.LogError(ex, "Error processing message {messageNumber}!", messageNumber);
833 }
834
835 await localActiveProcessingTask.ConfigureAwait(false);
836 }
837
838 activeProcessingTask = WrapProcessMessage();
839
840 messageTasks.Remove(completedMessageTaskKvp.Key);
841 }
842 }
843 }
844 catch (OperationCanceledException ex)
845 {
846 logger.LogTrace(ex, "Message processing loop cancelled!");
847 }
848 catch (Exception e)
849 {
850 logger.LogError(e, "Message loop crashed!");
851 }
852 finally
853 {
854 await activeProcessingTask.ConfigureAwait(false);
855 }
856
857 logger.LogTrace("Leaving message processing loop");
858 }
