78 {
79 GitHubClient client;
80 bool cacheHit;
81 lock (clientCache)
82 {
83 string cacheKey;
84 if (String.IsNullOrWhiteSpace(accessToken))
85 {
86 accessToken = null;
87 cacheKey = DefaultCacheKey;
88 }
89 else
90 cacheKey = accessToken;
91
92 cacheHit = clientCache.TryGetValue(cacheKey, out var tuple);
93
94 var now = DateTimeOffset.UtcNow;
95 if (!cacheHit)
96 {
97 logger.LogTrace("Creating new GitHubClient...");
98 client = new GitHubClient(
99 new ProductHeaderValue(
100 assemblyInformationProvider.ProductInfoHeaderValue.Product.Name,
101 assemblyInformationProvider.ProductInfoHeaderValue.Product.Version));
102
103 if (accessToken != null)
104 client.Credentials = new Credentials(accessToken);
105
106 clientCache.Add(cacheKey, (client, now));
107 }
108 else
109 {
110 logger.LogTrace("Cache hit for GitHubClient");
111 client = tuple.Item1;
112 tuple.Item2 = now;
113 }
114
115 // Prune the cache
116 var purgeCount = 0U;
117 var purgeAfter = now.AddDays(-ClientCacheDays);
118 foreach (var key in clientCache.Keys.ToList())
119 {
120 if (key == cacheKey)
121 continue; // save the hash lookup
122
123 tuple = clientCache[key];
124 if (tuple.Item2 <= purgeAfter)
125 {
126 clientCache.Remove(key);
127 ++purgeCount;
128 }
129 }
130
131 if (purgeCount > 0)
132 logger.LogDebug(
133 "Pruned {count} expired GitHub client(s) from cache that haven't been used in {purgeAfterHours} days.",
134 purgeCount,
135 ClientCacheDays);
136 }
137
138 var rateLimitInfo = client.GetLastApiInfo()?.RateLimit;
139 if (rateLimitInfo != null)
140 if (rateLimitInfo.Remaining == 0)
141 logger.LogWarning(
142 "Requested GitHub client has no requests remaining! Limit resets at {resetTime}",
143 rateLimitInfo.Reset.ToString("o"));
144 else if (rateLimitInfo.Remaining < 25) // good luck hitting these lines on codecov
145 logger.LogWarning(
146 "Requested GitHub client has only {remainingRequests} requests remaining! Limit resets at {resetTime}",
147 rateLimitInfo.Remaining,
148 rateLimitInfo.Reset.ToString("o"));
149 else
150 logger.LogDebug(
151 "Requested GitHub client has {remainingRequests} requests remaining. Limit resets at {resetTime}",
152 rateLimitInfo.Remaining,
153 rateLimitInfo.Reset.ToString("o"));
154
155 return client;
156 }
