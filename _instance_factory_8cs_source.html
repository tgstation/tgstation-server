264 public async ValueTask<IInstance> CreateInstance(IBridgeRegistrar bridgeRegistrar, Models.Instance metadata)
265 {
266 ArgumentNullException.ThrowIfNull(bridgeRegistrar);
267 ArgumentNullException.ThrowIfNull(metadata);
268
269 // Create the ioManager for the instance
270 var instanceIoManager = CreateInstanceIOManager(metadata);
271
272 // various other ioManagers
273 var repoIoManager = new ResolvingIOManager(instanceIoManager, "Repository");
274 var byondIOManager = new ResolvingIOManager(instanceIoManager, "Byond");
275 var gameIoManager = CreateGameIOManager(instanceIoManager);
276 var diagnosticsIOManager = new ResolvingIOManager(instanceIoManager, "Diagnostics");
277 var configurationIoManager = new ResolvingIOManager(instanceIoManager, "Configuration");
278
279 var metricFactory = this.metricFactory.WithLabels(
280 new Dictionary<string, string>
281 {
282 { "instance_name", metadata.Name! },
283 { "instance_id", metadata.Id!.Value.ToString(CultureInfo.InvariantCulture) },
284 });
285
286 var configuration = new StaticFiles.Configuration(
287 configurationIoManager,
288 synchronousIOManager,
289 linkFactory,
290 processExecutor,
291 postWriteHandler,
292 platformIdentifier,
293 fileTransferService,
294 loggerFactory.CreateLogger<StaticFiles.Configuration>(),
295 metadata,
296 generalConfiguration,
297 sessionConfiguration);
298 var eventConsumer = new EventConsumer(configuration);
299 var repoManager = repositoryManagerFactory.CreateRepositoryManager(repoIoManager, eventConsumer);
300 try
301 {
302 var engineManager = new EngineManager(
303 byondIOManager,
304 engineInstaller,
305 eventConsumer,
306 loggerFactory.CreateLogger<EngineManager>());
307
308 var dmbFactory = new DmbFactory(
309 databaseContextFactory,
310 gameIoManager,
311 remoteDeploymentManagerFactory,
312 eventConsumer,
313 asyncDelayer,
314 loggerFactory.CreateLogger<DmbFactory>(),
315 metadata);
316 try
317 {
318 var commandFactory = new CommandFactory(assemblyInformationProvider, engineManager, repoManager, databaseContextFactory, dmbFactory, metadata);
319
320 var chatManager = chatFactory.CreateChatManager(commandFactory, metadata.ChatSettings);
321 try
322 {
323 var reattachInfoHandler = new SessionPersistor(
324 databaseContextFactory,
325 dmbFactory,
326 processExecutor,
327 loggerFactory.CreateLogger<SessionPersistor>(),
328 metadata);
329
330 var sessionControllerFactory = new SessionControllerFactory(
331 processExecutor,
332 engineManager,
333 topicClientFactory,
334 cryptographySuite,
335 assemblyInformationProvider,
336 gameIoManager,
337 diagnosticsIOManager,
338 chatManager,
339 networkPromptReaper,
340 platformIdentifier,
341 bridgeRegistrar,
342 serverPortProvider,
343 eventConsumer,
344 asyncDelayer,
345 dotnetDumpService,
346 metricFactory,
347 loggerFactory,
348 loggerFactory.CreateLogger<SessionControllerFactory>(),
349 sessionConfiguration,
350 metadata);
351
352 var watchdog = watchdogFactory.CreateWatchdog(
353 chatManager,
354 dmbFactory,
355 reattachInfoHandler,
356 sessionControllerFactory,
357 gameIoManager,
358 diagnosticsIOManager,
359 configuration, // watchdog doesn't need itself as an event consumer
360 remoteDeploymentManagerFactory,
361 metricFactory,
362 metadata,
363 metadata.DreamDaemonSettings!);
364 try
365 {
366 eventConsumer.SetWatchdog(watchdog);
367 commandFactory.SetWatchdog(watchdog);
368
369 Instance? instance = null;
370 var dreamMaker = new DreamMaker(
371 engineManager,
372 gameIoManager,
373 configuration,
374 sessionControllerFactory,
375 eventConsumer,
376 chatManager,
377 processExecutor,
378 dmbFactory,
379 repoManager,
380 remoteDeploymentManagerFactory,
381 asyncDelayer,
382 metricFactory,
383 loggerFactory.CreateLogger<DreamMaker>(),
384 sessionConfiguration,
385 metadata);
386
387 instance = new Instance(
388 metadata,
389 repoManager,
390 engineManager,
391 dreamMaker,
392 watchdog,
393 chatManager,
394 configuration,
395 dmbFactory,
396 jobManager,
397 eventConsumer,
398 remoteDeploymentManagerFactory,
399 asyncDelayer,
400 loggerFactory.CreateLogger<Instance>());
401
402 return instance;
403 }
404 catch
405 {
406 await watchdog.DisposeAsync();
407 throw;
408 }
409 }
410 catch
411 {
412 await chatManager.DisposeAsync();
413 throw;
414 }
415 }
416 catch
417 {
418 dmbFactory.Dispose();
419 throw;
420 }
421 }
422 catch
423 {
424 repoManager.Dispose();
425 throw;
426 }
427 }
