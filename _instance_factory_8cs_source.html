252 public async ValueTask<IInstance> CreateInstance(IBridgeRegistrar bridgeRegistrar, Models.Instance metadata)
253 {
254 ArgumentNullException.ThrowIfNull(bridgeRegistrar);
255 ArgumentNullException.ThrowIfNull(metadata);
256
257 // Create the ioManager for the instance
258 var instanceIoManager = CreateInstanceIOManager(metadata);
259
260 // various other ioManagers
261 var repoIoManager = new ResolvingIOManager(instanceIoManager, "Repository");
262 var byondIOManager = new ResolvingIOManager(instanceIoManager, "Byond");
263 var gameIoManager = CreateGameIOManager(instanceIoManager);
264 var diagnosticsIOManager = new ResolvingIOManager(instanceIoManager, "Diagnostics");
265 var configurationIoManager = new ResolvingIOManager(instanceIoManager, "Configuration");
266
267 var configuration = new StaticFiles.Configuration(
268 configurationIoManager,
269 synchronousIOManager,
270 linkFactory,
271 processExecutor,
272 postWriteHandler,
273 platformIdentifier,
274 fileTransferService,
275 loggerFactory.CreateLogger<StaticFiles.Configuration>(),
276 generalConfiguration,
277 sessionConfiguration);
278 var eventConsumer = new EventConsumer(configuration);
279 var repoManager = repositoryManagerFactory.CreateRepositoryManager(repoIoManager, eventConsumer);
280 try
281 {
282 var engineManager = new EngineManager(
283 byondIOManager,
284 engineInstaller,
285 eventConsumer,
286 loggerFactory.CreateLogger<EngineManager>());
287
288 var dmbFactory = new DmbFactory(
289 databaseContextFactory,
290 gameIoManager,
291 remoteDeploymentManagerFactory,
292 eventConsumer,
293 loggerFactory.CreateLogger<DmbFactory>(),
294 metadata);
295 try
296 {
297 var commandFactory = new CommandFactory(assemblyInformationProvider, engineManager, repoManager, databaseContextFactory, dmbFactory, metadata);
298
299 var chatManager = chatFactory.CreateChatManager(commandFactory, metadata.ChatSettings);
300 try
301 {
302 var reattachInfoHandler = new SessionPersistor(
303 databaseContextFactory,
304 dmbFactory,
305 processExecutor,
306 loggerFactory.CreateLogger<SessionPersistor>(),
307 metadata);
308
309 var sessionControllerFactory = new SessionControllerFactory(
310 processExecutor,
311 engineManager,
312 topicClientFactory,
313 cryptographySuite,
314 assemblyInformationProvider,
315 gameIoManager,
316 diagnosticsIOManager,
317 chatManager,
318 networkPromptReaper,
319 platformIdentifier,
320 bridgeRegistrar,
321 serverPortProvider,
322 eventConsumer,
323 asyncDelayer,
324 dotnetDumpService,
325 loggerFactory,
326 loggerFactory.CreateLogger<SessionControllerFactory>(),
327 sessionConfiguration,
328 metadata);
329
330 var watchdog = watchdogFactory.CreateWatchdog(
331 chatManager,
332 dmbFactory,
333 reattachInfoHandler,
334 sessionControllerFactory,
335 gameIoManager,
336 diagnosticsIOManager,
337 configuration, // watchdog doesn't need itself as an event consumer
338 remoteDeploymentManagerFactory,
339 metadata,
340 metadata.DreamDaemonSettings!);
341 try
342 {
343 eventConsumer.SetWatchdog(watchdog);
344 commandFactory.SetWatchdog(watchdog);
345
346 Instance? instance = null;
347 var dreamMaker = new DreamMaker(
348 engineManager,
349 gameIoManager,
350 configuration,
351 sessionControllerFactory,
352 eventConsumer,
353 chatManager,
354 processExecutor,
355 dmbFactory,
356 repoManager,
357 remoteDeploymentManagerFactory,
358 asyncDelayer,
359 loggerFactory.CreateLogger<DreamMaker>(),
360 sessionConfiguration,
361 metadata);
362
363 instance = new Instance(
364 metadata,
365 repoManager,
366 engineManager,
367 dreamMaker,
368 watchdog,
369 chatManager,
370 configuration,
371 dmbFactory,
372 jobManager,
373 eventConsumer,
374 remoteDeploymentManagerFactory,
375 asyncDelayer,
376 loggerFactory.CreateLogger<Instance>());
377
378 return instance;
379 }
380 catch
381 {
382 await watchdog.DisposeAsync();
383 throw;
384 }
385 }
386 catch
387 {
388 await chatManager.DisposeAsync();
389 throw;
390 }
391 }
392 catch
393 {
394 dmbFactory.Dispose();
395 throw;
396 }
397 }
398 catch
399 {
400 repoManager.Dispose();
401 throw;
402 }
403 }
