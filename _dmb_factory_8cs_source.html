249 compileJob.Job.StoppedAt = DateTimeOffset.UtcNow;

260 var newProvider = new DmbProvider(compileJob, ioManager, CleanupAction);

263 const string LegacyADirectoryName = "A";

327 public async Task CleanUnusedCompileJobs(CancellationToken cancellationToken)
328 {
329 List<long> jobIdsToSkip;
330
331 // don't clean locked directories
332 lock (jobLockCounts)
333 jobIdsToSkip = jobLockCounts.Select(x => x.Key).ToList();
334
335 List<string> jobUidsToNotErase = null;

337 // find the uids of locked directories
338 if (jobIdsToSkip.Any())

342 jobUidsToNotErase = (await db

344 .AsQueryable()
345 .Where(
346 x => x.Job.Instance.Id == metadata.Id
347 && jobIdsToSkip.Contains(x.Id.Value))
348 .Select(x => x.DirectoryName.Value)
349 .ToListAsync(cancellationToken))
350 .Select(x => x.ToString())
351 .ToList();
352 });
353 }
354 else
355 jobUidsToNotErase = new List<string>();
356
357 jobUidsToNotErase.Add(SwappableDmbProvider.LiveGameDirectory);
358
359 logger.LogTrace("We will not clean the following directories: {0}", String.Join(", ", jobUidsToNotErase));
360
361 // cleanup
362 var gameDirectory = ioManager.ResolvePath();
363 await ioManager.CreateDirectory(gameDirectory, cancellationToken);
364 var directories = await ioManager.GetDirectories(gameDirectory, cancellationToken);
365 int deleting = 0;
366 var tasks = directories.Select(async x =>
367 {
368 var nameOnly = ioManager.GetFileName(x);
369 if (jobUidsToNotErase.Contains(nameOnly))
370 return;
371 logger.LogDebug("Cleaning unused game folder: {0}...", nameOnly);
372 try
373 {
374 ++deleting;
375 await DeleteCompileJobContent(x, cancellationToken);
376 }
377 catch (OperationCanceledException)
378 {
379 throw;
380 }
381 catch (Exception e)
382 {
383 logger.LogWarning(e, "Error deleting directory {0}!", x);
384 }
385 }).ToList();
386 if (deleting > 0)
387 await Task.WhenAll(tasks);
388 }

403 void CleanRegisteredCompileJob(CompileJob job)
404 {
405 async Task HandleCleanup()
406 {
407 // First kill the GitHub deployment
408 var remoteDeploymentManager = remoteDeploymentManagerFactory.CreateRemoteDeploymentManager(metadata, job);
409
410 // DCT: None available
411 var deploymentJob = remoteDeploymentManager.MarkInactive(job, default);
412
413 var deleteTask = DeleteCompileJobContent(job.DirectoryName.ToString(), cleanupCts.Token);
414 var otherTask = cleanupTask;
415 await Task.WhenAll(otherTask, deleteTask, deploymentJob);
416 }
417
418 lock (jobLockCounts)
419 if (jobLockCounts.TryGetValue(job.Id.Value, out var currentVal))
420 if (currentVal == 1)
421 {
422 jobLockCounts.Remove(job.Id.Value);
423 logger.LogDebug("Cleaning lock-free compile job {0} => {1}", job.Id, job.DirectoryName);
424 cleanupTask = HandleCleanup();
425 }
426 else
427 {
428 var decremented = --jobLockCounts[job.Id.Value];
429 logger.LogTrace("Compile job {0} lock count now: {1}", job.Id, decremented);
430 }
431 else
432 logger.LogError("Extra Dispose of DmbProvider for CompileJob {compileJobId}!", job.Id);
433 }
