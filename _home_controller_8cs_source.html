222 public async Task<IActionResult> CreateToken(CancellationToken cancellationToken)
223 {
224 if (ApiHeaders == null)
225 {
226 Response.Headers.Add(HeaderNames.WWWAuthenticate, new StringValues("basic realm=\"Create TGS bearer token\""));
227 return HeadersIssue(false);
228 }
229
230 if (ApiHeaders.IsTokenAuthentication)
231 return BadRequest(new ErrorMessageResponse(ErrorCode.TokenWithToken));
232
233 var oAuthLogin = ApiHeaders.OAuthProvider.HasValue;
234
235 ISystemIdentity systemIdentity = null;
236 if (!oAuthLogin)
237 try
238 {
239 // trust the system over the database because a user's name can change while still having the same SID
240 systemIdentity = await systemIdentityFactory.CreateSystemIdentity(ApiHeaders.Username, ApiHeaders.Password, cancellationToken);
241 }
242 catch (NotImplementedException ex)
243 {
244 Logger.LogTrace(ex, "System identities not implemented!");
245 }
246
247 using (systemIdentity)
248 {
249 // Get the user from the database
250 IQueryable<Models.User> query = DatabaseContext.Users.AsQueryable();
251 if (oAuthLogin)
252 {
253 var oAuthProvider = ApiHeaders.OAuthProvider.Value;
254 string externalUserId;
255 try
256 {
257 var validator = oAuthProviders
258 .GetValidator(oAuthProvider);
259
260 if (validator == null)
261 return BadRequest(new ErrorMessageResponse(ErrorCode.OAuthProviderDisabled));
262
263 externalUserId = await validator
264 .ValidateResponseCode(ApiHeaders.Token, cancellationToken)
265 ;
266
267 Logger.LogTrace("External {oAuthProvider} UID: {externalUserId}", oAuthProvider, externalUserId);
268 }
269 catch (RateLimitExceededException ex)
270 {
271 return RateLimit(ex);
272 }
273
274 if (externalUserId == null)
275 return Unauthorized();
276
277 query = query.Where(
278 x => x.OAuthConnections.Any(
279 y => y.Provider == oAuthProvider
280 && y.ExternalUserId == externalUserId));
281 }
282 else
283 {
284 var canonicalUserName = Models.User.CanonicalizeName(ApiHeaders.Username);
285 if (canonicalUserName == Models.User.CanonicalizeName(Models.User.TgsSystemUserName))
286 return Unauthorized();
287
288 if (systemIdentity == null)
289 query = query.Where(x => x.CanonicalName == canonicalUserName);
290 else
291 query = query.Where(x => x.CanonicalName == canonicalUserName || x.SystemIdentifier == systemIdentity.Uid);
292 }
293
294 var users = await query
295 .Select(x => new Models.User
296 {
297 Id = x.Id,
298 PasswordHash = x.PasswordHash,
299 Enabled = x.Enabled,
300 Name = x.Name,
301 })
302 .ToListAsync(cancellationToken)
303 ;
304
305 // Pick the DB user first
306 var user = users
307 .OrderByDescending(dbUser => dbUser.SystemIdentifier == null)
308 .FirstOrDefault();
309
310 // No user? You're not allowed
311 if (user == null)
312 return Unauthorized();
313
314 // A system user may have had their name AND password changed to one in our DB...
315 // Or a DB user was created that had the same user/pass as a system user
316 // Dumb admins...
317 // FALLBACK TO THE DB USER HERE, DO NOT REVEAL A SYSTEM LOGIN!!!
318 // This of course, allows system users to discover TGS users in this (HIGHLY IMPROBABLE) case but that is not our fault
319 var originalHash = user.PasswordHash;
320 var isDbUser = originalHash != null;
321 bool usingSystemIdentity = systemIdentity != null && !isDbUser;
322 if (!oAuthLogin)
323 if (!usingSystemIdentity)
324 {
325 // DB User password check and update
326 if (!cryptographySuite.CheckUserPassword(user, ApiHeaders.Password))
327 return Unauthorized();
328 if (user.PasswordHash != originalHash)
329 {
330 Logger.LogDebug("User ID {userId}'s password hash needs a refresh, updating database.", user.Id);
331 var updatedUser = new Models.User
332 {
333 Id = user.Id,
334 };
335 DatabaseContext.Users.Attach(updatedUser);
336 updatedUser.PasswordHash = user.PasswordHash;
337 await DatabaseContext.Save(cancellationToken);
338 }
339 }
340 else if (systemIdentity.Username != user.Name)
341 {
342 // System identity username change update
343 Logger.LogDebug("User ID {userId}'s system identity needs a refresh, updating database.", user.Id);
344 DatabaseContext.Users.Attach(user);
345 user.Name = systemIdentity.Username;
346 user.CanonicalName = Models.User.CanonicalizeName(user.Name);
347 await DatabaseContext.Save(cancellationToken);
348 }
349
350 // Now that the bookeeping is done, tell them to fuck off if necessary
351 if (!user.Enabled.Value)
352 {
353 Logger.LogTrace("Not logging in disabled user {userId}.", user.Id);
354 return Forbid();
355 }
356
357 var token = await tokenFactory.CreateToken(user, oAuthLogin, cancellationToken);
358 if (usingSystemIdentity)
359 {
360 // expire the identity slightly after the auth token in case of lag
361 var identExpiry = token.ExpiresAt;
362 identExpiry += tokenFactory.ValidationParameters.ClockSkew;
363 identExpiry += TimeSpan.FromSeconds(15);
364 identityCache.CacheSystemIdentity(user, systemIdentity, identExpiry);
365 }
366
367 Logger.LogDebug("Successfully logged in user {userId}!", user.Id);
368
369 return Json(token);
370 }
371 }
