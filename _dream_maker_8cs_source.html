419 async ValueTask<TimeSpan?> CalculateExpectedDeploymentTime(IDatabaseContext databaseContext, CancellationToken cancellationToken)
420 {
421 var previousCompileJobs = await databaseContext
422 .CompileJobs
423 .AsQueryable()
424 .Where(x => x.Job.Instance!.Id == metadata.Id)
425 .OrderByDescending(x => x.Job.StoppedAt)
426 .Take(10)
427 .Select(x => new
428 {
429 StoppedAt = x.Job.StoppedAt!.Value,
430 StartedAt = x.Job.StartedAt!.Value,
431 })
432 .ToListAsync(cancellationToken);
433
434 TimeSpan? averageSpan = null;
435 if (previousCompileJobs.Count != 0)
436 {
437 var totalSpan = TimeSpan.Zero;
438 foreach (var previousCompileJob in previousCompileJobs)
439 totalSpan += previousCompileJob.StoppedAt - previousCompileJob.StartedAt;
440 averageSpan = totalSpan / previousCompileJobs.Count;
441 }
442
443 return averageSpan;
444 }

708 async ValueTask ProgressTask(JobProgressReporter progressReporter, TimeSpan? estimatedDuration, CancellationToken cancellationToken)
709 {
710 double? lastReport = estimatedDuration.HasValue ? 0 : null;
711 progressReporter.ReportProgress(lastReport);
712
713 var minimumSleepInterval = TimeSpan.FromMilliseconds(250);
714 var sleepInterval = estimatedDuration.HasValue ? estimatedDuration.Value / 100 : minimumSleepInterval;
715
716 if (estimatedDuration.HasValue)
717 {
718 logger.LogDebug("Compile is expected to take: {estimatedDuration}", estimatedDuration);
719 }
720 else
721 {
722 logger.LogTrace("No metric to estimate compile time.");
723 }
724
725 try
726 {
727 for (var iteration = 0; iteration < (estimatedDuration.HasValue ? 99 : Int32.MaxValue); ++iteration)
728 {
729 if (estimatedDuration.HasValue)
730 {
731 var nextInterval = DateTimeOffset.UtcNow + sleepInterval;
732 do
733 {
734 var remainingSleepThisInterval = nextInterval - DateTimeOffset.UtcNow;
735 var nextSleepSpan = remainingSleepThisInterval < minimumSleepInterval ? minimumSleepInterval : remainingSleepThisInterval;
736
737 await asyncDelayer.Delay(nextSleepSpan, cancellationToken);
738 progressReporter.ReportProgress(lastReport);
739 }
740 while (DateTimeOffset.UtcNow < nextInterval);
741 }
742 else
743 await asyncDelayer.Delay(minimumSleepInterval, cancellationToken);
744
745 lastReport = estimatedDuration.HasValue ? sleepInterval * (iteration + 1) / estimatedDuration.Value : null;
746 progressReporter.ReportProgress(lastReport);
747 }
748 }
749 catch (OperationCanceledException ex)
750 {
751 logger.LogTrace(ex, "ProgressTask aborted.");
752 }
753 catch (Exception ex)
754 {
755 logger.LogError(ex, "ProgressTask crashed!");
756 }
757 }

837 case ApiValidationStatus.Incompatible:
838 throw new JobException(ErrorCode.DeploymentInvalidValidation);
839 case ApiValidationStatus.UnaskedValidationRequest:
840 default:
841 throw new InvalidOperationException(
842 $"Session controller returned unexpected ApiValidationStatus: {validationStatus}");
843 }

890 async ValueTask ModifyDme(Models.CompileJob job, CancellationToken cancellationToken)
891 {
892 var dmeFileName = String.Join('.', job.DmeName, DmeExtension);
893 var stringDirectoryName = job.DirectoryName!.Value.ToString();
894 var dmePath = ioManager.ConcatPath(stringDirectoryName, dmeFileName);
895 var dmeReadTask = ioManager.ReadAllBytes(dmePath, cancellationToken);
896
897 var dmeModificationsTask = configuration.CopyDMFilesTo(
898 dmeFileName,
899 ioManager.ResolvePath(
900 ioManager.ConcatPath(
901 stringDirectoryName,
902 ioManager.GetDirectoryName(dmeFileName))),
903 cancellationToken);
904
905 var dmeBytes = await dmeReadTask;
906 var dme = Encoding.UTF8.GetString(dmeBytes);
907
908 var dmeModifications = await dmeModificationsTask;
909
910 if (dmeModifications == null || dmeModifications.TotalDmeOverwrite)
911 {
912 if (dmeModifications != null)
913 logger.LogDebug(".dme replacement configured!");
914 else
915 logger.LogTrace("No .dme modifications required.");
916 return;
917 }
918
919 var dmeLines = new List<string>(dme.Split('\n', StringSplitOptions.None));
920 for (var dmeLineIndex = 0; dmeLineIndex < dmeLines.Count; ++dmeLineIndex)
921 {
922 var line = dmeLines[dmeLineIndex];
923 if (line.Contains("BEGIN_INCLUDE", StringComparison.Ordinal) && dmeModifications.HeadIncludeLine != null)
924 {
925 var headIncludeLineNumber = dmeLineIndex + 1;
926 logger.LogDebug(
