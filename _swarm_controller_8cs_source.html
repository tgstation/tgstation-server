214 public override async Task OnActionExecutionAsync(ActionExecutingContext context, ActionExecutionDelegate next)
215 {
216 using (LogContext.PushProperty(SerilogContextHelper.RequestPathContextProperty, $"{Request.Method} {Request.Path}"))
217 {
218 logger.LogTrace("Swarm request from {remoteIP}...", Request.HttpContext.Connection.RemoteIpAddress);
219 if (swarmConfiguration.PrivateKey == null)
220 {
221 logger.LogDebug("Attempted swarm request without private key configured!");
222 await Forbid().ExecuteResultAsync(context);
223 return;
224 }
225
226 if (!(Request.Headers.TryGetValue(SwarmConstants.ApiKeyHeader, out var apiKeyHeaderValues)
227 && apiKeyHeaderValues.Count == 1
228 && apiKeyHeaderValues.First() == swarmConfiguration.PrivateKey))
229 {
230 logger.LogDebug("Unauthorized swarm request!");
231 await Unauthorized().ExecuteResultAsync(context);
232 return;
233 }
234
235 if (!(Request.Headers.TryGetValue(SwarmConstants.RegistrationIdHeader, out var registrationHeaderValues)
236 && registrationHeaderValues.Count == 1
237 && Guid.TryParse(registrationHeaderValues.First(), out var registrationId)))
238 {
239 logger.LogDebug("Swarm request without registration ID!");
240 await BadRequest().ExecuteResultAsync(context);
241 return;
242 }
243
244 // we validate the registration itself on a case-by-case basis
245 if (ModelState?.IsValid == false)
246 {
247 var errorMessages = ModelState
248 .SelectMany(x => x.Value.Errors)
249 .Select(x => x.ErrorMessage);
250
251 logger.LogDebug(
252 "Swarm request model validation failed!{newLine}{messages}",
253 Environment.NewLine,
254 String.Join(Environment.NewLine, errorMessages));
255 await BadRequest().ExecuteResultAsync(context);
256 return;
257 }
258
259 logger.LogTrace("Starting swarm request processing...");
260 await base.OnActionExecutionAsync(context, next);
261 }
262 }
