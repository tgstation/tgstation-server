292 public Task<BridgeResponse> ProcessBridgeRequest(BridgeParameters parameters, CancellationToken cancellationToken)
293 {
294 if (parameters == null)
295 throw new ArgumentNullException(nameof(parameters));
296
297 static Task<BridgeResponse> Error(string message) => Task.FromResult(new BridgeResponse
298 {
299 ErrorMessage = message,
300 });
301
302 using (LogContext.PushProperty("Instance", metadata.Id))
303 {
304 logger.LogTrace("Handling bridge request...");
305 initialBridgeRequestTcs.TrySetResult(null);
306
307 var response = new BridgeResponse();
308 switch (parameters.CommandType)
309 {
310 case BridgeCommandType.ChatSend:
311 if (parameters.ChatMessage == null)
312 return Error("Missing chatMessage field!");
313
314 if (parameters.ChatMessage.ChannelIds == null)
315 return Error("Missing channelIds field in chatMessage!");
316
317 if (parameters.ChatMessage.ChannelIds.Any(channelIdString => !UInt64.TryParse(channelIdString, out var _)))
318 return Error("Invalid channelIds in chatMessage!");
319
320 if (parameters.ChatMessage.Text == null)
321 return Error("Missing message field in chatMessage!");
322
323 chat.QueueMessage(
324 parameters.ChatMessage.Text,
325 parameters.ChatMessage.ChannelIds.Select(UInt64.Parse));
326 break;
327 case BridgeCommandType.Prime:
328 var oldPrimeTcs = primeTcs;
329 primeTcs = new TaskCompletionSource<object>();
330 oldPrimeTcs.SetResult(null);
331 break;
332 case BridgeCommandType.Kill:
333 logger.LogInformation("Bridge requested process termination!");
334 TerminationWasRequested = true;
335 process.Terminate();
336 break;
337 case BridgeCommandType.PortUpdate:
338 lock (synchronizationLock)
339 {
340 if (!parameters.CurrentPort.HasValue)
341 {
343 logger.LogWarning("DreamDaemon sent new port command without providing it's own!");
344 return Error("Missing stringified port as data parameter!");
345 }
346
347 var currentPort = parameters.CurrentPort.Value;
348 if (!nextPort.HasValue)
349 ReattachInformation.Port = parameters.CurrentPort.Value; // not ready yet, so what we'll do is accept the random port DD opened on for now and change it later when we decide to
350 else
351 {
352 // nextPort is ready, tell DD to switch to that
353 // if it fails it'll kill itself
354 response.NewPort = nextPort.Value;
355 ReattachInformation.Port = nextPort.Value;
356 nextPort = null;
357
358 // we'll also get here from SetPort so complete that task
359 var tmpTcs = portAssignmentTcs;
360 portAssignmentTcs = null;
361 tmpTcs.SetResult(true);
362 }
363
364 portClosedForReboot = false;
365 }
366
367 break;
368 case BridgeCommandType.Startup:
369 apiValidationStatus = ApiValidationStatus.BadValidationRequest;
370 if (parameters.Version == null)
371 return Error("Missing dmApiVersion field!");
372
373 DMApiVersion = parameters.Version;
374 if (DMApiVersion.Major != DMApiConstants.InteropVersion.Major)
375 {
376 apiValidationStatus = ApiValidationStatus.Incompatible;
377 return Error("Incompatible dmApiVersion!");
378 }
379
380 switch (parameters.MinimumSecurityLevel)
381 {
382 case DreamDaemonSecurity.Ultrasafe:
383 apiValidationStatus = ApiValidationStatus.RequiresUltrasafe;
384 break;
385 case DreamDaemonSecurity.Safe:
386 apiValidationStatus = ApiValidationStatus.RequiresSafe;
387 break;
388 case DreamDaemonSecurity.Trusted:
389 apiValidationStatus = ApiValidationStatus.RequiresTrusted;
390 break;
391 case null:
392 return Error("Missing minimumSecurityLevel field!");
393 default:
394 return Error("Invalid minimumSecurityLevel!");
395 }
396
397 logger.LogTrace("ApiValidationStatus set to {0}", apiValidationStatus);
398
399 response.RuntimeInformation = new RuntimeInformation(
400 chatTrackingContext,
401 ReattachInformation.Dmb,
402 ReattachInformation.RuntimeInformation.ServerVersion,
403 ReattachInformation.RuntimeInformation.InstanceName,
404 ReattachInformation.RuntimeInformation.SecurityLevel,
405 ReattachInformation.RuntimeInformation.Visibility,
406 ReattachInformation.RuntimeInformation.ServerPort,
407 ReattachInformation.RuntimeInformation.ApiValidateOnly);
408
409 // Load custom commands
410 chatTrackingContext.CustomCommands = parameters.CustomCommands;
411 break;
412 case BridgeCommandType.Reboot:
413 if (ClosePortOnReboot)
414 {
415 chatTrackingContext.Active = false;
416 response.NewPort = 0;
417 portClosedForReboot = true;
418 }
419
420 var oldRebootTcs = rebootTcs;
421 rebootTcs = new TaskCompletionSource<object>();
422 oldRebootTcs.SetResult(null);
423 break;
424 case null:
425 response.ErrorMessage = "Missing commandType!";
426 break;
427 default:
428 response.ErrorMessage = "Requested commandType not supported!";
429 break;
430 }
431
432 return Task.FromResult(response);
433 }
434 }

455 public async Task<CombinedTopicResponse> SendCommand(TopicParameters parameters, CancellationToken cancellationToken)
456 {
457 if (parameters == null)
458 throw new ArgumentNullException(nameof(parameters));
459
460 if (Lifetime.IsCompleted)
461 {
462 logger.LogWarning(
463 "Attempted to send a command to an inactive SessionController: {0}",
464 parameters.CommandType);
465 return null;
466 }
467
468 if (!DMApiAvailable)
469 {
470 logger.LogTrace("Not sending topic request {0} to server without/with incompatible DMAPI!", parameters.CommandType);
471 return null;
472 }
473
474 parameters.AccessIdentifier = ReattachInformation.AccessIdentifier;
475
476 var json = JsonConvert.SerializeObject(parameters, DMApiConstants.SerializerSettings);
477 logger.LogTrace("Topic request: {0}", json);
478 try
479 {
480 var commandString = String.Format(
481 CultureInfo.InvariantCulture,
482 "?{0}={1}",
483 byondTopicSender.SanitizeString(DMApiConstants.TopicData),
484 byondTopicSender.SanitizeString(json));
485
486 var targetPort = ReattachInformation.Port;
487
488 var topicResponse = await byondTopicSender.SendTopic(
489 new IPEndPoint(IPAddress.Loopback, targetPort),
490 commandString,
491 cancellationToken);
492
493 var topicReturn = topicResponse.StringData;
494
495 Interop.Topic.TopicResponse interopResponse = null;
496 if (topicReturn != null)
497 try
498 {
499 interopResponse = JsonConvert.DeserializeObject<Interop.Topic.TopicResponse>(topicReturn, DMApiConstants.SerializerSettings);
500 if (interopResponse.ErrorMessage != null)
501 {
502 logger.LogWarning("Errored topic response for command {0}: {1}", parameters.CommandType, interopResponse.ErrorMessage);
503 }
504
505 logger.LogTrace("Interop response: {0}", topicReturn);
506 }
507 catch (Exception ex)
508 {
509 logger.LogWarning(ex, "Invalid interop response: {0}", topicReturn);
510 }
511
512 return new CombinedTopicResponse(topicResponse, interopResponse);
513 }
514 catch (OperationCanceledException ex)
515 {
516 logger.LogTrace(
517 ex,
518 "Topic request {0}!",
519 cancellationToken.IsCancellationRequested
520 ? "aborted"
521 : "timed out");
522 cancellationToken.ThrowIfCancellationRequested();
523 }
524 catch (Exception e)
525 {
526 logger.LogWarning(e, "Send command exception!");
527 }
528
529 return null;
530 }

634 async Task<LaunchResult> GetLaunchResult(
635 IAssemblyInformationProvider assemblyInformationProvider,
636 uint? startupTimeout,
637 bool reattached,
638 bool apiValidate)
639 {
640 var startTime = DateTimeOffset.UtcNow;
641 var useBridgeRequestForLaunchResult = !reattached && (apiValidate || DMApiAvailable);
642 var startupTask = useBridgeRequestForLaunchResult
643 ? initialBridgeRequestTcs.Task
644 : process.Startup;
645 var toAwait = Task.WhenAny(startupTask, process.Lifetime);
646
647 if (startupTimeout.HasValue)
648 toAwait = Task.WhenAny(toAwait, Task.Delay(TimeSpan.FromSeconds(startupTimeout.Value)));
649
650 logger.LogTrace(
651 "Waiting for LaunchResult based on {0}{1}...",
652 useBridgeRequestForLaunchResult ? "initial bridge request" : "process startup",
653 startupTimeout.HasValue ? $" with a timeout of {startupTimeout.Value}s" : String.Empty);
654
655 await toAwait;
656
657 var result = new LaunchResult
658 {
659 ExitCode = process.Lifetime.IsCompleted ? (int?)await process.Lifetime : null,
660 StartupTime = startupTask.IsCompleted ? (TimeSpan?)(DateTimeOffset.UtcNow - startTime) : null,
661 };
662
663 logger.LogTrace("Launch result: {0}", result);
664
665 if (!result.ExitCode.HasValue && reattached && !disposed)
666 {
667 var reattachResponse = await SendCommand(
668 new TopicParameters(
669 assemblyInformationProvider.Version,
670 ReattachInformation.RuntimeInformation.ServerPort),
671 reattachTopicCts.Token)
672 ;
673
674 if (reattachResponse != null)
675 {
676 if (reattachResponse.InteropResponse?.CustomCommands != null)
677 chatTrackingContext.CustomCommands = reattachResponse.InteropResponse.CustomCommands;
678 else if (reattachResponse.InteropResponse != null)
679 logger.LogWarning(
680 "DMAPI v{0} isn't returning the TGS custom commands list. Functionality added in v5.2.0.",
681 CompileJob.DMApiVersion.Semver());
682 }
683 }
684
685 return result;
686 }
