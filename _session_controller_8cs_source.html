536 async Task<LaunchResult> GetLaunchResult(
537 IAssemblyInformationProvider assemblyInformationProvider,
538 IAsyncDelayer asyncDelayer,
539 uint? startupTimeout,
540 bool reattached,
541 bool apiValidate)
542 {
543 var startTime = DateTimeOffset.UtcNow;
544 var useBridgeRequestForLaunchResult = !reattached && (apiValidate || DMApiAvailable);
545 var startupTask = useBridgeRequestForLaunchResult
546 ? initialBridgeRequestTcs.Task
547 : process.Startup;
548 var toAwait = Task.WhenAny(startupTask, process.Lifetime);
549
550 if (startupTimeout.HasValue)
551 toAwait = Task.WhenAny(toAwait, asyncDelayer.Delay(TimeSpan.FromSeconds(startupTimeout.Value), default)); // DCT: None available, task will clean up after delay
552
553 Logger.LogTrace(
554 "Waiting for LaunchResult based on {launchResultCompletionCause}{possibleTimeout}...",
555 useBridgeRequestForLaunchResult ? "initial bridge request" : "process startup",
556 startupTimeout.HasValue ? $" with a timeout of {startupTimeout.Value}s" : String.Empty);
557
558 await toAwait;
559
560 var result = new LaunchResult
561 {
562 ExitCode = process.Lifetime.IsCompleted ? await process.Lifetime : null,
563 StartupTime = startupTask.IsCompleted ? (DateTimeOffset.UtcNow - startTime) : null,
564 };
565
566 Logger.LogTrace("Launch result: {launchResult}", result);
567
568 if (!result.ExitCode.HasValue && reattached && !disposed)
569 {
570 var reattachResponse = await SendCommand(
571 new TopicParameters(
572 assemblyInformationProvider.Version,
573 ReattachInformation.RuntimeInformation.ServerPort),
574 reattachTopicCts.Token);
575
576 if (reattachResponse != null)
577 {
578 if (reattachResponse?.CustomCommands != null)
579 chatTrackingContext.CustomCommands = reattachResponse.CustomCommands;
580 else if (reattachResponse != null)
581 Logger.Log(
582 CompileJob.DMApiVersion >= new Version(5, 2, 0)
583 ? LogLevel.Warning
584 : LogLevel.Debug,
585 "DMAPI Interop v{interopVersion} isn't returning the TGS custom commands list. Functionality added in v5.2.0.",
586 CompileJob.DMApiVersion.Semver());
587 }
588 }
589
590 return result;
591 }

608 async Task<BridgeResponse> ProcessBridgeCommand(BridgeParameters parameters, CancellationToken cancellationToken)
609 {
610 var response = new BridgeResponse();
611 switch (parameters.CommandType)
612 {
613 case BridgeCommandType.ChatSend:
614 if (parameters.ChatMessage == null)
615 return BridgeError("Missing chatMessage field!");
616
617 if (parameters.ChatMessage.ChannelIds == null)
618 return BridgeError("Missing channelIds field in chatMessage!");
619
620 if (parameters.ChatMessage.ChannelIds.Any(channelIdString => !UInt64.TryParse(channelIdString, out var _)))
621 return BridgeError("Invalid channelIds in chatMessage!");
622
623 if (parameters.ChatMessage.Text == null)
624 return BridgeError("Missing message field in chatMessage!");
625
626 var anyFailed = false;
627 var parsedChannels = parameters.ChatMessage.ChannelIds.Select(
628 channelString =>
629 {
630 anyFailed |= !UInt64.TryParse(channelString, out var channelId);
631 return channelId;
632 });
633
634 if (anyFailed)
635 return BridgeError("Failed to parse channelIds as U64!");
636
637 chat.QueueMessage(
638 parameters.ChatMessage,
639 parsedChannels);
640 break;
641 case BridgeCommandType.Prime:
642 var oldPrimeTcs = primeTcs;
643 primeTcs = new TaskCompletionSource();
644 oldPrimeTcs.SetResult();
645 break;
646 case BridgeCommandType.Kill:
647 Logger.LogInformation("Bridge requested process termination!");
648 TerminationWasRequested = true;
649 process.Terminate();
650 break;
651 case BridgeCommandType.PortUpdate:
652 lock (synchronizationLock)
653 {
654 if (!parameters.CurrentPort.HasValue)
655 {
657 Logger.LogWarning("DreamDaemon sent new port command without providing it's own!");
658 return BridgeError("Missing stringified port as data parameter!");
659 }
660
661 var currentPort = parameters.CurrentPort.Value;
662 if (!nextPort.HasValue)
663 ReattachInformation.Port = parameters.CurrentPort.Value; // not ready yet, so what we'll do is accept the random port DD opened on for now and change it later when we decide to
664 else
665 {
666 // nextPort is ready, tell DD to switch to that
667 // if it fails it'll kill itself
668 response.NewPort = nextPort.Value;
669 ReattachInformation.Port = nextPort.Value;
670 nextPort = null;
671
672 // we'll also get here from SetPort so complete that task
673 var tmpTcs = portAssignmentTcs;
674 portAssignmentTcs = null;
675 tmpTcs.SetResult(true);
676 }
677
678 portClosedForReboot = false;
679 }
680
681 break;
682 case BridgeCommandType.Startup:
683 apiValidationStatus = ApiValidationStatus.BadValidationRequest;
684 if (parameters.Version == null)
685 return BridgeError("Missing dmApiVersion field!");
686
687 DMApiVersion = parameters.Version;
688 if (DMApiVersion.Major != DMApiConstants.InteropVersion.Major)
689 {
690 apiValidationStatus = ApiValidationStatus.Incompatible;
691 return BridgeError("Incompatible dmApiVersion!");
692 }
693
694 switch (parameters.MinimumSecurityLevel)
695 {
696 case DreamDaemonSecurity.Ultrasafe:
697 apiValidationStatus = ApiValidationStatus.RequiresUltrasafe;
698 break;
699 case DreamDaemonSecurity.Safe:
700 apiValidationStatus = ApiValidationStatus.RequiresSafe;
701 break;
702 case DreamDaemonSecurity.Trusted:
703 apiValidationStatus = ApiValidationStatus.RequiresTrusted;
704 break;
705 case null:
706 return BridgeError("Missing minimumSecurityLevel field!");
707 default:
708 return BridgeError("Invalid minimumSecurityLevel!");
709 }
710
711 Logger.LogTrace("ApiValidationStatus set to {apiValidationStatus}", apiValidationStatus);
712
713 response.RuntimeInformation = new RuntimeInformation(
714 chatTrackingContext,
715 ReattachInformation.Dmb,
716 ReattachInformation.RuntimeInformation.ServerVersion,
717 ReattachInformation.RuntimeInformation.InstanceName,
718 ReattachInformation.RuntimeInformation.SecurityLevel,
719 ReattachInformation.RuntimeInformation.Visibility,
720 ReattachInformation.RuntimeInformation.ServerPort,
721 ReattachInformation.RuntimeInformation.ApiValidateOnly);
722
723 // Load custom commands
724 chatTrackingContext.CustomCommands = parameters.CustomCommands;
725 break;
726 case BridgeCommandType.Reboot:
727 if (ClosePortOnReboot)
728 {
729 chatTrackingContext.Active = false;
730 response.NewPort = 0;
731 portClosedForReboot = true;
732 }
733
734 var oldRebootTcs = rebootTcs;
735 rebootTcs = new TaskCompletionSource();
736 oldRebootTcs.SetResult();
737 break;
738 case BridgeCommandType.Chunk:
739 return await ProcessChunk<BridgeParameters, BridgeResponse>(ProcessBridgeCommand, BridgeError, parameters.Chunk, cancellationToken);
740 case null:
741 return BridgeError("Missing commandType!");
742 default:
743 return BridgeError($"commandType {parameters.CommandType} not supported!");
744 }
745
746 return response;
747 }

769 async Task<CombinedTopicResponse> SendTopicRequest(TopicParameters parameters, CancellationToken cancellationToken)
770 {
771 parameters.AccessIdentifier = ReattachInformation.AccessIdentifier;
772
773 var fullCommandString = GenerateQueryString(parameters, out var json);
774 if (LogTopicRequests)
775 Logger.LogTrace("Topic request: {json}", json);
776 var fullCommandByteCount = Encoding.UTF8.GetByteCount(fullCommandString);
777 var topicPriority = parameters.IsPriority;
778 if (fullCommandByteCount <= DMApiConstants.MaximumTopicRequestLength)
779 return await SendRawTopic(fullCommandString, topicPriority, cancellationToken);
780
781 var interopChunkingVersion = new Version(5, 6, 0);
782 if (ReattachInformation.Dmb.CompileJob.DMApiVersion < interopChunkingVersion)
783 {
784 Logger.LogWarning(
785 "Cannot send topic request as it is exceeds the single request limit of {limitBytes}B ({actualBytes}B) and requires chunking and the current compile job's interop version must be at least {chunkingVersionRequired}!",
786 DMApiConstants.MaximumTopicRequestLength,
787 fullCommandByteCount,
788 interopChunkingVersion);
789 return null;
790 }
791
792 var payloadId = NextPayloadId;
793
794 // AccessIdentifer is just noise in a chunked request
795 parameters.AccessIdentifier = null;
796 GenerateQueryString(parameters, out json);
797
798 // yes, this straight up ignores unicode, precalculating it is useless when we don't
799 // even know if the UTF8 bytes of the url encoded chunk will fit the window until we do said encoding
800 var fullPayloadSize = (uint)json.Length;
801
802 List<string> chunkQueryStrings = null;
803 for (var chunkCount = 2; chunkQueryStrings == null; ++chunkCount)
804 {
805 var standardChunkSize = fullPayloadSize / chunkCount;
806 var bigChunkSize = standardChunkSize + (fullPayloadSize % chunkCount);
807 if (bigChunkSize > DMApiConstants.MaximumTopicRequestLength)
808 continue;
809
810 chunkQueryStrings = new List<string>();
811 for (var i = 0U; i < chunkCount; ++i)
812 {
813 var startIndex = i * standardChunkSize;
814 var subStringLength = Math.Min(
815 fullPayloadSize - startIndex,
816 i == chunkCount - 1
817 ? bigChunkSize
818 : standardChunkSize);
819 var chunkPayload = json.Substring((int)startIndex, (int)subStringLength);
820
821 var chunk = new ChunkData
822 {
823 Payload = chunkPayload,
824 PayloadId = payloadId,
825 SequenceId = i,
826 TotalChunks = (uint)chunkCount,
827 };
828
829 var chunkParameters = new TopicParameters(chunk)
830 {
831 AccessIdentifier = ReattachInformation.AccessIdentifier,
832 };
833
834 var chunkCommandString = GenerateQueryString(chunkParameters, out _);
835 if (Encoding.UTF8.GetByteCount(chunkCommandString) > DMApiConstants.MaximumTopicRequestLength)
836 {
837 // too long when encoded, need more chunks
838 chunkQueryStrings = null;
839 break;
840 }
841
842 chunkQueryStrings.Add(chunkCommandString);
843 }
844 }
845
846 Logger.LogTrace("Chunking topic request ({totalChunks} total)...", chunkQueryStrings.Count);
847
848 CombinedTopicResponse combinedResponse = null;
849 bool LogRequestIssue(bool possiblyFromCompletedRequest)
850 {
851 if (combinedResponse?.InteropResponse == null || combinedResponse.InteropResponse.ErrorMessage != null)
852 {
853 Logger.LogWarning(
854 "Topic request {chunkingStatus} failed!{potentialRequestError}",
855 possiblyFromCompletedRequest ? "final chunk" : "chunking",
856 combinedResponse?.InteropResponse?.ErrorMessage != null
857 ? $" Request error: {combinedResponse.InteropResponse.ErrorMessage}"
858 : String.Empty);
859 return true;
860 }
861
862 return false;
863 }
864
865 foreach (var chunkCommandString in chunkQueryStrings)
866 {
867 combinedResponse = await SendRawTopic(chunkCommandString, topicPriority, cancellationToken);
868 if (LogRequestIssue(chunkCommandString == chunkQueryStrings.Last()))
869 return null;
870 }
871
872 while ((combinedResponse.InteropResponse.MissingChunks?.Count ?? 0) > 0)
873 {
874 Logger.LogWarning("DD is still missing some chunks of topic request P{payloadId}! Sending missing chunks...", payloadId);
875 var lastIndex = combinedResponse.InteropResponse.MissingChunks.Last();
876 foreach (var missingChunkIndex in combinedResponse.InteropResponse.MissingChunks)
877 {
878 var chunkCommandString = chunkQueryStrings[(int)missingChunkIndex];
879 combinedResponse = await SendRawTopic(chunkCommandString, topicPriority, cancellationToken);
880 if (LogRequestIssue(missingChunkIndex == lastIndex))
881 return null;
882 }
883 }
884
885 return combinedResponse;
886 }

912 async Task<CombinedTopicResponse> SendRawTopic(string queryString, bool priority, CancellationToken cancellationToken)
913 {
914 var targetPort = ReattachInformation.Port;
915 var killedOrRebootedTask = Task.WhenAny(Lifetime, OnReboot);
916 global::Byond.TopicSender.TopicResponse byondResponse = null;
917 var firstSend = true;
918
919 const int PrioritySendAttempts = 5;
920 for (var i = PrioritySendAttempts - 1; i >= 0 && (priority || firstSend); --i)
921 try
922 {
923 firstSend = false;
924 if (!killedOrRebootedTask.IsCompleted)
925 byondResponse = await byondTopicSender.SendTopic(
926 new IPEndPoint(IPAddress.Loopback, targetPort),
927 queryString,
928 cancellationToken);
929
930 break;
931 }
932 catch (Exception ex)
933 {
934 Logger.LogWarning(ex, "SendTopic exception!{retryDetails}", priority ? $" {i} attempts remaining." : String.Empty);
935
936 if (priority && i > 0)
937 {
938 var delayTask = asyncDelayer.Delay(TimeSpan.FromSeconds(2), cancellationToken);
939 await Task.WhenAny(killedOrRebootedTask, delayTask);
940 }
941 }
942
943 if (byondResponse == null)
944 {
945 if (priority)
946 if (killedOrRebootedTask.IsCompleted)
947 Logger.LogWarning(
948 "Unable to send priority topic \"{queryString}\" DreamDaemon {stateClearAction}!",
949 queryString,

957 }

966 }

969 Logger.LogWarning(ex, "Invalid interop response: {topicReturnString}", topicReturn);
970 }
971
972 return new CombinedTopicResponse(byondResponse, interopResponse);
973 }
