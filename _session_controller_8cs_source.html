329 public async ValueTask DisposeAsync()
330 {
331 lock (synchronizationLock)
332 {
333 if (disposed)
334 return;
335 disposed = true;
336 }
337
338 Logger.LogTrace("Disposing...");
339
340 // yield then acquire the topic semaphore to prevent new calls from starting
341 await Task.Yield();
342 (await topicSendSemaphore.Lock(CancellationToken.None)).Dispose(); // DCT: None available
343
344 if (!released)
345 {
346 process.Terminate();
347 await process.Lifetime;
348 }
349
350 await process.DisposeAsync();
351 byondLock.Dispose();
352 bridgeRegistration?.Dispose();
353 ReattachInformation.Dmb.Dispose();
354 ReattachInformation.InitialDmb?.Dispose();
355 chatTrackingContext.Dispose();
356 reattachTopicCts.Dispose();
357
358 if (!released)
359 await Lifetime; // finish the async callback
360
361 topicSendSemaphore.Dispose();
362 }

400 public async ValueTask<TopicResponse> SendCommand(TopicParameters parameters, CancellationToken cancellationToken)
401 {
402 ArgumentNullException.ThrowIfNull(parameters);
403
404 if (Lifetime.IsCompleted || disposed)
405 {
406 Logger.LogWarning(
407 "Attempted to send a command to an inactive SessionController: {commandType}",
408 parameters.CommandType);
409 return null;
410 }
411
412 if (!DMApiAvailable)
413 {
414 Logger.LogTrace("Not sending topic request {commandType} to server without/with incompatible DMAPI!", parameters.CommandType);
415 return null;

419 try
420 {
421 var combinedResponse = await SendTopicRequest(parameters, cancellationToken);

423 void LogCombinedResponse()
424 {
425 if (LogTopicRequests && combinedResponse != null)
426 Logger.LogTrace("Topic response: {topicString}", combinedResponse.ByondTopicResponse.StringData ?? "(NO STRING DATA)");
427 }
428
429 LogCombinedResponse();
430
431 if (combinedResponse?.InteropResponse?.Chunk != null)
432 {
433 Logger.LogTrace("Topic response is chunked...");
434
435 ChunkData nextChunk = combinedResponse.InteropResponse.Chunk;

439 (completedResponse, cancellationToken) =>

447 return null;
448 },

450 cancellationToken);
451
452 if (nextRequest != null)

455 combinedResponse = await SendTopicRequest(nextRequest, cancellationToken);
456 LogCombinedResponse();
457 nextChunk = combinedResponse?.InteropResponse?.Chunk;
458 }

460 nextChunk = null;
461 }

463 }

466 }

588 async Task<LaunchResult> GetLaunchResult(
589 IAssemblyInformationProvider assemblyInformationProvider,
590 IAsyncDelayer asyncDelayer,
591 uint? startupTimeout,
592 bool reattached,
593 bool apiValidate)
594 {
595 var startTime = DateTimeOffset.UtcNow;
596 var useBridgeRequestForLaunchResult = !reattached && (apiValidate || DMApiAvailable);
597 var startupTask = useBridgeRequestForLaunchResult
598 ? initialBridgeRequestTcs.Task
599 : process.Startup;
600 var toAwait = Task.WhenAny(startupTask, process.Lifetime);
601
602 if (startupTimeout.HasValue)
603 toAwait = Task.WhenAny(
604 toAwait,
605 asyncDelayer.Delay(
606 TimeSpan.FromSeconds(startupTimeout.Value),
607 CancellationToken.None)); // DCT: None available, task will clean up after delay
608
609 Logger.LogTrace(
610 "Waiting for LaunchResult based on {launchResultCompletionCause}{possibleTimeout}...",
611 useBridgeRequestForLaunchResult ? "initial bridge request" : "process startup",
612 startupTimeout.HasValue ? $" with a timeout of {startupTimeout.Value}s" : String.Empty);
613
614 await toAwait;
615
616 var result = new LaunchResult
617 {
618 ExitCode = process.Lifetime.IsCompleted ? await process.Lifetime : null,
619 StartupTime = startupTask.IsCompleted ? (DateTimeOffset.UtcNow - startTime) : null,
620 };
621
622 Logger.LogTrace("Launch result: {launchResult}", result);
623
624 if (!result.ExitCode.HasValue && reattached && !disposed)
625 {
626 var reattachResponse = await SendCommand(
627 new TopicParameters(
628 assemblyInformationProvider.Version,
629 ReattachInformation.RuntimeInformation.ServerPort),
630 reattachTopicCts.Token);
631
632 if (reattachResponse != null)
633 {
634 if (reattachResponse?.CustomCommands != null)
635 chatTrackingContext.CustomCommands = reattachResponse.CustomCommands;
636 else if (reattachResponse != null)
637 Logger.Log(
638 CompileJob.DMApiVersion >= new Version(5, 2, 0)
639 ? LogLevel.Warning
640 : LogLevel.Debug,
641 "DMAPI Interop v{interopVersion} isn't returning the TGS custom commands list. Functionality added in v5.2.0.",
642 CompileJob.DMApiVersion.Semver());
643 }
644 }
645
646 return result;
647 }

664 async ValueTask<BridgeResponse> ProcessBridgeCommand(BridgeParameters parameters, CancellationToken cancellationToken)
665 {
666 var response = new BridgeResponse();
667 switch (parameters.CommandType)
668 {
669 case BridgeCommandType.ChatSend:
670 if (parameters.ChatMessage == null)
671 return BridgeError("Missing chatMessage field!");
672
673 if (parameters.ChatMessage.ChannelIds == null)
674 return BridgeError("Missing channelIds field in chatMessage!");
675
676 if (parameters.ChatMessage.ChannelIds.Any(channelIdString => !UInt64.TryParse(channelIdString, out var _)))
677 return BridgeError("Invalid channelIds in chatMessage!");
678
679 if (parameters.ChatMessage.Text == null)
680 return BridgeError("Missing message field in chatMessage!");
681
682 var anyFailed = false;
683 var parsedChannels = parameters.ChatMessage.ChannelIds.Select(
684 channelString =>
685 {
686 anyFailed |= !UInt64.TryParse(channelString, out var channelId);
687 return channelId;
688 });
689
690 if (anyFailed)
691 return BridgeError("Failed to parse channelIds as U64!");
692
693 chat.QueueMessage(
694 parameters.ChatMessage,
695 parsedChannels);
696 break;
697 case BridgeCommandType.Prime:
698 Interlocked.Exchange(ref primeTcs, new TaskCompletionSource()).SetResult();
699 break;
700 case BridgeCommandType.Kill:
701 Logger.LogInformation("Bridge requested process termination!");
702 TerminationWasRequested = true;
703 process.Terminate();
704 break;
705 case BridgeCommandType.PortUpdate:
706 lock (synchronizationLock)
707 {
708 if (!parameters.CurrentPort.HasValue)
709 {
711 Logger.LogWarning("DreamDaemon sent new port command without providing it's own!");
712 return BridgeError("Missing stringified port as data parameter!");
713 }
714
715 var currentPort = parameters.CurrentPort.Value;
716 if (!nextPort.HasValue)
717 ReattachInformation.Port = parameters.CurrentPort.Value; // not ready yet, so what we'll do is accept the random port DD opened on for now and change it later when we decide to
718 else
719 {
720 // nextPort is ready, tell DD to switch to that
721 // if it fails it'll kill itself
722 response.NewPort = nextPort.Value;
723 ReattachInformation.Port = nextPort.Value;
724 nextPort = null;
725
726 // we'll also get here from SetPort so complete that task
727 var tmpTcs = portAssignmentTcs;
728 portAssignmentTcs = null;
729 tmpTcs.SetResult(true);
730 }
731
732 portClosedForReboot = false;
733 }
734
735 break;
736 case BridgeCommandType.Startup:
737 apiValidationStatus = ApiValidationStatus.BadValidationRequest;
738 if (parameters.Version == null)
739 return BridgeError("Missing dmApiVersion field!");
740
741 DMApiVersion = parameters.Version;
742 if (DMApiVersion.Major != DMApiConstants.InteropVersion.Major)
743 {
744 apiValidationStatus = ApiValidationStatus.Incompatible;
745 return BridgeError("Incompatible dmApiVersion!");
746 }
747
748 switch (parameters.MinimumSecurityLevel)
749 {
750 case DreamDaemonSecurity.Ultrasafe:
751 apiValidationStatus = ApiValidationStatus.RequiresUltrasafe;
752 break;
753 case DreamDaemonSecurity.Safe:
754 apiValidationStatus = ApiValidationStatus.RequiresSafe;
755 break;
756 case DreamDaemonSecurity.Trusted:
757 apiValidationStatus = ApiValidationStatus.RequiresTrusted;
758 break;
759 case null:
760 return BridgeError("Missing minimumSecurityLevel field!");
761 default:
762 return BridgeError("Invalid minimumSecurityLevel!");
763 }
764
765 Logger.LogTrace("ApiValidationStatus set to {apiValidationStatus}", apiValidationStatus);
766
767 response.RuntimeInformation = new RuntimeInformation(
768 chatTrackingContext,
769 ReattachInformation.Dmb,
770 ReattachInformation.RuntimeInformation.ServerVersion,
771 ReattachInformation.RuntimeInformation.InstanceName,
772 ReattachInformation.RuntimeInformation.SecurityLevel,
773 ReattachInformation.RuntimeInformation.Visibility,
774 ReattachInformation.RuntimeInformation.ServerPort,
775 ReattachInformation.RuntimeInformation.ApiValidateOnly);
776
777 // Load custom commands
778 chatTrackingContext.CustomCommands = parameters.CustomCommands;
779 Interlocked.Exchange(ref startupTcs, new TaskCompletionSource()).SetResult();
780 break;
781 case BridgeCommandType.Reboot:
782 Interlocked.Increment(ref rebootBridgeRequestsProcessing);
783 try
784 {
785 if (ClosePortOnReboot)
786 {
787 chatTrackingContext.Active = false;
788 response.NewPort = 0;
789 portClosedForReboot = true;
790 }
791
792 Interlocked.Exchange(ref rebootTcs, new TaskCompletionSource()).SetResult();
793 await RebootGate.WaitAsync(cancellationToken);
794 }
795 finally
796 {
797 Interlocked.Decrement(ref rebootBridgeRequestsProcessing);
798 }
799
800 break;
801 case BridgeCommandType.Chunk:
802 return await ProcessChunk<BridgeParameters, BridgeResponse>(ProcessBridgeCommand, BridgeError, parameters.Chunk, cancellationToken);
803 case null:
804 return BridgeError("Missing commandType!");
805 default:
806 return BridgeError($"commandType {parameters.CommandType} not supported!");
807 }
808
809 return response;
810 }

832 async ValueTask<CombinedTopicResponse> SendTopicRequest(TopicParameters parameters, CancellationToken cancellationToken)
833 {
834 parameters.AccessIdentifier = ReattachInformation.AccessIdentifier;
835
836 var fullCommandString = GenerateQueryString(parameters, out var json);
837 if (LogTopicRequests)
838 Logger.LogTrace("Topic request: {json}", json);
839 var fullCommandByteCount = Encoding.UTF8.GetByteCount(fullCommandString);
840 var topicPriority = parameters.IsPriority;
841 if (fullCommandByteCount <= DMApiConstants.MaximumTopicRequestLength)
842 return await SendRawTopic(fullCommandString, topicPriority, cancellationToken);
843
844 var interopChunkingVersion = new Version(5, 6, 0);
845 if (ReattachInformation.Dmb.CompileJob.DMApiVersion < interopChunkingVersion)
846 {
847 Logger.LogWarning(
848 "Cannot send topic request as it is exceeds the single request limit of {limitBytes}B ({actualBytes}B) and requires chunking and the current compile job's interop version must be at least {chunkingVersionRequired}!",
849 DMApiConstants.MaximumTopicRequestLength,
850 fullCommandByteCount,
851 interopChunkingVersion);
852 return null;
853 }
854
855 var payloadId = NextPayloadId;
856
857 // AccessIdentifer is just noise in a chunked request
858 parameters.AccessIdentifier = null;
859 GenerateQueryString(parameters, out json);
860
861 // yes, this straight up ignores unicode, precalculating it is useless when we don't
862 // even know if the UTF8 bytes of the url encoded chunk will fit the window until we do said encoding
863 var fullPayloadSize = (uint)json.Length;
864
865 List<string> chunkQueryStrings = null;
866 for (var chunkCount = 2; chunkQueryStrings == null; ++chunkCount)
867 {
868 var standardChunkSize = fullPayloadSize / chunkCount;
869 var bigChunkSize = standardChunkSize + (fullPayloadSize % chunkCount);
870 if (bigChunkSize > DMApiConstants.MaximumTopicRequestLength)
871 continue;
872
873 chunkQueryStrings = new List<string>();
874 for (var i = 0U; i < chunkCount; ++i)
875 {
876 var startIndex = i * standardChunkSize;
877 var subStringLength = Math.Min(
878 fullPayloadSize - startIndex,
879 i == chunkCount - 1
880 ? bigChunkSize
881 : standardChunkSize);
882 var chunkPayload = json.Substring((int)startIndex, (int)subStringLength);
883
884 var chunk = new ChunkData
885 {
886 Payload = chunkPayload,
887 PayloadId = payloadId,
888 SequenceId = i,
889 TotalChunks = (uint)chunkCount,
890 };
891
892 var chunkParameters = new TopicParameters(chunk)
893 {
894 AccessIdentifier = ReattachInformation.AccessIdentifier,
895 };
896
897 var chunkCommandString = GenerateQueryString(chunkParameters, out _);
898 if (Encoding.UTF8.GetByteCount(chunkCommandString) > DMApiConstants.MaximumTopicRequestLength)
899 {
900 // too long when encoded, need more chunks
901 chunkQueryStrings = null;
902 break;
903 }
904
905 chunkQueryStrings.Add(chunkCommandString);
906 }
907 }
908
909 Logger.LogTrace("Chunking topic request ({totalChunks} total)...", chunkQueryStrings.Count);
910
911 CombinedTopicResponse combinedResponse = null;
912 bool LogRequestIssue(bool possiblyFromCompletedRequest)
913 {
914 if (combinedResponse?.InteropResponse == null || combinedResponse.InteropResponse.ErrorMessage != null)
915 {
916 Logger.LogWarning(
917 "Topic request {chunkingStatus} failed!{potentialRequestError}",
918 possiblyFromCompletedRequest ? "final chunk" : "chunking",
919 combinedResponse?.InteropResponse?.ErrorMessage != null
920 ? $" Request error: {combinedResponse.InteropResponse.ErrorMessage}"
921 : String.Empty);
922 return true;
923 }
924
925 return false;
926 }
927
928 foreach (var chunkCommandString in chunkQueryStrings)
929 {
930 combinedResponse = await SendRawTopic(chunkCommandString, topicPriority, cancellationToken);
931 if (LogRequestIssue(chunkCommandString == chunkQueryStrings.Last()))
932 return null;
933 }
934
935 while ((combinedResponse.InteropResponse.MissingChunks?.Count ?? 0) > 0)
936 {
937 Logger.LogWarning("DD is still missing some chunks of topic request P{payloadId}! Sending missing chunks...", payloadId);
938 var lastIndex = combinedResponse.InteropResponse.MissingChunks.Last();
939 foreach (var missingChunkIndex in combinedResponse.InteropResponse.MissingChunks)
940 {
941 var chunkCommandString = chunkQueryStrings[(int)missingChunkIndex];
942 combinedResponse = await SendRawTopic(chunkCommandString, topicPriority, cancellationToken);
943 if (LogRequestIssue(missingChunkIndex == lastIndex))
944 return null;
945 }
946 }
947
948 return combinedResponse;
949 }

975 async ValueTask<CombinedTopicResponse> SendRawTopic(string queryString, bool priority, CancellationToken cancellationToken)
976 {
977 if (disposed)
978 {
979 Logger.LogWarning(
980 "Attempted to send a topic on a disposed SessionController");
981 return null;
982 }
983
984 var targetPort = ReattachInformation.Port;
985 var killedOrRebootedTask = Task.WhenAny(Lifetime, OnReboot);
986 global::Byond.TopicSender.TopicResponse byondResponse = null;
987 var firstSend = true;
988
989 using (await topicSendSemaphore.Lock(cancellationToken))
990 {
991 const int PrioritySendAttempts = 5;
992 var endpoint = new IPEndPoint(IPAddress.Loopback, targetPort);
993 for (var i = PrioritySendAttempts - 1; i >= 0 && (priority || firstSend); --i)
994 try
995 {
996 firstSend = false;
997 if (!killedOrRebootedTask.IsCompleted)
998 {
999 Logger.LogTrace("Begin topic request");
1000 byondResponse = await byondTopicSender.SendTopic(
1001 endpoint,
1002 queryString,
1003 cancellationToken);
1004
1005 Logger.LogTrace("End topic request");
1006 }
1007
1008 break;
1009 }
1010 catch (Exception ex) when (!cancellationToken.IsCancellationRequested)
1011 {
1012 Logger.LogWarning(ex, "SendTopic exception!{retryDetails}", priority ? $" {i} attempts remaining." : String.Empty);
1013
1014 if (priority && i > 0)
1015 {
1016 var delayTask = asyncDelayer.Delay(TimeSpan.FromSeconds(2), cancellationToken);
1017 await Task.WhenAny(killedOrRebootedTask, delayTask);
1018 }
1019 }
1020 }
1021
1022 if (byondResponse == null)
1023 {
1024 if (priority)
1025 if (killedOrRebootedTask.IsCompleted)
1026 Logger.LogWarning(
1027 "Unable to send priority topic \"{queryString}\" DreamDaemon {stateClearAction}!",
1028 queryString,
1029 Lifetime.IsCompleted ? "process ended" : "rebooted");
1030 else
1031 Logger.LogError(
1032 "Unable to send priority topic \"{queryString}\"!",
1033 queryString);
1034
1035 return null;
1036 }
1037
1038 var topicReturn = byondResponse.StringData;
1039
1040 TopicResponse interopResponse = null;
1041 if (topicReturn != null)
1042 try
1043 {
1044 interopResponse = JsonConvert.DeserializeObject<TopicResponse>(topicReturn, DMApiConstants.SerializerSettings);
1045 }
1046 catch (Exception ex)
1047 {
1048 Logger.LogWarning(ex, "Invalid interop response: {topicReturnString}", topicReturn);
1049 }
1050
1051 return new CombinedTopicResponse(byondResponse, interopResponse);
1052 }
