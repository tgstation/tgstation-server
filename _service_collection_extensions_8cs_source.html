34 public static IServiceCollection UseStandardConfig<TConfig>(this IServiceCollection serviceCollection, IConfiguration configuration) where TConfig : class
35 {
36 if (serviceCollection == null)

41 const string SectionFieldName = nameof(GeneralConfiguration.Section);
42
43 var configType = typeof(TConfig);
44 var sectionField = configType.GetField(SectionFieldName);
45 if (sectionField == null)
46 throw new InvalidOperationException(String.Format(CultureInfo.InvariantCulture, "{0} has no {1} field!", configType, SectionFieldName));

52 var sectionName = (string)sectionField.GetValue(null);
